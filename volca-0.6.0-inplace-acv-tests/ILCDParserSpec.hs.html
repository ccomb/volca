   16     describe "ILCD Process Parser" $ do
   17         it "parses classification from classificationInformation" $
   18             M.lookup "ILCDCategories" (classOf ilcdProcessWithClassification)
   19                 `shouldBe` Just "End-of-life treatment/Material recycling"
   20 
   21         it "parses multiple classification systems" $ do
   22             let cls = classOf ilcdProcessWithTwoClassifications
   23             M.lookup "ILCDCategories" cls `shouldBe` Just "Energy/Electricity"
   24             M.lookup "EcoSpold" cls `shouldBe` Just "Supply"
   25 
   26         it "produces empty classifications when none present" $
   27             classOf ilcdProcessNoClassification `shouldBe` M.empty
