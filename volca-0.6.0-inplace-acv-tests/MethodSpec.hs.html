  240     describe "LCIA Score Computation" $ do
  241         describe "computeLCIAScore" $ do
  242             it "computes score as sum of inventory * CF for mapped flows" $ do
  243                 let co2Uuid = UUID.fromWords 1 2 3 4
  244                     ch4Uuid = UUID.fromWords 5 6 7 8
  245                     -- Inventory: CO2 = 10 kg, CH4 = 2 kg
  246                     inventory = M.fromList [(co2Uuid, 10.0), (ch4Uuid, 2.0)]
  247                     -- Method: CO2 CF = 1.0, CH4 CF = 28.0
  248                     co2CF = MethodCF co2Uuid "Carbon dioxide" Output 1.0 Nothing Nothing
  249                     ch4CF = MethodCF ch4Uuid "Methane" Output 28.0 Nothing Nothing
  250                     co2Flow = mkTestFlow co2Uuid "Carbon dioxide"
  251                     ch4Flow = mkTestFlow ch4Uuid "Methane"
  252                     mappings = [ (co2CF, Just (co2Flow, ByUUID))
  253                                , (ch4CF, Just (ch4Flow, ByUUID))
  254                                ]
  255                 -- Score = 10*1 + 2*28 = 10 + 56 = 66
  256                 computeLCIAScore inventory mappings `shouldBe` 66.0
  257 
  258             it "ignores unmapped flows" $ do
  259                 let co2Uuid = UUID.fromWords 1 2 3 4
  260                     ch4Uuid = UUID.fromWords 5 6 7 8
  261                     inventory = M.fromList [(co2Uuid, 10.0), (ch4Uuid, 2.0)]
  262                     co2CF = MethodCF co2Uuid "Carbon dioxide" Output 1.0 Nothing Nothing
  263                     ch4CF = MethodCF ch4Uuid "Methane" Output 28.0 Nothing Nothing
  264                     co2Flow = mkTestFlow co2Uuid "Carbon dioxide"
  265                     mappings = [ (co2CF, Just (co2Flow, ByUUID))
  266                                , (ch4CF, Nothing)  -- CH4 not mapped
  267                                ]
  268                 -- Score = 10*1 = 10 (CH4 ignored because not mapped)
  269                 computeLCIAScore inventory mappings `shouldBe` 10.0
  270 
  271             it "returns 0 for flows not in inventory" $ do

  278                     n2oFlow = mkTestFlow n2oUuid "Dinitrogen monoxide"
  279                     mappings = [ (n2oCF, Just (n2oFlow, ByName)) ]
