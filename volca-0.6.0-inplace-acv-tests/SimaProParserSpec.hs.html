  250 -- Helper: find technosphere input by name
  251 findInput :: Activity -> T.Text -> Maybe Exchange
  252 -- never enteredfindInput act query = case [e | e@TechnosphereExchange{} <- exchanges act
  253                               , techIsInput e, not (techIsReference e)] of
  254     exs -> case filter (matchesName query) exs of
  255         (e:_) -> Just e
  256         _     -> Nothing
  257   where
  258     matchesName _ _ = True  -- We check by position since we can't easily get flow names here
  259 
  260 -- Helper: get all tech input amounts
  261 techInputAmounts :: Activity -> [Double]
  262 -- entered 3 timestechInputAmounts act = [techAmount e | e@TechnosphereExchange{} <- exchanges act
  263                                      , techIsInput e, not (techIsReference e)]

  267 -- entered oncerefProductAmount act = case [techAmount e | e@TechnosphereExchange{} <- exchanges act

  272 spec :: Spec
  273 -- entered oncespec = do
  274     describe "SimaPro expression evaluator" $ do
  275         it "evaluates numeric literals" $ do
  276             evaluate M.empty "42" `shouldBe` Right 42.0
  277             evaluate M.empty "3.14" `shouldBe` Right 3.14
  278 
  279         it "evaluates arithmetic" $ do
  280             evaluate M.empty "2+3" `shouldBe` Right 5.0
  281             evaluate M.empty "10-4" `shouldBe` Right 6.0
  282             evaluate M.empty "3*4" `shouldBe` Right 12.0
  283             evaluate M.empty "10/4" `shouldBe` Right 2.5
  284 
  285         it "evaluates parenthesized expressions" $ do
  286             evaluate M.empty "(2+3)*4" `shouldBe` Right 20.0
  287             evaluate M.empty "2*(3+4)" `shouldBe` Right 14.0
  288 
  289         it "evaluates variables" $ do
  290             let env = M.fromList [("Qm", 20.53), ("Qb", 1.0)]
  291             evaluate env "Qm" `shouldBe` Right 20.53
  292             evaluate env "Qb" `shouldBe` Right 1.0
  293 
  294         it "evaluates complex expressions with variables" $ do
  295             let env = M.fromList [("Qb", 1.0), ("DMb", 0.82), ("Qm", 20.53), ("DMm", 0.118)]
  296             -- Butter allocation formula: (Qb*DMb/(Qb*DMb+Qm*DMm))*100
  297             let result = evaluate env "(Qb*DMb/(Qb*DMb+Qm*DMm))*100"
  298             case result of
  299                 Right v -> v `shouldSatisfy` (\x -> abs (x - 25.29) < 0.1)
  300                 Left e  -> expectationFailure $ "Evaluation failed: " ++ e
  301 
  302         it "evaluates yield correction chains" $ do
  303             let env = M.fromList [("weight_kg", 0.25), ("yield1", 0.95), ("yield2", 0.90)]
  304             let result = evaluate env "weight_kg/yield1/yield2"
  305             case result of
  306                 Right v -> v `shouldSatisfy` (\x -> abs (x - 0.2924) < 0.001)
  307                 Left e  -> expectationFailure $ "Evaluation failed: " ++ e
  308 
  309         it "evaluates power operator" $ do
  310             evaluate M.empty "2^3" `shouldBe` Right 8.0
  311             evaluate M.empty "3^2" `shouldBe` Right 9.0
  312 
  313         it "evaluates unary minus" $ do
  314             evaluate M.empty "-5" `shouldBe` Right (-5.0)
