   99 isInfixOf :: String -> String -> Bool
  100 -- entered 5 timesisInfixOf needle haystack = any (needle `isPrefixOf'`) (tails' haystack)
  101   where
  102     isPrefixOf' [] _       = True
  103     isPrefixOf' _ []       = False
  104     isPrefixOf' (x:xs) (y:ys) = x == y && isPrefixOf' xs ys
  105     tails' []     = [[]]
  106     tails' s@(_:rest) = s : tails' rest
