   22 -- entered 19 timesmkFlow fid name cat msub = Flow
   23     { flowId             = fid
   24     , flowName           = name

   26     , flowSubcompartment = msub

   28     , flowType           = Biosphere

   34 mkCF :: Text -> Maybe Text -> Double -> MethodCF
   35 -- entered 4 timesmkCF name mCas val = MethodCF
   36     { mcfFlowRef     = nil
   37     , mcfFlowName    = name
   38     , mcfDirection   = Output
   39     , mcfValue       = val
   40     , mcfCompartment = Nothing
   41     , mcfCAS         = mCas
   42     , mcfUnit        = "kg"
   43     }

   46 -- entered 3 timesmkCFComp name medium subcomp val = (mkCF name Nothing val)
