  264 -- | Build a minimal TreeExport from a list of (id, parentId, name) and edges (from,to)
  265 mkTreeExport :: [(Text, Maybe Text, Text)] -> [(Text, Text)] -> TreeExport
  266 -- entered 5 timesmkTreeExport nodeSpecs edgeSpecs =
  267     let mkNode (nid, parent, name) = (nid, ExportNode
  268             { enId = nid, enName = name, enDescription = []
  269             , enLocation = "", enUnit = "kg"
  270             , enNodeType = ActivityNode, enDepth = 0
  271             , enLoopTarget = Nothing, enParentId = parent
  272             , enChildrenCount = 0, enCompartment = Nothing
  273             })
  274         nodes = M.fromList (map mkNode nodeSpecs)
  275         dummyFlow = FlowInfo nil "" ""
  276         mkEdge (f, t) = TreeEdge f t dummyFlow 1.0 "kg" TechnosphereEdge
  277         edges = map mkEdge edgeSpecs
  278         meta  = TreeMetadata "" 1 (M.size nodes) 0 0 0
  279     in TreeExport meta nodes edges
