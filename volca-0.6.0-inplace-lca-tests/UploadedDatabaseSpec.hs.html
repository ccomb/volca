   80     -- -----------------------------------------------------------------------
   81     -- parseMetaToml

   83     describe "parseMetaToml" $ do
   84         it "parses minimal meta without description" $ do

   86             parseMetaToml toml `shouldBe` Just UploadMeta
   87                 { umVersion = 1, umDisplayName = "My DB"

   90 
   91         it "parses meta with description" $ do
   92             let toml = "version = 1\ndisplayName = \"DB\"\ndescription = \"Desc\"\nformat = \"simapro\"\ndataPath = \"d\"\n"
   93             fmap umDescription (parseMetaToml toml) `shouldBe` Just (Just "Desc")
   94 
   95         it "returns Nothing for missing required field" $
   96             parseMetaToml "displayName = \"x\"\n" `shouldBe` Nothing
   97 
   98         it "ignores comment lines" $ do

  100             fmap umFormat (parseMetaToml toml) `shouldBe` Just ILCDProcess

  102     -- -----------------------------------------------------------------------

  105     describe "formatMetaToml / parseMetaToml roundtrip" $ do
  106         it "round-trips a meta without description" $

  108 
  109         it "round-trips a meta with description" $ do

  112 
  113         it "round-trips all database formats" $

  117                 ) [EcoSpold2, EcoSpold1, SimaProCSV, ILCDProcess, UnknownFormat]
  118 
  119         it "round-trips a path with spaces" $ do
  120             let meta = baseMeta { umDataPath = "my data/sub dir" }
