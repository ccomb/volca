    7 -- | Parse ILCD flow XML files to extract baseName, compartment, and CAS.
    8 --
    9 -- ILCD method packages contain a @flows/@ directory with one XML per flow.
   10 -- Each XML has: UUID, baseName, elementaryFlowCategorization (compartment),
   11 -- and optionally a CASNumber. This module builds a lookup map from UUID
   12 -- to enrichment data, used to annotate MethodCFs during method parsing.
   13 --
   14 -- Parsed flow data is cached to disk (zstd-compressed) to avoid re-parsing
   15 -- 94K+ XML files on every startup (~30s → <1s).
   16 module Method.FlowResolver
   17     ( ILCDFlowInfo(..)
   18     , resolveFlowDirectory
   19     , parseFlowDirectory
   20     , parseFlowXML
   21     ) where

   47 -- | Enrichment data extracted from an ILCD flow XML
   48 data ILCDFlowInfo = ILCDFlowInfo
   49     { ilcdBaseName       :: !Text
   50     , ilcdCompartment    :: !(Maybe Compartment)
   51     , ilcdCAS            :: !(Maybe Text)  -- normalized CAS
   52     , ilcdSynonyms       :: ![Text]        -- from <common:synonyms xml:lang="en">, semicolon-split
   53     , ilcdFlowType       :: !Text          -- "Elementary flow" / "Product flow" / "Waste flow"
   54     , ilcdFlowPropertyRef :: !(Maybe UUID) -- reference flow property UUID (for unit resolution)
   55     } deriving (Show, Generic, NFData, Store)

   57 -- | Given a method directory (containing lciamethods/), find the sibling flows/ directory.
   58 -- ILCD packages have structure: ILCD/{lciamethods/, flows/, ...}
   59 resolveFlowDirectory :: FilePath -> IO (Maybe FilePath)
   60 resolveFlowDirectory methodDir = do
   61     -- methodDir is the directory containing method XMLs (e.g., .../ILCD/lciamethods)
   62     -- flows/ is a sibling: .../ILCD/flows
   63     let parent = takeParentDir methodDir
   64         flowsDir = parent </> "flows"
   65     exists <- doesDirectoryExist flowsDir
   66     return $ if exists then Just flowsDir else Nothing
   67   where
   68     -- Go up one directory level
   69     takeParentDir = reverse . dropWhile (/= '/') . dropWhile (== '/') . reverse

   71 -- | Parse all flow XMLs in a directory, returning UUID → ILCDFlowInfo map.
   72 --   Uses a zstd-compressed cache to skip re-parsing on subsequent startups.
   73 --   Falls back to worker-based parallel parsing on cache miss.
   74 parseFlowDirectory :: FilePath -> IO (M.Map UUID ILCDFlowInfo)
   75 parseFlowDirectory dir = do
   76     let cacheFile = flowCacheFile dir
   77     cached <- loadFlowCache cacheFile dir
   78     case cached of
   79         Just info -> return info
   80         Nothing -> do
   81             info <- parseFlowDirectoryFresh dir
   82             saveFlowCache cacheFile info
   83             return info
   84 
   85 -- | Parse all flow XMLs from scratch using worker-based parallelism
   86 parseFlowDirectoryFresh :: FilePath -> IO (M.Map UUID ILCDFlowInfo)
   87 parseFlowDirectoryFresh dir = do
   88     files <- listDirectory dir
   89     let xmlFiles = [dir </> f | f <- files, map toLower (takeExtension f) == ".xml"]
   90     numWorkers <- getNumCapabilities
   91     let workers = distributeFiles numWorkers xmlFiles
   92     workerResults <- mapConcurrently parseWorker workers
   93     return $! M.unions workerResults
   94   where
   95     parseWorker paths = do
   96         results <- mapM parseOneFile paths
   97         return $! M.fromList [(uuid, info) | Just (uuid, info) <- results]
   98     parseOneFile path = do
   99         bytes <- BS.readFile path
  100         return $ parseFlowXML bytes
  101 
  102 -- | Cache file path for a flows directory
  103 flowCacheFile :: FilePath -> FilePath
  104 flowCacheFile dir = dir </> ".volca.flows.cache.zst"
  105 
  106 -- | Load flow info from cache if valid (file exists and is newer than directory)
  107 loadFlowCache :: FilePath -> FilePath -> IO (Maybe (M.Map UUID ILCDFlowInfo))
  108 loadFlowCache cacheFile dir = do
  109     exists <- doesFileExist cacheFile
  110     if not exists then return Nothing
  111     else catch
  112         (do cacheTime <- getModificationTime cacheFile
  113             dirTime <- getModificationTime dir
  114             if cacheTime < dirTime
  115                 then do
  116                     reportProgress Info "[FLOWS-CACHE] Stale, reparsing"
  117                     removeFile cacheFile
  118                     return Nothing
  119                 else do
  120                     compressed <- BS.readFile cacheFile
  121                     case Zstd.decompress compressed of
  122                         Zstd.Decompress raw -> do
  123                             let !info = decodeEx raw
  124                             result <- evaluate (force info)
  125                             reportProgress Info $ "[FLOWS-CACHE] Loaded " <> show (M.size result) <> " flow definitions from cache"
  126                             return (Just result)
  127                         _ -> do
  128                             reportProgress Warning "[FLOWS-CACHE] Decompression failed, reparsing"
  129                             removeFile cacheFile
  130                             return Nothing)
  131         (\(_ :: SomeException) -> return Nothing)
  132 
  133 -- | Save parsed flow info to cache
  134 saveFlowCache :: FilePath -> M.Map UUID ILCDFlowInfo -> IO ()
  135 saveFlowCache cacheFile info = catch
  136     (do let serialized = encode info
  137             compressed = Zstd.compress 1 serialized
  138         BS.writeFile cacheFile compressed
  139         reportProgress Info $ "[FLOWS-CACHE] Saved " <> show (M.size info) <> " flow definitions to cache")
  140     (\(_ :: SomeException) -> return ())

  142 -- | SAX parse state for flow XML
  143 data FlowParseState = FlowParseState
  144     { fpsUUID        :: !Text
  145     , fpsBaseName    :: !Text
  146     , fpsCAS         :: !Text
  147     , fpsCategories  :: ![Text]  -- category level="0", "1", "2" in order
  148     , fpsPath        :: ![BS.ByteString]
  149     , fpsTextAccum   :: ![BS.ByteString]
  150     , fpsInBaseName  :: !Bool
  151     , fpsSynonyms    :: ![Text]  -- accumulated synonym names
  152     , fpsInSynonyms  :: !Bool    -- inside <common:synonyms>
  153     , fpsLangIsEn    :: !Bool    -- current synonyms element has xml:lang="en"
  154     , fpsFlowType    :: !Text    -- <typeOfDataSet> text
  155     , fpsRefFlowPropIdx :: !Int  -- referenceToReferenceFlowProperty index
  156     , fpsCurrentFPIdx   :: !Int  -- current flowProperty dataSetInternalID
  157     , fpsFlowPropRef    :: !(Maybe UUID) -- resolved flow property UUID
  158     , fpsInFlowProperty :: !Bool -- inside <flowProperty>
  159     }
  160 
  161 initialFlowState :: FlowParseState
  162 initialFlowState = FlowParseState "" "" "" [] [] [] False [] False False "" 0 (-1) Nothing False
  163 
  164 -- | Parse a single ILCD flow XML from bytes
  165 parseFlowXML :: BS.ByteString -> Maybe (UUID, ILCDFlowInfo)
  166 parseFlowXML bytes =
  167     case X.fold openTag attr endOpen txt closeTag cdata initialFlowState bytes of
  168         Left _  -> Nothing
  169         Right s -> buildFlowInfo s
  170   where
  171     openTag s tag =
  172         let !inSyn = isElement tag "synonyms" || fpsInSynonyms s
  173             !inBase = isElement tag "baseName" || fpsInBaseName s
  174             !inFP = isElement tag "flowProperty" || fpsInFlowProperty s
  175         in s { fpsPath = tag : fpsPath s, fpsTextAccum = []
  176              , fpsInBaseName = inBase, fpsInSynonyms = inSyn
  177              , fpsLangIsEn = if isElement tag "synonyms" then False else fpsLangIsEn s
  178              , fpsInFlowProperty = inFP }
  179 
  180     attr s name value
  181         | fpsInSynonyms s && isElement name "lang" && bsToText value == "en" =
  182             s { fpsLangIsEn = True }
  183         | isElement name "dataSetInternalID" && fpsInFlowProperty s =
  184             s { fpsCurrentFPIdx = case TR.decimal (bsToText value) of
  185                     Right (n, _) -> n
  186                     Left _       -> -1 }
  187         | isElement name "refObjectId" && fpsInFlowProperty s
  188             && fpsCurrentFPIdx s == fpsRefFlowPropIdx s =
  189             s { fpsFlowPropRef = UUID.fromText (bsToText value) }
  190         | isElement name "level" = s
  191         | otherwise = s
  192 
  193     endOpen s _ = s
  194 
  195     txt s content =
  196         let trimmed = BS.dropWhile (== 32) $ BS.dropWhileEnd (== 32) content
  197         in if BS.null trimmed then s
  198            else s { fpsTextAccum = trimmed : fpsTextAccum s }
  199 
  200     closeTag s tag
  201         | isElement tag "UUID" && T.null (fpsUUID s) =
  202             s { fpsPath = dropPath s, fpsUUID = accum s, fpsTextAccum = [] }
  203         | isElement tag "baseName" =
  204             s { fpsPath = dropPath s, fpsBaseName = accum s
  205               , fpsTextAccum = [], fpsInBaseName = False }
  206         | isElement tag "CASNumber" =
  207             s { fpsPath = dropPath s, fpsCAS = accum s, fpsTextAccum = [] }
  208         | isElement tag "typeOfDataSet" && T.null (fpsFlowType s) =
  209             s { fpsPath = dropPath s, fpsFlowType = accum s, fpsTextAccum = [] }
  210         | isElement tag "referenceToReferenceFlowProperty" =
  211             s { fpsPath = dropPath s
  212               , fpsRefFlowPropIdx = case TR.decimal (accum s) of
  213                     Right (n, _) -> n
  214                     Left _       -> 0
  215               , fpsTextAccum = [] }
  216         | isElement tag "flowProperty" =
  217             s { fpsPath = dropPath s, fpsTextAccum = [], fpsInFlowProperty = False }
  218         | isElement tag "synonyms" =
  219             let !syns = if fpsLangIsEn s
  220                         then filter (not . T.null) $ map T.strip $ T.splitOn ";" (accum s)
  221                         else []
  222             in s { fpsPath = dropPath s, fpsTextAccum = []
  223                  , fpsInSynonyms = False, fpsLangIsEn = False
  224                  , fpsSynonyms = fpsSynonyms s ++ syns }
  225         | isElement tag "category" =
  226             s { fpsPath = dropPath s
  227               , fpsCategories = fpsCategories s ++ [accum s]
  228               , fpsTextAccum = [] }
  229         | otherwise =
  230             s { fpsPath = dropPath s, fpsTextAccum = [] }
  231 
  232     cdata = txt
  233 
  234     dropPath s = case fpsPath s of
  235         (_:rest) -> rest
  236         []       -> []
  237 
  238     accum s = T.strip $ T.concat $ reverse $ map bsToText (fpsTextAccum s)
  239 
  240     buildFlowInfo s = do
  241         uuid <- UUID.fromText (fpsUUID s)
  242         let baseName = fpsBaseName s
  243         if T.null baseName then Nothing
  244         else Just (uuid, ILCDFlowInfo
  245             { ilcdBaseName       = baseName
  246             , ilcdCompartment    = parseCompartment (fpsCategories s)
  247             , ilcdCAS            = if T.null (fpsCAS s) then Nothing
  248                                    else Just (normalizeCAS (fpsCAS s))
  249             , ilcdSynonyms       = fpsSynonyms s
  250             , ilcdFlowType       = fpsFlowType s
  251             , ilcdFlowPropertyRef = fpsFlowPropRef s
  252             })
  253 
  254 -- | Parse compartment from ILCD category levels.
  255 -- Level 0: "Emissions" or "Resources"
  256 -- Level 1: "Emissions to air" → medium = "air"
  257 -- Level 2: "Emissions to air, indoor" → subcompartment = "indoor"
  258 parseCompartment :: [Text] -> Maybe Compartment
  259 parseCompartment [] = Nothing
  260 parseCompartment cats =
  261     let -- Level 1 typically contains medium: "Emissions to air", "Emissions to water", etc.
  262         medium = case drop 1 cats of
  263             (lvl1:_) -> extractMedium lvl1
  264             []       -> extractMedium (head cats)  -- cats is non-empty ([] case handled above)
  265         -- Level 2 is the subcompartment
  266         subcomp = case drop 2 cats of
  267             (lvl2:_) -> extractSubcompartment lvl2
  268             []       -> ""
  269     in if T.null medium then Nothing
  270        else Just (Compartment medium subcomp "")
  271   where
  272     -- Extract medium from "Emissions to air" → "air", "Resources from ground" → "natural resource"
  273     extractMedium txt
  274         | "Emissions to air" `T.isPrefixOf` txt      = "air"
  275         | "Emissions to water" `T.isPrefixOf` txt    = "water"
  276         | "Emissions to soil" `T.isPrefixOf` txt     = "soil"
  277         | "Resources" `T.isPrefixOf` txt             = "natural resource"
  278         | "Emissions to fresh water" `T.isPrefixOf` txt = "water"
  279         | "Emissions to sea water" `T.isPrefixOf` txt  = "water"
  280         | otherwise = T.toLower $ T.strip txt
  281 
  282     -- Extract subcompartment: take the last part after comma or "to X,"
  283     extractSubcompartment txt
  284         | "Emissions to air, " `T.isPrefixOf` txt =
  285             T.strip $ T.drop (T.length "Emissions to air, ") txt
  286         | "Emissions to water, " `T.isPrefixOf` txt =
  287             T.strip $ T.drop (T.length "Emissions to water, ") txt
  288         | "Emissions to soil, " `T.isPrefixOf` txt =
  289             T.strip $ T.drop (T.length "Emissions to soil, ") txt
  290         | "Emissions to " `T.isPrefixOf` txt =
  291             T.strip $ T.drop (T.length "Emissions to ") txt
  292         | "Resources " `T.isPrefixOf` txt =
  293             T.strip $ T.drop (T.length "Resources ") txt
  294         | otherwise = T.toLower $ T.strip txt
