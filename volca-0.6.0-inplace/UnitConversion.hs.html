   98 -- | Convert amount from one unit to another.
   99 -- Returns Nothing if units are incompatible or unknown.
  100 convertUnit :: UnitConfig -> Text -> Text -> Double -> Maybe Double
  101 -- entered 94 timesconvertUnit cfg fromUnit toUnit amount = do
  102     UnitDef dimFrom factorFrom <- lookupUnitDef cfg fromUnit
  103     UnitDef dimTo factorTo     <- lookupUnitDef cfg toUnit
  104     if dimFrom == dimTo && factorTo /= 0
  105         then Just (amount * factorFrom / factorTo)
  106         else Nothing

  108 -- | Parse a dimension expression like "mass*length/time" into an exponent vector.
  109 parseDimension :: [Text] -> Text -> Either Text Dimension
  110 -- entered 4742 timesparseDimension dimOrder expr
  111     | T.null (T.strip expr) = Left "Empty dimension expression"
  112     | otherwise = do
  113         let baseVec = replicate (length dimOrder) 0
  114             (numExpr, denExpr) = case T.breakOn "/" expr of
  115                 (num, rest) | T.null rest -> (num, "")
  116                             | otherwise   -> (num, T.drop 1 rest)
  117             numParts = filter (not . T.null) $ map T.strip $ T.splitOn "*" numExpr
  118             denParts = if T.null denExpr
  119                        then []
  120                        else concatMap (filter (not . T.null) . map T.strip . T.splitOn "*")
  121                                       (T.splitOn "/" denExpr)
  122         vec1 <- foldlM (addExp dimOrder 1) baseVec numParts
  123         foldlM (addExp dimOrder (-1)) vec1 denParts
  124   where
  125     addExp :: [Text] -> Int -> Dimension -> Text -> Either Text Dimension
  126     addExp order delta vec dimName =
  127         case elemIndex dimName order of
  128             Just idx -> Right $ modifyAt idx (+ delta) vec
  129             Nothing  -> Left $ "Unknown dimension: " <> dimName
  130                             <> " (valid: " <> T.intercalate ", " order <> ")"
  131 
  132     modifyAt :: Int -> (Int -> Int) -> [Int] -> [Int]
  133     modifyAt idx f xs = zipWith (\i x -> if i == idx then f x else x) [0..] xs
  134 
  135     foldlM :: (b -> a -> Either e b) -> b -> [a] -> Either e b
  136     foldlM _ acc []     = Right acc
  137     foldlM f acc (x:xs) = case f acc x of
  138         Left err   -> Left err
  139         Right acc' -> foldlM f acc' xs

  141 -- | Build a UnitConfig from CSV content (three columns: name, dimension, factor).
  142 buildFromCSV :: BL.ByteString -> Either Text UnitConfig
  143 -- entered 27 timesbuildFromCSV csvData =
  144     case decode HasHeader csvData of
  145         Left err -> Left $ "CSV parse error: " <> T.pack err
  146         Right rows ->
  147             let dimOrder = defaultDimensionOrder
  148             in buildFromRows dimOrder (V.toList (rows :: V.Vector (Text, Text, Double)))
  149   where
  150     buildFromRows dimOrder rows = do
  151         pairs <- mapM (parseRow dimOrder) rows
  152         let units = M.fromList pairs
  153         Right UnitConfig
  154             { ucDimensionOrder = dimOrder
  155             , ucUnits = units
  156             , ucOriginalKeys = M.mapWithKey (\k _ -> k) units
  157             }
  158 
  159     parseRow dimOrder (name, dimExpr, factor) = do
  160         dim <- parseDimension dimOrder dimExpr
  161         Right (normalizeUnit name, UnitDef dim factor)

  202 -- | Warn if a unit is unknown (deduplicated).
  203 warnIfUnknownUnit :: UnknownUnitTracker -> Text -> IO ()
  204 -- never enteredwarnIfUnknownUnit tracker unit = do
  205     let normalized = normalizeUnit unit
  206     unless (isKnownUnit (uutConfig tracker) unit || T.null normalized) $ do
  207         seen <- readIORef (uutSeen tracker)
  208         unless (S.member normalized seen) $ do
  209             modifyIORef' (uutSeen tracker) (S.insert normalized)
  210             reportProgress Info $ "[WARNING] Unknown unit: \"" <> T.unpack unit
  211                                <> "\" - add to [[units]] CSV"
