    4 -- | Parser for CSV-based LCIA method files.
    5 --
    6 -- A single CSV file encodes multiple impact categories as columns.
    7 -- Supports two header layouts (both after optional @# methodology:@ comment):
    8 --
    9 -- __2-row__ (category defaults to name):
   10 --
   11 -- > ;;global warming (GWP100);acidification;...      ← names
   12 -- > ;;kg CO2 eq.;kg SO2 eq.;...                      ← units
   13 -- > substance;compartment;;;...                       ← label row
   14 --
   15 -- __3-row__ (explicit categories):
   16 --
   17 -- > ;;Climate change;Acidification;...                ← categories
   18 -- > ;;global warming (GWP100);acidification;...       ← names
   19 -- > ;;kg CO2 eq.;kg SO2 eq.;...                       ← units
   20 -- > substance;compartment;;;...                        ← label row
   21 --
   22 -- One CSV file → multiple 'Method' values (one per column).
   23 module Method.ParserCSV
   24     ( parseMethodCSVBytes
   25     ) where

   42 -- | Pure parser, exported for testing.
   43 parseMethodCSVBytes :: BS.ByteString -> Either String [Method]
   44 -- entered 9 timesparseMethodCSVBytes bytes =
   45     let allLines  = map stripCR $ BC.lines (stripBOM bytes)
   46         (comments, dataLines) = span (\l -> BC.isPrefixOf "#" l || BS.null l) allLines
   47         methodology = extractMethodology comments
   48     in case splitHeaderFromData dataLines of
   49         Nothing -> Left "CSV method file needs at least 3 header rows (names, units, column labels)"
   50         Just (cats, names, units, rows, delim) ->
   51             Right [ mkMethod methodology catName impactName unit colIdx rows delim
   52                   | (colIdx, catName, impactName, unit) <- zip4 [0..] cats names units
   53                   , not (T.null impactName)
   54                   ]

   56 -- | Find the label row (first column starts with "substance") and split header
   57 -- rows from data rows.  Returns (categories, names, units, dataRows, delimiter).
   58 -- 2-row layout → categories = names.  3-row layout → distinct.
   59 splitHeaderFromData :: [BS.ByteString] -> Maybe ([Text], [Text], [Text], [BS.ByteString], Word8)
   60 -- entered 9 timessplitHeaderFromData dataLines =
   61     case break isLabelRow dataLines of
   62         (headerRows, _labelRow : rows) ->
   63             let nonBlank = filter (not . BS.null) headerRows
   64             in case nonBlank of
   65                 [nameRow, unitRow] ->
   66                     let !delim = detectDelimiter nameRow
   67                         names = drop 2 $ splitRow delim nameRow
   68                         units = drop 2 $ splitRow delim unitRow
   69                     in Just (names, names, units, rows, delim)  -- categories = names
   70                 [catRow, nameRow, unitRow] ->
   71                     let !delim = detectDelimiter catRow
   72                         cats  = drop 2 $ splitRow delim catRow
   73                         names = drop 2 $ splitRow delim nameRow
   74                         units = drop 2 $ splitRow delim unitRow
   75                     in Just (cats, names, units, rows, delim)
   76                 _ -> Nothing
   77         _ -> Nothing

   79 -- | Detect the label row: first cell is a variation of "substance".
   80 isLabelRow :: BS.ByteString -> Bool
   81 -- entered 30 timesisLabelRow line =
   82     let first = T.toCaseFold . T.strip . head' . splitRow (detectDelimiter line) $ line
   83     in first == "substance"
   84   where
   85     head' (x:_) = x
   86     head' []    = ""

   88 -- | Build a single Method from one column of the CSV.
   89 mkMethod :: Maybe Text -> Text -> Text -> Text -> Int -> [BS.ByteString] -> Word8 -> Method
   90 -- entered 5 timesmkMethod methodology catName impactName unit colIdx rows delim =
   91     let !ns       = csvMethodNamespace
   92         !mId      = UUID5.generateNamed ns (bsKey $ "method:" <> impactName)
   93         !factors  = [ MethodCF
   94                         { mcfFlowRef     = UUID5.generateNamed ns (bsKey $ sub <> "::" <> comp)
   95                         , mcfFlowName    = sub
   96                         , mcfDirection   = directionFromCompartment comp
   97                         , mcfValue       = v
   98                         , mcfCompartment = parseCSVCompartment comp
   99                         , mcfCAS         = Nothing
  100                         , mcfUnit        = unit
  101                         }
  102                     | row <- rows
  103                     , let cells = splitRow delim row
  104                     , length cells > colIdx + 2
  105                     , let !sub  = T.strip (cells !! 0)
  106                           !comp = T.strip (cells !! 1)
  107                           !raw  = T.strip (cells !! (colIdx + 2))
  108                     , not (T.null sub)
  109                     , not (T.null raw)
  110                     , Just v <- [parseDouble raw]
  111                     ]
  112     in Method
  113         { methodId          = mId
  114         , methodName        = impactName
  115         , methodDescription = Nothing
  116         , methodUnit        = if T.null unit then "unknown" else unit
  117         , methodCategory    = catName
  118         , methodMethodology = methodology
  119         , methodFactors     = factors
  120         }

  132 -- | Parse compartment from CSV compartment column.
  133 -- Format: "Emissions to air", "Emissions to fresh water", "Resources", etc.
  134 parseCSVCompartment :: Text -> Maybe Compartment
  135 -- never enteredparseCSVCompartment comp
  136     | T.null comp = Nothing
  137     | "air" `T.isInfixOf` lc = Just (Compartment "air" "" "")
  138     | "water" `T.isInfixOf` lc = Just (Compartment "water" "" "")
  139     | "soil" `T.isInfixOf` lc = Just (Compartment "soil" "" "")
  140     | "resource" `T.isInfixOf` lc = Just (Compartment "natural resource" "" "")
  141     | otherwise = Nothing
  142   where lc = T.toCaseFold comp

  144 -- | Auto-detect delimiter: semicolon if the line contains ';', else comma.
  145 detectDelimiter :: BS.ByteString -> Word8
  146 -- entered 38 timesdetectDelimiter line
  147     | BC.elem ';' line = 0x3B  -- ';'
  148     | otherwise        = 0x2C  -- ','

  156 -- entered onceextractMethodology = go

  158     go [] = Nothing
