  240 -- | Build the final Method from parsed state, enriching CFs with flow XML data
  241 buildMethod :: M.Map UUID ILCDFlowInfo -> ParseState -> Either String Method
  242 -- entered 5 timesbuildMethod flowInfo state = do
  243     when (T.null (psMethodId state)) $
  244         Left "Missing method UUID"
  245     when (T.null (psMethodName state)) $
  246         Left "Missing method name"
  247 
  248     methodUUID <- case UUID.fromText (psMethodId state) of
  249         Just u -> Right u
  250         Nothing -> Left $ "Invalid method UUID: " ++ T.unpack (psMethodId state)
  251 
  252     let factors = reverse (psFactors state)
  253         enrichedFactors = map (enrichCF flowInfo) factors
  254 
  255     Right $ Method
  256         { methodId = methodUUID
  257         , methodName = psMethodName state
  258         , methodDescription = if T.null (psDescription state)
  259                                 then Nothing
  260                                 else Just (psDescription state)
  261         , methodUnit = if T.null (psUnit state) then "unknown" else psUnit state
  262         , methodCategory = if T.null (psCategory state) then "unknown" else psCategory state
  263         , methodMethodology = if T.null (psMethodology state)
  264                                 then Nothing
  265                                 else Just (psMethodology state)
  266         , methodFactors = enrichedFactors
  267         }

  312 -- | Extract flow name from shortDescription
  313 -- Format: "methane (fossil) (Mass, kg, Emissions to ...)"
  314 -- The flow name may contain parentheses (e.g. "methane (fossil)"),

  316 extractFlowName :: Text -> Text
  317 -- entered 2 timesextractFlowName txt =
  318     let markers = ["(Mass", "(Volume", "(Area", "(Energy", "(Length", "(Number of items"]
  319         findMarker [] = Nothing
  320         findMarker (m:ms) = case T.breakOn m txt of
  321             (before, rest) | not (T.null rest) -> Just before
  322             _ -> findMarker ms
  323     in T.strip $ case findMarker markers of
  324         Just before -> before

  327 -- | Extract compartment info from shortDescription as fallback
  328 -- Format: "... Emissions to air, non-urban ..."
  329 extractCompartmentFromDesc :: Text -> Maybe Compartment
  330 -- entered 2 timesextractCompartmentFromDesc txt
  331     | "Emissions to air" `T.isInfixOf` txt =
  332         Just (Compartment "air" (extractSubFromDesc "Emissions to air" txt) "")
  333     | "Emissions to water" `T.isInfixOf` txt =
  334         Just (Compartment "water" (extractSubFromDesc "Emissions to water" txt) "")
  335     | "Emissions to soil" `T.isInfixOf` txt =
  336         Just (Compartment "soil" (extractSubFromDesc "Emissions to soil" txt) "")
  337     | "Resources" `T.isInfixOf` txt =
  338         Just (Compartment "natural resource" "" "")
  339     | otherwise = Nothing
  340   where
  341     extractSubFromDesc prefix t =
  342         let (_, after) = T.breakOn prefix t
  343             rest = T.drop (T.length prefix) after
  344             -- Rest might be ", non-urban air or from high stacks)"
  345             cleaned = T.dropWhile (\c -> c == ',' || c == ' ') rest
  346             sub = T.takeWhile (/= ')') cleaned
  347         in T.strip sub
