   21 {- | Build complete database with pre-computed sparse matrices
   22 
   23 SIGN CONVENTION:
   24 - Technosphere triplets are stored as POSITIVE values (input coefficients per unit output)
   25 - Matrix.hs negates these when constructing (I-A) system matrix for solving
   26 - The biosphere matrix stores ALL flows as POSITIVE (emissions AND resource extractions)
   27 - Resource extractions represent "outputs" from nature into the technosphere (positive like emissions)
   28 - This follows Ecoinvent convention where B matrix contains positive values for all environmental flows
   29 
   30 Matrix Construction:
   31 - Accepts a Map with (UUID, UUID) keys and converts to Vector internally
   32 - Builds sparse triplets for technosphere (A) and biosphere (B) matrices
   33 - Normalizes exchanges by NET reference product amounts (gross output - internal consumption)
   34 - SELF-LOOPS (internal consumption) are EXCLUDED from matrix triplets but affect normalization
   35 - Example: Electricity market with 1.0 kWh output and 0.012 kWh internal loss
   36   * Normalization factor: 1.0 - 0.012 = 0.988 kWh (net output)
   37   * All inputs divided by 0.988, giving ~1.2% increase in coefficients
   38   * Self-loop NOT exported as matrix entry (matches Ecoinvent convention)
   39 - Solver constructs (I-A) by adding identity and negating technosphere triplets
   40 -}

  286 -- | Build indexes with ProcessIds
  287 buildIndexesWithProcessIds :: V.Vector Activity -> V.Vector (UUID, UUID) -> FlowDB -> Indexes
  288 -- entered 35 timesbuildIndexesWithProcessIds activityVec processIdTable flowDB =

  289     let

  293         activities = V.toList activityVec

  294         activityPairs = zip activityUUIDs activities

  296         -- Build indexes using activity UUIDs

  297         nameIdx =

  316                 [(activityUnit activity, [uuid]) | (uuid, activity) <- activityPairs]

  318         flowCatIdx =

  319             M.fromListWith

  320                 (++)

  321                 [(flowCategory flow, [flowId]) | (flowId, flow) <- M.toList flowDB]

  334             { idxByName = nameIdx

  342 -- | Build ProductIndex for product-based lookups
  343 -- Used for: (1) SimaPro upstream link resolution, (2) future product search
  344 -- Maps product flow UUIDs and names to the activities that produce them
  345 buildProductIndex :: V.Vector Activity -> V.Vector (UUID, UUID) -> FlowDB -> ProductIndex
  346 -- entered 35 timesbuildProductIndex activities processIdTable flowDb =

  347     let -- Build list of (ProcessId, productUUID, productName, location) for each activity

  348         entries =

  349             [ (pid, prodUUID, prodName, actLoc)

  350             | (pid, (_, prodUUID)) <- zip [0..] (V.toList processIdTable)

  351             , let act = activities V.! fromIntegral pid

  352             , let actLoc = activityLocation act

  353             , Just flow <- [M.lookup prodUUID flowDb]

  354             , let prodName = T.toLower (flowName flow)

  355             ]

  356     in ProductIndex

  357         { piByUUID = M.fromList [(prodUUID, pid) | (pid, prodUUID, _, _) <- entries]

  358         , piByName = M.fromListWith (++) [(name, [pid]) | (pid, _, name, _) <- entries]

  359         , piByLocation = M.fromListWith (++) [(loc, [pid]) | (pid, _, _, loc) <- entries, not (T.null loc)]

  360         }

  368         -- Filter by name: split into words, each word must match name OR location

  373                     -- Check if all words match (each word must be in name OR location)
