  162 -- | Helper function to parse OutputFormat from string
  163 parseOutputFormat :: String -> Maybe OutputFormat
  164 -- never enteredparseOutputFormat s = case map toLower s of
  165   "json" -> Just JSON
  166   "csv" -> Just CSV
  167   "table" -> Just Table
  168   "pretty" -> Just Pretty
  169   _ -> Nothing
  170   where
  171     toLower c = if c >= 'A' && c <= 'Z' then toEnum (fromEnum c + 32) else c
