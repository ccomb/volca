  205 -- | Check if path contains files with a given extension
  206 hasFilesWithExt :: String -> FilePath -> IO Bool
  207 -- never enteredhasFilesWithExt ext path = do
  208     isFile <- doesFileExist path
  209     if isFile
  210         then pure $ map toLower (takeExtension path) == ext
  211         else do
  212             isDir <- doesDirectoryExist path
  213             if isDir
  214                 then do
  215                     files <- listDirectory path
  216                     pure $ any (\f -> map toLower (takeExtension f) == ext) files
  217                 else pure False

  223 -- | Name-based flow search: substring match on flow name and synonyms
  224 nameSearcher :: SearchHandle
  225 -- entered oncenameSearcher = SearchHandle
  226     { shName     = "name-searcher"
  227     , shBackend  = Builtin
  228     , shPriority = 0
  229     , shSearch   = \db query -> do
  230         let queryLower = T.toLower (sqText query)
  231             flows = M.elems (dbFlows db)
  232             matches = [ (f, score) | f <- flows
  233                       , let nameLower = T.toLower (flowName f)
  234                             synMatches = any (\syns -> any (T.isInfixOf queryLower . T.toLower) (S.toList syns))
  235                                             (M.elems (flowSynonyms f))
  236                       , T.isInfixOf queryLower nameLower || synMatches
  237                       , let score = if queryLower == nameLower then 1.0
  238                                     else if T.isPrefixOf queryLower nameLower then 0.9
  239                                     else 0.7
  240                       ]
  241             limited = take (sqLimit query) matches
  242         pure [ SearchResult (flowId f) (flowName f) s M.empty | (f, s) <- limited ]
  243     }

  245 -- | CAS number search: exact CAS match on dbFlowsByCAS index
  246 casSearcher :: SearchHandle
  247 -- entered oncecasSearcher = SearchHandle
  248     { shName     = "cas-searcher"
  249     , shBackend  = Builtin
  250     , shPriority = 10
  251     , shSearch   = \db query -> do
  252         let cas = sqText query
  253         pure $ case M.lookup cas (dbFlowsByCAS db) of
  254             Just flows ->
  255                 [ SearchResult (flowId f) (flowName f) 1.0 (M.singleton "cas" cas) | f <- take (sqLimit query) flows ]
  256             Nothing -> []
  257     }

  259 -- | Run search across all plugin searchers, merge and deduplicate by UUID
  260 searchWithPlugins :: [SearchHandle] -> Database -> SearchQuery -> IO [SearchResult]
  261 -- entered oncesearchWithPlugins searchers db query = do
  262     allResults <- concat <$> mapM (\s -> shSearch s db query) searchers
  263     -- Deduplicate: keep highest-scoring result per UUID
  264     let byUUID = M.fromListWith (\a b -> if srScore a >= srScore b then a else b)
  265                     [(srId r, r) | r <- allResults]
  266     pure $ M.elems byUUID

  272 -- | Hotspot analyzer: for each method, return top-N flows by contribution to the score.
  273 -- This answers "why is my product 2.5 kg CO2e?" by showing which flows contribute most.
  274 hotspotAnalyzer :: AnalyzeHandle
  275 -- entered oncehotspotAnalyzer = AnalyzeHandle
  276     { ahName    = "hotspot"
  277     , ahBackend = Builtin
  278     , ahAnalyze = \ctx -> do
  279         let db = acDatabase ctx
  280             inv = acInventory ctx
  281             methods = acMethods ctx
  282             mapCtx = Mapping.buildMapContext db
  283             topN = 20
  284         results <- mapM (analyzeMethod mapCtx inv topN) methods
  285         pure $ toJSON results
  286     }
  287   where
  288     analyzeMethod mapCtx inv topN method = do
  289         mappings <- Mapping.mapMethodFlows defaultMappers mapCtx method
  290         let contributions = mapMaybe (flowContribution inv) mappings
  291             sorted = take topN $ sortOn (\(_,_,c) -> negate (abs c)) contributions
  292             total = sum [qty * mcfValue cf | (cf, Just (flow, _)) <- mappings
  293                                             , Just qty <- [M.lookup (flowId flow) inv]]
  294         pure $ toJSON $ M.fromList
  295             [ ("method" :: String, toJSON (methodName method))
  296             , ("unit", toJSON (methodUnit method))
  297             , ("totalScore", toJSON total)
  298             , ("topFlows", toJSON [ toJSON $ M.fromList
  299                 [ ("flowName" :: String, toJSON (flowName f))
  300                 , ("cfName", toJSON (mcfFlowName cf))
  301                 , ("contribution", toJSON contrib)
  302                 , ("percent", toJSON (if total /= 0 then contrib / total * 100 else 0 :: Double))
  303                 ]
  304                 | (cf, f, contrib) <- sorted
  305                 ])
  306             ]
  307 
  308     flowContribution inv (cf, Just (flow, _))
  309         | Just qty <- M.lookup (flowId flow) inv =
  310             Just (cf, flow, qty * mcfValue cf)
  311     flowContribution _ _ = Nothing
