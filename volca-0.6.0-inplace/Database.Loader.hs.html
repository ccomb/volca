 1073 {- | Fix activity links using cross-database lookup.
 1074 
 1075 For each unlinked technosphere input (where activityLinkId is nil),
 1076 search across other loaded databases to find a matching supplier.
 1077 
 1078 Matching criteria:
 1079 - Product name must match (exact, synonym, or fuzzy)
 1080 - Units must be compatible
 1081 - Location scoring with hierarchy fallback
 1082 
 1083 Cross-database links are stored in CrossDBLinkingStats.cdlLinks for use
 1084 in chained inventory solving. The exchanges are NOT modified - they
 1085 remain "unlinked" from the perspective of the internal matrix, but the
 1086 CrossDBLinks provide the information needed to resolve them at solve time.
 1087 -}
 1088 fixActivityLinksWithCrossDB
 1089     :: [IndexedDatabase]            -- ^ Pre-built indexes from other databases
 1090     -> SynonymDB                    -- ^ Synonym database
 1091     -> UC.UnitConfig                -- ^ Unit configuration
 1092     -> M.Map T.Text [T.Text]        -- ^ Location hierarchy (code → parent codes)
 1093     -> SimpleDatabase               -- ^ Database to fix
 1094     -> IO (SimpleDatabase, CrossDBLinkingStats)
 1095 -- never enteredfixActivityLinksWithCrossDB indexedDbs synonymDB unitConfig locationHier db = do
 1096     -- Count unlinked exchanges before
 1097     let unlinkedBefore = countUnlinkedExchanges db
 1098         !totalInputs = countTotalTechInputs db
 1099 
 1100     -- If no unlinked exchanges, skip
 1101     if unlinkedBefore == 0
 1102         then do
 1103             reportProgress Info "No unlinked exchanges to resolve via cross-DB linking"
 1104             return (db, emptyCrossDBLinkingStats { cdlTotalInputs = totalInputs })
 1105         else do
 1106             reportProgress Info $
 1107                 printf "Cross-database linking: %d unlinked exchanges, searching %d database(s)..."
 1108                     unlinkedBefore (length indexedDbs)
 1109 
 1110             -- Report index stats
 1111             forM_ indexedDbs $ \idb ->
 1112                 reportProgress Info $
 1113                     printf "  - %s: %d products indexed"
 1114                         (T.unpack (idbName idb))
 1115                         (M.size (idbByProductName idb))
 1116 
 1117             -- Build the linking context with pre-built indexes
 1118             let linkingCtx = LinkingContext
 1119                     { lcIndexedDatabases  = indexedDbs
 1120                     , lcSynonymDB         = synonymDB
 1121                     , lcUnitConfig        = unitConfig
 1122                     , lcThreshold         = defaultLinkingThreshold
 1123                     , lcLocationHierarchy = if M.null locationHier then locationHierarchy else locationHier
 1124                     }
 1125 
 1126             -- Process all activities to find cross-DB links
 1127             reportProgress Info "Finding cross-database suppliers..."
 1128             let stats = findAllCrossDBLinks
 1129                     linkingCtx (sdbFlows db) (sdbUnits db) (sdbActivities db)
 1130 
 1131             -- Report statistics
 1132             let !stats' = stats { cdlTotalInputs = totalInputs }
 1133             reportCrossDBLinkingStats (M.size (sdbActivities db)) stats'
 1134 
 1135             -- Return the original database unchanged, along with the cross-DB links
 1136             -- The links will be stored in the Database.dbCrossDBLinks field later
 1137             return (db, stats')

 1250 -- | Report cross-database linking statistics
 1251 reportCrossDBLinkingStats :: Int -> CrossDBLinkingStats -> IO ()
 1252 -- never enteredreportCrossDBLinkingStats nActivities stats = do
 1253     let !nInputs = cdlTotalInputs stats
 1254         !nCrossDB = crossDBLinksCount stats
 1255         !nUnresolved = unresolvedCount stats
 1256         !nInternal = max 0 (nInputs - nCrossDB - nUnresolved)
 1257         !nResolved = nInternal + nCrossDB
 1258 
 1259     -- Summary line (skip "0/0" for databases without technosphere input tracking)
 1260     if nInputs > 0
 1261         then do
 1262             let !completeness = 100.0 * fromIntegral nResolved / fromIntegral nInputs :: Double
 1263             reportProgress Info $
 1264                 printf "Supply chain: %.1f%% complete (%d/%d inputs resolved), %d activities"
 1265                     completeness nResolved nInputs nActivities
 1266             reportProgress Info $
 1267                 printf "  Internal: %d, Cross-DB: %d, Unresolved: %d" nInternal nCrossDB nUnresolved
 1268         else
 1269             reportProgress Info $
 1270                 printf "Supply chain: %d activities (no technosphere inputs)" nActivities
 1271 
 1272     -- Per-database breakdown
 1273     forM_ (M.toList (crossDBBySource stats)) $ \(srcDb, count) ->
 1274         reportProgress Info $
 1275             printf "  - %s: %d links" (T.unpack srcDb) count
 1276 
 1277     -- Missing suppliers
 1278     let !missing = sortOn (\(_, (cnt, _)) -> Down cnt) $ M.toList (cdlUnresolvedProducts stats)
 1279     when (not (null missing)) $ do
 1280         reportProgress Warning $
 1281             printf "Missing suppliers: %d products unresolved" (length missing)
 1282         forM_ (take 20 missing) $ \(name, (cnt, blocker)) ->
 1283             reportProgress Warning $
 1284                 printf "  - %s (%d activities) — %s" (T.unpack name) cnt (showBlocker blocker)
 1285         when (length missing > 20) $
 1286             reportProgress Warning $
 1287                 printf "  ... and %d more" (length missing - 20)
 1288 
 1289     -- Unknown units
 1290     let !unknowns = S.toList (cdlUnknownUnits stats)
 1291     when (not (null unknowns)) $
 1292         reportProgress Warning $
 1293             printf "Unknown units: %s" (T.unpack $ T.intercalate ", " unknowns)
 1294 
 1295     -- Location fallbacks (deduplicated)
 1296     let !uniqueFallbacks = deduplicateFallbacks (cdlLocationFallbacks stats)
 1297         !nFallbacks = length uniqueFallbacks
 1298     when (nFallbacks > 0) $ do
 1299         reportProgress Info $
 1300             printf "Location fallbacks: %d unique products matched with different location" nFallbacks
 1301         forM_ uniqueFallbacks $ \(prod, requested, actual) ->
 1302             reportProgress Info $
 1303                 printf "  - %s: %s → %s" (T.unpack prod) (T.unpack requested) (T.unpack actual)
