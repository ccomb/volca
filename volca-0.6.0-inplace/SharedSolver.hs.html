    4 Module      : SharedSolver
    5 Description : Shared PETSc solver with lazy factorization and thread synchronization
    6 
    7 Implements lazy factorization: the PETSc KSP is not built at startup, but
    8 on the first solve request. This eliminates ~3s × N databases of startup
    9 latency while keeping sub-second solves after the first query.
   10 -}

   18

   24 import Control.Exception (catch, SomeException)
   25 import Data.Text (Text)
   26 import Progress

   30 -- | Shared solver with lazy factorization and thread synchronization.
   31 --   Factorization happens on first solve, not at startup.
   32 data SharedSolver = SharedSolver
   33     { -- never enteredsolverLock :: MVar ()                                -- ^ Serialize access to solver
   34     , -- never enteredsolverFactorizationVar :: MVar (Maybe MatrixFactorization) -- ^ Lazy: Nothing until first solve
   35     , -- never enteredsolverTechTriples :: [(Int, Int, Double)]             -- ^ Technosphere matrix data
   36     , -- never enteredsolverActivityCount :: Int                            -- ^ Number of activities
   37     , -- never enteredsolverDbName :: Text                                  -- ^ Database name (for PETSc cache key)
   38     }
   39 
   40 -- | Create a shared solver. No factorization happens here — deferred to first solve.
   41 createSharedSolver :: Text -> [(Int, Int, Double)] -> Int -> IO SharedSolver
   42 -- never enteredcreateSharedSolver dbName techTriples activityCount = do
   43     reportProgress Info $ "Creating shared solver for '" ++ show dbName ++ "' (factorization deferred)"
   44     lock <- newMVar ()
   45     factVar <- newMVar Nothing
   46     return $ SharedSolver lock factVar techTriples activityCount dbName
   47 
   48 -- | Solve using shared solver. On first call, triggers lazy factorization.
   49 solveWithSharedSolver :: SharedSolver -> Vector -> IO Vector
   50 -- never enteredsolveWithSharedSolver solver demandVector = do
   51     withMVar (solverLock solver) $ \_ -> do
   52         maybeFact <- readMVar (solverFactorizationVar solver)
   53         case maybeFact of
   54             Just factorization -> do
   55                 reportProgress Solver "Using cached factorization for sub-second solve"
   56                 solveSparseLinearSystemWithFactorization factorization demandVector
   57             Nothing -> do
   58                 -- First solve: factorize, cache, then solve
   59                 reportProgress Info $ "First solve for '" ++ show (solverDbName solver) ++ "' — computing factorization..."
   60                 factResult <- catch
   61                     (do factorization <- precomputeMatrixFactorization
   62                             (solverDbName solver)
   63                             (solverTechTriples solver)
   64                             (solverActivityCount solver)
   65                         -- Store for subsequent solves (replace the MVar contents)
   66                         _ <- modifyMVar (solverFactorizationVar solver) $ \_ -> return (Just factorization, ())
   67                         reportProgress Info "Factorization complete — solving"
   68                         result <- solveSparseLinearSystemWithFactorization factorization demandVector
   69                         return (Just result))
   70                     (\e -> do
   71                         reportProgress Solver $ "Factorization failed: " ++ show (e :: SomeException) ++ " — using fallback solver"
   72                         return Nothing)
   73                 case factResult of
   74                     Just result -> return result
   75                     Nothing ->
   76                         solveSparseLinearSystem (solverTechTriples solver) (solverActivityCount solver) demandVector
