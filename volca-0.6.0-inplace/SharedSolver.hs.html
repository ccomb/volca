    3 {-|
    4 Module      : SharedSolver
    5 Description : Shared MUMPS solver with lazy factorization and thread synchronization
    6 
    7 Implements lazy factorization: the MUMPS solver is not built at startup, but
    8 on the first solve request. This eliminates ~3s x N databases of startup
    9 latency while keeping sub-second solves after the first query.
   10 -}
   11 
   12 module SharedSolver (
   13     -- * Shared solver types
   14     SharedSolver,
   15 
   16     -- * Solver management
   17     createSharedSolver,
   18 
   19     -- * Concurrent solving
   20     solveWithSharedSolver,
   21     getFactorization
   22 ) where
   23 
   24 import Control.Concurrent.MVar (MVar, newMVar, withMVar, readMVar, modifyMVar)
   25 import Control.Exception (catch, SomeException)
   26 import Data.Text (Text)
   27 import Progress
   28 import Types
   29 import Matrix (Vector, solveSparseLinearSystem, solveSparseLinearSystemWithFactorization, precomputeMatrixFactorization)
   30 
   31 -- | Shared solver with lazy factorization and thread synchronization.
   32 --   Factorization happens on first solve, not at startup.
   33 data SharedSolver = SharedSolver
   34     { solverLock :: MVar ()                                -- ^ Serialize access to solver
   35     , solverFactorizationVar :: MVar (Maybe MatrixFactorization) -- ^ Lazy: Nothing until first solve
   36     , solverTechTriples :: [(Int, Int, Double)]             -- ^ Technosphere matrix data
   37     , solverActivityCount :: Int                            -- ^ Number of activities
   38     , solverDbName :: Text                                  -- ^ Database name (for solver cache key)
   39     }
   40 
   41 -- | Create a shared solver. No factorization happens here — deferred to first solve.
   42 createSharedSolver :: Text -> [(Int, Int, Double)] -> Int -> IO SharedSolver
   43 createSharedSolver dbName techTriples activityCount = do
   44     reportProgress Info $ "Creating shared solver for '" ++ show dbName ++ "' (factorization deferred)"
   45     lock <- newMVar ()
   46     factVar <- newMVar Nothing
   47     return $ SharedSolver lock factVar techTriples activityCount dbName
   48 
   49 -- | Solve using shared solver. On first call, triggers lazy factorization.
   50 solveWithSharedSolver :: SharedSolver -> Vector -> IO Vector
   51 solveWithSharedSolver solver demandVector = do
   52     withMVar (solverLock solver) $ \_ -> do
   53         maybeFact <- readMVar (solverFactorizationVar solver)
   54         case maybeFact of
   55             Just factorization -> do
   56                 reportProgress Solver "Using cached factorization for sub-second solve"
   57                 solveSparseLinearSystemWithFactorization factorization demandVector
   58             Nothing -> do
   59                 -- First solve: factorize, cache, then solve
   60                 reportProgress Info $ "First solve for '" ++ show (solverDbName solver) ++ "' — computing factorization..."
   61                 factResult <- catch
   62                     (do factorization <- precomputeMatrixFactorization
   63                             (solverDbName solver)
   64                             (solverTechTriples solver)
   65                             (solverActivityCount solver)
   66                         -- Store for subsequent solves (replace the MVar contents)
   67                         _ <- modifyMVar (solverFactorizationVar solver) $ \_ -> return (Just factorization, ())
   68                         reportProgress Info "Factorization complete — solving"
   69                         result <- solveSparseLinearSystemWithFactorization factorization demandVector
   70                         return (Just result))
   71                     (\e -> do
   72                         reportProgress Solver $ "Factorization failed: " ++ show (e :: SomeException) ++ " — using fallback solver"
   73                         return Nothing)
   74                 case factResult of
   75                     Just result -> return result
   76                     Nothing ->
   77                         solveSparseLinearSystem (solverTechTriples solver) (solverActivityCount solver) demandVector
   78 
   79 -- | Read the cached factorization without solving. Returns Nothing until the first solve.
   80 getFactorization :: SharedSolver -> IO (Maybe MatrixFactorization)
   81 getFactorization solver = readMVar (solverFactorizationVar solver)
