   39 -- | Pure parser for SimaPro method CSV bytes.
   40 parseSimaProMethodCSVBytes :: BS.ByteString -> Either String MethodCollection
   41 -- entered 10 timesparseSimaProMethodCSVBytes raw =
   42     let !utf8 = ensureUtf8 raw
   43         lns = BS8.lines utf8
   44         cfg = parseConfig lns
   45         methodName' = parseMethodName cfg lns
   46         result = foldl' (step cfg methodName') initState lns
   47     in Right (finalize result)

   58 data ParseState = ParseState

   67     , -- entered 30 timespsDcUnit     :: !Text               -- current damage category unit
   68     , -- entered 60 timespsDcImpacts  :: ![(Text, Double)]   -- current damage category impacts (reversed)
   69     -- Normalization/Weighting
   70     , -- entered 20 timespsNWsets     :: ![NormWeightSet]     -- completed NW sets (reversed)
   71     , -- entered 10 timespsNWname     :: !Text               -- current NW set name
   72     , -- entered 50 timespsNormMap    :: !(M.Map Text Double) -- current normalization factors
   73     , -- entered 40 timespsWeightMap  :: !(M.Map Text Double) -- current weighting factors

   76 data Phase
   77     = PhHeader          -- reading {key: value} header lines
   78     | PhMethodMeta      -- reading method-level metadata (Name, Version, etc.)
   79     | PhExpectCategory  -- expecting "Impact category" or "Damage category" etc.
   80     | PhExpectCatLine   -- expecting the "Name;Unit" line after "Impact category"
   81     | PhExpectSubst     -- expecting blank or "Substances" marker
   82     | PhReadingCFs      -- reading substance/CF rows
   83     -- Damage categories
   84     | PhExpectDcLine    -- expecting "Name;Unit" after "Damage category"
   85     | PhExpectDcImpacts -- expecting "Impact categories" marker
   86     | PhReadingDcImpacts -- reading impact category rows in damage category
   87     -- Normalization/Weighting
   88     | PhExpectNWname    -- expecting NW set name line
   89     | PhExpectNWsection -- expecting "Normalization" or "Weighting" or next section
   90     | PhReadingNorm     -- reading normalization rows
   91     | PhReadingWeight   -- reading weighting rows
   92 
   93 initState :: ParseState
   94 -- entered onceinitState = ParseState PhHeader "" "" [] [] [] "" "" [] [] "" M.empty M.empty

  100 step :: SimaProConfig -> Text -> ParseState -> BS.ByteString -> ParseState
  101 -- entered 960 timesstep cfg methodologyName st line = case psPhase st of
  102     PhHeader
  103         | BS8.isPrefixOf "{" line -> st
  104         | otherwise -> st { psPhase = PhMethodMeta }
  105 
  106     PhMethodMeta
  107         | stripped == "Impact category" -> st { psPhase = PhExpectCatLine }
  108         | stripped == "Damage category" -> st { psPhase = PhExpectDcLine }
  109         | isNWsetMarker stripped -> st { psPhase = PhExpectNWname }
  110         | otherwise -> st
  111 
  112     PhExpectCategory
  113         | stripped == "Impact category" -> st { psPhase = PhExpectCatLine }
  114         | stripped == "Damage category" -> st { psPhase = PhExpectDcLine }
  115         | isNWsetMarker stripped -> st { psPhase = PhExpectNWname }
  116         | stripped == "End" -> st
  117         | otherwise -> st
  118 
  119     PhExpectCatLine
  120         | isBlank line -> st
  121         | otherwise ->
  122             let fields = splitCSV (spDelimiter cfg) line
  123                 catName = decodeBS (BS8.strip (head' fields))
  124                 catUnit = if length fields > 1
  125                           then decodeBS (BS8.strip (fields !! 1))
  126                           else ""
  127             in st { psPhase = PhExpectSubst
  128                   , psCatName = catName
  129                   , psCatUnit = catUnit
  130                   , psFactors = []
  131                   }
  132 
  133     PhExpectSubst
  134         | isBlank line -> st
  135         | stripped == "Substances" -> st { psPhase = PhReadingCFs }
  136         | otherwise -> st
  137 
  138     PhReadingCFs
  139         | isBlank line -> finishCategory st
  140         | stripped == "Impact category" ->
  141             (finishCategory st) { psPhase = PhExpectCatLine }
  142         | stripped == "Damage category" ->
  143             (finishCategory st) { psPhase = PhExpectDcLine }
  144         | isNWsetMarker stripped ->
  145             (finishCategory st) { psPhase = PhExpectNWname }
  146         | stripped == "End" -> finishCategory st
  147         | otherwise ->
  148             let fields = splitCSV (spDelimiter cfg) line
  149             in case fields of
  150                 (comp:sub:name:cas:cfVal:_unit:_) ->
  151                     let !cf = MethodCF
  152                             { mcfFlowRef     = makeFlowUUID name comp sub
  153                             , mcfFlowName    = decodeBS (BS8.strip name)
  154                             , mcfDirection   = direction comp
  155                             , mcfValue       = parseAmount (spDecimal cfg) (BS8.strip cfVal)
  156                             , mcfCompartment = mkCompartment comp sub
  157                             , mcfCAS         = normalizeCAS (decodeBS (BS8.strip cas))
  158                             }
  159                     in st { psFactors = cf : psFactors st }
  160                 _ -> st
  161 
  162     -- Damage category parsing
  163     PhExpectDcLine
  164         | isBlank line -> st
  165         | otherwise ->
  166             let fields = splitCSV (spDelimiter cfg) line
  167                 dcn = decodeBS (BS8.strip (head' fields))
  168                 dcu = if length fields > 1 then decodeBS (BS8.strip (fields !! 1)) else ""
  169             in st { psPhase = PhExpectDcImpacts, psDcName = dcn, psDcUnit = dcu, psDcImpacts = [] }
  170 
  171     PhExpectDcImpacts
  172         | isBlank line -> st
  173         | stripped == "Impact categories" -> st { psPhase = PhReadingDcImpacts }
  174         | otherwise -> st
  175 
  176     PhReadingDcImpacts
  177         | isBlank line -> finishDamageCategory st
  178         | stripped == "Damage category" ->
  179             (finishDamageCategory st) { psPhase = PhExpectDcLine }
  180         | isNWsetMarker stripped ->
  181             (finishDamageCategory st) { psPhase = PhExpectNWname }
  182         | stripped == "End" -> finishDamageCategory st
  183         | otherwise ->
  184             let fields = splitCSV (spDelimiter cfg) line
  185             in case fields of
  186                 (name:val:_) ->
  187                     let n = decodeBS (BS8.strip name)
  188                         v = parseAmount (spDecimal cfg) (BS8.strip val)
  189                     in st { psDcImpacts = (n, v) : psDcImpacts st }
  190                 _ -> st
  191 
  192     -- Normalization/Weighting parsing
  193     PhExpectNWname
  194         | isBlank line -> st
  195         | otherwise -> st { psPhase = PhExpectNWsection, psNWname = decodeBS (BS8.strip line) }
  196 
  197     PhExpectNWsection
  198         | isBlank line -> st
  199         | stripped == "Normalization" -> st { psPhase = PhReadingNorm }
  200         | stripped == "Weighting" -> st { psPhase = PhReadingWeight }
  201         | stripped == "Damage category" ->
  202             (finishNWset st) { psPhase = PhExpectDcLine }
  203         | isNWsetMarker stripped ->
  204             (finishNWset st) { psPhase = PhExpectNWname }
  205         | stripped == "End" -> finishNWset st
  206         | otherwise -> st
  207 
  208     PhReadingNorm
  209         | isBlank line -> st { psPhase = PhExpectNWsection }
  210         | stripped == "Weighting" -> st { psPhase = PhReadingWeight }
  211         | stripped == "End" -> finishNWset st
  212         | otherwise ->
  213             let fields = splitCSV (spDelimiter cfg) line
  214             in case fields of
  215                 (name:val:_) ->
  216                     let n = decodeBS (BS8.strip name)
  217                         v = parseAmount (spDecimal cfg) (BS8.strip val)
  218                     in st { psNormMap = M.insert n v (psNormMap st) }
  219                 _ -> st
  220 
  221     PhReadingWeight
  222         | isBlank line -> st { psPhase = PhExpectNWsection }
  223         | isNWsetMarker stripped ->
  224             (finishNWset st) { psPhase = PhExpectNWname }
  225         | stripped == "End" -> finishNWset st
  226         | otherwise ->
  227             let fields = splitCSV (spDelimiter cfg) line
  228             in case fields of
  229                 (name:val:_) ->
  230                     let n = decodeBS (BS8.strip name)
  231                         v = parseAmount (spDecimal cfg) (BS8.strip val)
  232                     in st { psWeightMap = M.insert n v (psWeightMap st) }
  233                 _ -> st
  234   where
  235     stripped = BS8.strip line
  236 
  237     finishCategory s =
  238         let !m = Method
  239                 { methodId          = UUID5.generateNamed simaproNamespace
  240                     (BS.unpack $ TE.encodeUtf8 $ "method:" <> psCatName s)
  241                 , methodName        = psCatName s
  242                 , methodDescription = Nothing
  243                 , methodUnit        = psCatUnit s
  244                 , methodCategory    = psCatName s
  245                 , methodMethodology = Just methodologyName
  246                 , methodFactors     = reverse (psFactors s)
  247                 }
  248         in s { psPhase = PhExpectCategory, psFactors = [], psMethods = m : psMethods s }
  249 
  250     finishDamageCategory s =
  251         let !dc = DamageCategory (psDcName s) (psDcUnit s) (reverse (psDcImpacts s))
  252         in s { psPhase = PhExpectCategory, psDcImpacts = [], psDamageCats = dc : psDamageCats s }
  253 
  254     finishNWset s
  255         | M.null (psNormMap s) && M.null (psWeightMap s) = s { psPhase = PhExpectCategory }
  256         | otherwise =
  257             let !nw = NormWeightSet (psNWname s) (psNormMap s) (psWeightMap s)
  258             in s { psPhase = PhExpectCategory
  259                  , psNormMap = M.empty, psWeightMap = M.empty, psNWname = ""
  260                  , psNWsets = nw : psNWsets s }
  261 
  262 finalize :: ParseState -> MethodCollection
  263 -- entered 10 timesfinalize st =
  264     let methods = case psPhase st of
  265             PhReadingCFs | not (null (psFactors st)) ->
  266                 let !m = Method
  267                         { methodId          = UUID5.generateNamed simaproNamespace
  268                             (BS.unpack $ TE.encodeUtf8 $ "method:" <> psCatName st)
  269                         , methodName        = psCatName st
  270                         , methodDescription = Nothing
  271                         , methodUnit        = psCatUnit st
  272                         , methodCategory    = psCatName st
  273                         , methodMethodology = Nothing
  274                         , methodFactors     = reverse (psFactors st)
  275                         }
  276                 in reverse (m : psMethods st)
  277             _ -> reverse (psMethods st)
  278         -- Flush any pending NW set
  279         nwSets = case psPhase st of
  280             PhReadingWeight | not (M.null (psWeightMap st)) ->
  281                 let !nw = NormWeightSet (psNWname st) (psNormMap st) (psWeightMap st)
  282                 in reverse (nw : psNWsets st)
  283             PhReadingNorm | not (M.null (psNormMap st)) ->
  284                 let !nw = NormWeightSet (psNWname st) (psNormMap st) (psWeightMap st)
  285                 in reverse (nw : psNWsets st)
  286             _ -> reverse (psNWsets st)
  287     in MethodCollection methods (reverse (psDamageCats st)) nwSets

  297 parseConfig :: [BS.ByteString] -> SimaProConfig
  298 -- entered onceparseConfig = foldl' go defaultConfig
  299   where
  300     go cfg line
  301         | "{CSV separator: Semicolon}" `BS.isInfixOf` line = cfg { spDelimiter = ';' }
  302         | "{CSV separator: Comma}" `BS.isInfixOf` line     = cfg { spDelimiter = ',' }
  303         | "{CSV separator: Tab}" `BS.isInfixOf` line       = cfg { spDelimiter = '\t' }
  304         | "{Decimal separator: .}" `BS.isInfixOf` line     = cfg { spDecimal = '.' }
  305         | "{Decimal separator: ,}" `BS.isInfixOf` line     = cfg { spDecimal = ',' }
  306         | otherwise = cfg
