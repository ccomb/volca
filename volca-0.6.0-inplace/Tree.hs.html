   13 -- | Nouvelle version optimisée avec les variants Exchange
   14 isTechnosphereInput :: FlowDB -> Exchange -> Bool
   15 isTechnosphereInput _ ex =
   16     case ex of
   17         TechnosphereExchange _ _ _ isInput isRef _ _ _ -> isInput && not isRef
   18         BiosphereExchange _ _ _ _ _ -> False

   20 -- | Get converted exchange amount ensuring unit compatibility
   21 -- Converts exchange amount to the target activity's reference unit for proper scaling
   22 getConvertedExchangeAmount :: UnitConfig -> Database -> Exchange -> UUID -> Double
   23 getConvertedExchangeAmount unitCfg db exchange targetActivityUUID =
   24     let originalAmount = exchangeAmount exchange
   25         -- Get exchange unit name
   26         exchangeUnitName = case M.lookup (exchangeUnitId exchange) (dbUnits db) of
   27             Just unit -> unitName unit
   28             Nothing -> "unknown"
   29         -- Get target activity's reference unit (lookup by activity UUID for backward compat)
   30         targetReferenceUnit = case findActivityByActivityUUID db targetActivityUUID of
   31             Just targetActivity -> activityUnit targetActivity
   32             Nothing -> "unknown"
   33     in if exchangeUnitName == "unknown" || targetReferenceUnit == "unknown"
   34        then originalAmount  -- No conversion possible, use original
   35        else convertExchangeAmount unitCfg exchangeUnitName targetReferenceUnit originalAmount

   37 -- | Build loop-aware tree for SVG export with maximum depth limit
   38 buildLoopAwareTree :: UnitConfig -> Database -> UUID -> Int -> LoopAwareTree
   39 buildLoopAwareTree unitCfg db rootUUID maxDepth =
   40     let maxNodes = 300  -- Maximum total nodes to prevent performance issues
   41         (tree, _) = buildLoopAwareTreeWithVisited unitCfg db rootUUID S.empty 0 maxDepth maxNodes
   42     in tree
   43 
   44 -- | Helper function with visited set, depth tracking, and node count limit
   45 -- Returns (tree, remainingNodeBudget)
   46 buildLoopAwareTreeWithVisited :: UnitConfig -> Database -> UUID -> S.Set UUID -> Int -> Int -> Int -> (LoopAwareTree, Int)
   47 buildLoopAwareTreeWithVisited unitCfg db activityUUID visited depth maxDepth remainingNodes
   48     -- Stop if node budget exhausted
   49     | remainingNodes <= 0 =
   50         case findActivityByActivityUUID db activityUUID of
   51             Nothing -> (TreeLoop activityUUID "Missing Activity (node limit)" depth, 0)
   52             Just activity -> (TreeLoop activityUUID (activityName activity) depth, 0)
   53     -- Stop at maximum depth
   54     | depth >= maxDepth =
   55         case findActivityByActivityUUID db activityUUID of
   56             Nothing -> (TreeLoop activityUUID "Missing Activity" depth, remainingNodes - 1)
   57             Just activity -> (TreeLoop activityUUID (activityName activity) depth, remainingNodes - 1)
   58     -- Detect loop
   59     | activityUUID `S.member` visited =
   60         case findActivityByActivityUUID db activityUUID of
   61             Nothing -> (TreeLoop activityUUID "Missing Activity" depth, remainingNodes - 1)
   62             Just activity -> (TreeLoop activityUUID (activityName activity) depth, remainingNodes - 1)
   63     -- Build subtree
   64     | otherwise =
   65         case findActivityByActivityUUID db activityUUID of
   66             Nothing -> (TreeLoop activityUUID "Missing Activity" depth, remainingNodes - 1)
   67             Just activity ->
   68                 let visited' = S.insert activityUUID visited
   69                     techInputs = [ ex | ex <- exchanges activity
   70                                      , isTechnosphereInput (dbFlows db) ex ]
   71                     -- Build children with node budget tracking
   72                     (children, finalBudget) = buildChildrenWithBudget techInputs visited' (depth + 1) maxDepth (remainingNodes - 1)
   73                     buildChildrenWithBudget :: [Exchange] -> S.Set UUID -> Int -> Int -> Int -> ([(Double, Flow, LoopAwareTree)], Int)
   74                     buildChildrenWithBudget [] _ _ _ budget = ([], budget)
   75                     buildChildrenWithBudget (ex:exs) vis d maxD budget
   76                         | budget <= 0 = ([], 0)  -- Stop if budget exhausted
   77                         | otherwise =
   78                             case (exchangeActivityLinkId ex, M.lookup (exchangeFlowId ex) (dbFlows db)) of
   79                                 (Just targetUUID, Just flow) ->
   80                                     let convertedAmount = getConvertedExchangeAmount unitCfg db ex targetUUID
   81                                         (subtree, budget') = buildLoopAwareTreeWithVisited unitCfg db targetUUID vis d maxD budget
   82                                         (restChildren, finalBudget') = buildChildrenWithBudget exs vis d maxD budget'
   83                                     in ((convertedAmount, flow, subtree) : restChildren, finalBudget')
   84                                 _ -> buildChildrenWithBudget exs vis d maxD budget
   85                 in if null children
   86                    then (TreeLeaf activity, finalBudget)
   87                    else (TreeNode activity children, finalBudget)
