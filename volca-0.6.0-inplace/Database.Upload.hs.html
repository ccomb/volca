   96 -- | Handle a database upload
   97 -- Extracts ZIP archive to uploadsDir/{slug}/ and detects format
   98 handleUpload :: FilePath -> UploadData -> (ProgressEvent -> IO ()) -> IO (Either Text UploadResult)
   99 -- never enteredhandleUpload uploadsDir UploadData{..} reportProgress = do
  100     let slug = slugify udName
  101 
  102     -- Check for valid slug
  103     if T.null slug
  104         then return $ Left "Invalid database name - cannot generate slug"
  105         else do
  106             let targetDir = uploadsDir </> T.unpack slug
  107 
  108             -- Check if directory already exists
  109             exists <- doesDirectoryExist targetDir
  110             if exists
  111                 then return $ Left $ "Database already exists: " <> slug
  112                 else do
  113                     -- Create target directory
  114                     createDirectoryIfMissing True targetDir
  115 
  116                     reportProgress $ ProgressEvent "extracting" 10 "Extracting archive..."
  117 
  118                     result <- try $ extractUpload udZipData targetDir
  119                     case result of
  120                         Left (e :: SomeException) -> do
  121                             _ <- try @SomeException $ removeDirectoryRecursive targetDir
  122                             return $ Left $ "Failed to extract archive: " <> T.pack (show e)
  123                         Right (Left err) -> do
  124                             _ <- try @SomeException $ removeDirectoryRecursive targetDir
  125                             return $ Left err
  126                         Right (Right ()) -> do
  127                             -- Find the actual data directory (may be nested)
  128                             reportProgress $ ProgressEvent "locating" 70 "Locating data files..."
  129                             dataDir <- findDataDirectory targetDir
  130 
  131                             -- Detect format
  132                             reportProgress $ ProgressEvent "detecting" 80 "Detecting database format..."
  133                             format <- detectDatabaseFormat dataDir
  134 
  135                             -- Count data files
  136                             reportProgress $ ProgressEvent "counting" 90 "Counting data files..."
  137                             fileCount <- countDataFiles dataDir format
  138 
  139                             reportProgress $ ProgressEvent "complete" 100 "Upload complete!"
  140 
  141                             return $ Right UploadResult
  142                                 { urSlug = slug
  143                                 , urPath = dataDir
  144                                 , urFormat = format
  145                                 , urFileCount = fileCount
  146                                 }

  148 -- | Supported archive format (detected by magic bytes)
  149 data ArchiveFormat
  150     = ArchiveZip     -- ZIP: 50 4B (PK)
  151     | Archive7z      -- 7z:  37 7A BC AF 27 1C
  152     | ArchiveGzip    -- gzip (tar.gz): 1F 8B
  153     | ArchiveXz      -- XZ (tar.xz): FD 37 7A 58 5A 00
  154     | ArchivePlainXML -- Plain XML file (EcoSpold1)
  155     | ArchivePlainCSV
  156     | ArchiveUnknown
  157     deriving (-- never entered-- never entered-- never enteredShow, -- never entered-- never enteredEq)

  159 -- | Detect archive format from magic bytes
  160 detectArchiveFormat :: BL.ByteString -> ArchiveFormat
  161 -- never entereddetectArchiveFormat content
  162     | BL.null content = ArchiveUnknown
  163     | matchesMagic [0x50, 0x4B] = ArchiveZip
  164     | matchesMagic [0x37, 0x7A, 0xBC, 0xAF, 0x27, 0x1C] = Archive7z
  165     | matchesMagic [0x1F, 0x8B] = ArchiveGzip
  166     | matchesMagic [0xFD, 0x37, 0x7A, 0x58, 0x5A, 0x00] = ArchiveXz
  167     | isXmlFile = ArchivePlainXML  -- Check XML before plain text
  168     | isPlainText = ArchivePlainCSV
  169     | otherwise = ArchiveUnknown
  170   where
  171     bytes = BL.unpack (BL.take 10 content)
  172     matchesMagic magic = take (length magic) bytes == magic
  173     -- Check for XML declaration "<?xml" (0x3C 0x3F 0x78 0x6D 0x6C)
  174     -- or "<ecoSpold" (0x3C 0x65 0x63 0x6F)
  175     -- or UTF-8 BOM (0xEF 0xBB 0xBF) followed by "<?xml"
  176     isXmlFile = matchesMagic [0x3C, 0x3F, 0x78, 0x6D, 0x6C]
  177              || matchesMagic [0x3C, 0x65, 0x63, 0x6F]
  178              || matchesMagic [0xEF, 0xBB, 0xBF, 0x3C, 0x3F]
  179     -- Check if first byte is printable ASCII (plain text / CSV)
  180     isPlainText = let b = BL.head content in b == 0x7B || (b >= 0x20 && b < 0x7F)

  189 -- | Extract upload data to target directory.
  190 -- Supports ZIP, 7z, tar.gz, tar.xz archives (auto-detected), plain XML, and CSV files.
  191 extractUpload :: BL.ByteString -> FilePath -> IO (Either Text ())
  192 -- never enteredextractUpload content targetDir = do
  193     let format = detectArchiveFormat content
  194     case format of
  195         ArchivePlainXML -> do
  196             -- Plain XML: write directly (e.g., EcoSpold1 multi-dataset file)
  197             BL.writeFile (targetDir </> "data.xml") content
  198             return $ Right ()
  199         ArchivePlainCSV -> do
  200             -- Plain CSV: write directly
  201             BL.writeFile (targetDir </> "data.csv") content
  202             return $ Right ()
  203         ArchiveUnknown ->
  204             return $ Left "Unsupported file format. Please upload a ZIP, 7z, tar.gz, tar.xz archive, XML, or CSV file."
  205         Archive7z -> extract7z content targetDir
  206         ArchiveZip -> extractZip content targetDir
  207         ArchiveGzip -> extractArchive ArchiveGzip content targetDir
  208         ArchiveXz -> extractArchive ArchiveXz content targetDir

  210 -- | Extract 7z archive using external 7z command
  211 -- The Haskell libarchive package doesn't properly support 7z extraction
  212 extract7z :: BL.ByteString -> FilePath -> IO (Either Text ())
  213 -- never enteredextract7z archiveData targetDir = do
  214     -- Write archive to a temp file
  215     let tempArchive = targetDir </> ".temp.7z"
  216     BL.writeFile tempArchive archiveData
  217 
  218     -- Get list of 7z commands to try (platform-specific)
  219     -- Checks for bundled binary first, then system paths
  220     commands <- get7zCommands
  221 
  222     -- Try each command until one works
  223     result <- tryCommands commands tempArchive targetDir
  224 
  225     -- Clean up temp file
  226     _ <- try @SomeException $ removeFile tempArchive
  227 
  228     case result of
  229         Left err -> return $ Left err
  230         Right () -> do
  231             -- Verify files were extracted
  232             files <- listDirectory targetDir
  233             -- Filter out any remaining temp files
  234             let actualFiles = filter (not . isPrefixOf ".temp") files
  235             if null actualFiles
  236                 then return $ Left "7z extraction produced no files. The archive may be empty or corrupted."
  237                 else return $ Right ()
  238   where
  239     -- Get platform-specific 7z command candidates
  240     -- Checks for bundled binary in VOLCA_DATA_DIR first, then system paths
  241     get7zCommands :: IO [FilePath]
  242     get7zCommands = do
  243         mDataDir <- lookupEnv "VOLCA_DATA_DIR"
  244         let bundledPaths = case mDataDir of
  245                 Just d  -> if isWindows
  246                     then [d </> "7z.exe", d </> "7z"]
  247                     else [d </> "7zz", d </> "7z"]
  248                 Nothing -> []
  249         return $ bundledPaths ++ systemPaths
  250       where
  251         isWindows = os == "mingw32" || os == "windows"
  252 
  253         systemPaths
  254             | isWindows = windowsCommands
  255             | otherwise = unixCommands
  256 
  257         -- Windows: try common installation paths
  258         windowsCommands =
  259             [ "7z"  -- If in PATH
  260             , "7z.exe"
  261             , "C:\\Program Files\\7-Zip\\7z.exe"
  262             , "C:\\Program Files (x86)\\7-Zip\\7z.exe"
  263             ]
  264 
  265         -- Unix/Linux/macOS: try various package names
  266         unixCommands =
  267             [ "7zz"   -- 7-Zip standalone (newer)
  268             , "7z"    -- p7zip-full
  269             , "7zr"   -- p7zip (reduced, but supports 7z)
  270             , "7za"   -- p7zip standalone
  271             ]
  272 
  273     tryCommands :: [FilePath] -> FilePath -> FilePath -> IO (Either Text ())
  274     tryCommands [] _ _ = return $ Left $ T.concat
  275         [ "7z extraction failed. Please install 7-Zip:\n"
  276         , if os == "mingw32" || os == "windows"
  277             then "  Windows: Download from https://7-zip.org/"
  278             else "  Linux: apt install 7zip (or p7zip-full)\n  macOS: brew install p7zip"
  279         ]
  280     tryCommands (cmd:rest) archive target = do
  281         result <- tryExtractWith cmd archive target
  282         case result of
  283             Right () -> return $ Right ()
  284             Left _ -> tryCommands rest archive target
  285 
  286     tryExtractWith :: FilePath -> FilePath -> FilePath -> IO (Either Text ())
  287     tryExtractWith cmd archive target = do
  288         result <- try @SomeException $
  289             readProcessWithExitCode cmd ["x", "-y", "-o" ++ target, archive] ""
  290         case result of
  291             Left _ -> return $ Left "command not found"
  292             Right (ExitSuccess, _, _) -> return $ Right ()
  293             Right (ExitFailure _, _, stderr) -> return $ Left (T.pack stderr)

  308         Left err -> return $ Left err

  313                 then return $ Left "ZIP extraction produced no files."

  318         mDataDir <- lookupEnv "VOLCA_DATA_DIR"

  320                 Just d -> if isWindows

  334     args7z    [archive, target] = ["x", "-y", "-o" ++ target, archive]
  335     args7z    _ = []

  342             else "  Linux: apt install unzip\n  macOS: brew install unzip (usually pre-installed)"

  345         let args = mkArgs [archive, target]
  346         result <- try @SomeException $ readProcessWithExitCode cmd args ""

  348             Left _ -> tryZipCommands rest archive target

  350             Right (ExitFailure _, _, _) -> tryZipCommands rest archive target

  389 -- | Find the actual data directory (picks the candidate with the most data files)
  390 -- Archives often contain multiple folders (e.g., "datasets", "MasterData")
  391 -- For ILCD: detects directories with a processes/ subdirectory first.
  392 findDataDirectory :: FilePath -> IO FilePath
  393 -- never enteredfindDataDirectory dir = do
  394     -- Check for ILCD format first (recursively find dirs with processes/ subdir)
  395     ilcdDir <- findILCDRoot dir
  396     case ilcdDir of
  397         Just d  -> return d
  398         Nothing -> do
  399             candidates <- findAllDataDirectories dir
  400             case candidates of
  401                 []    -> return dir
  402                 [one] -> return one
  403                 many  -> pickByFileCount many
  404   where
  405     pickByFileCount dirs = do
  406         counts <- mapM (\d -> (,) d <$> countDataFilesIn d) dirs
  407         let sorted = sortOn (Down . snd) counts
  408         return $ fst (head sorted)
  409 
  410     -- Recursively find the first directory containing a processes/ subdirectory
  411     findILCDRoot d = do
  412         hasProcesses <- doesDirectoryExist (d </> "processes")
  413         if hasProcesses then return (Just d)
  414         else do
  415             entries <- listDirectory d
  416             subdirs <- filterM doesDirectoryExist (map (d </>) entries)
  417             findFirst subdirs
  418 
  419     findFirst [] = return Nothing
  420     findFirst (d:ds) = do
  421         r <- findILCDRoot d
  422         case r of
  423             Just _  -> return r
  424             Nothing -> findFirst ds

  548         let path = d </> entry
  549         isDir <- doesDirectoryExist path
  550         if isDir
  551             then listDirectoryRecursive path
  552             else return [path]
  553     return $ concat results

  555 -- | Detect database format from extracted files or a single file
  556 detectDatabaseFormat :: FilePath -> IO DatabaseFormat
  557 -- never entereddetectDatabaseFormat path = do
  558     isFile <- doesFileExist path
  559     isDir <- doesDirectoryExist path
  560 
  561     if isFile
  562         then do
  563             let ext = map toLower (takeExtension path)
  564             case ext of
  565                 ".spold" -> return EcoSpold2
  566                 ".xml" -> do
  567                     isEcoSpold1 <- checkForEcoSpold1 [path]
  568                     return $ if isEcoSpold1 then EcoSpold1 else UnknownFormat
  569                 ".csv" -> do
  570                     isSimaPro <- checkForSimaProCSV [path]
  571                     return $ if isSimaPro then SimaProCSV else UnknownFormat
  572                 _ -> return UnknownFormat
  573         else if isDir
  574             then do
  575                 -- Check for ILCD format: has a processes/ subdirectory
  576                 hasProcesses <- doesDirectoryExist (path </> "processes")
  577                 if hasProcesses
  578                     then return ILCDProcess
  579                     else do
  580                         fs <- listDirectoryRecursive path
  581                         let extensions = map (map toLower . takeExtension) fs
  582                         let hasSpold = any (== ".spold") extensions
  583                             hasXml = any (== ".xml") extensions
  584                             hasCsv = any (== ".csv") extensions
  585                         if hasSpold
  586                             then return EcoSpold2
  587                             else if hasXml
  588                                 then do
  589                                     isEcoSpold1 <- checkForEcoSpold1 fs
  590                                     return $ if isEcoSpold1 then EcoSpold1 else UnknownFormat
  591                                 else if hasCsv
  592                                     then do
  593                                         isSimaPro <- checkForSimaProCSV fs
  594                                         return $ if isSimaPro then SimaProCSV else UnknownFormat
  595                                     else return UnknownFormat
  596             else return UnknownFormat
  597 
  598 -- | Check if XML files are EcoSpold1 format
  599 checkForEcoSpold1 :: [FilePath] -> IO Bool
  600 -- never enteredcheckForEcoSpold1 fs = do
  601     let xmlFiles = filter (\f -> ".xml" `isSuffixOf` map toLower f) fs
  602     case xmlFiles of
  603         [] -> return False
  604         (f:_) -> do
  605             result <- try $ TIO.readFile f
  606             case result of
  607                 Left (_ :: SomeException) -> return False
  608                 Right c ->
  609                     return $ T.isInfixOf "EcoSpold01" c || T.isInfixOf "ecoSpold" c
  610 
  611 -- | Check if CSV files are SimaPro format
  612 -- Uses ByteString to avoid UTF-8 encoding issues (SimaPro files often use Latin1)
  613 checkForSimaProCSV :: [FilePath] -> IO Bool
  614 -- never enteredcheckForSimaProCSV fs = do
  615     let csvFiles = filter (\f -> ".csv" `isSuffixOf` map toLower f) fs
  616     case csvFiles of
  617         [] -> return False
  618         (f:_) -> do
  619             result <- try $ BS.readFile f
  620             case result of
  621                 Left (_ :: SomeException) -> return False
  622                 Right c ->
  623                     return $ BS.isPrefixOf "{" c

  625 -- | Convert name to URL-safe slug
  626 slugify :: Text -> Text
  627 -- never enteredslugify = T.intercalate "-" . filter (not . T.null) . T.split (not . isValidSlugChar) . T.toLower
  628   where
  629     isValidSlugChar c = isAlphaNum c || c == '-' || c == '_'
