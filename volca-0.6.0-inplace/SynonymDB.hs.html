   42 -- | Build a SynonymDB from CSV content (two columns: name1, name2).
   43 -- Each row declares two names as direct synonyms. No transitive closure:
   44 -- if A↔B and B↔C, looking up A returns {A,B} but NOT C.
   45 -- This prevents chain pollution (e.g., sulfate→gypsum→calcium→calcite→carbonate→lithium carbonate).
   46 buildFromCSV :: BL.ByteString -> Either String SynonymDB
   47 -- entered 4 timesbuildFromCSV csvData =
   48     case decode HasHeader csvData of
   49         Left err -> Left $ "CSV parse error: " <> err
   50         Right rows -> Right $ buildFromPairs (V.toList (rows :: V.Vector (Text, Text)))

   52 -- | Build SynonymDB from direct pairs (no transitive closure).
   53 -- Each pair (A, B) creates a group containing both A and B.
   54 -- If A appears in multiple pairs, all its direct partners are in one group.
   55 -- But partners' partners are NOT included (no transitivity).
   56 buildFromPairs :: [(Text, Text)] -> SynonymDB
   57 -- entered 20 timesbuildFromPairs pairs =
   58     let -- Collect direct neighbors for each name (star topology, not transitive)
   59         directNeighbors = foldl addPair M.empty pairs
   60         -- Filter: reject names with too many direct synonyms (overly generic terms)
   61         maxDirectSynonyms = 50
   62         validNeighbors = M.filter ((<= maxDirectSynonyms) . length) directNeighbors
   63         -- Build groups: each name's group = itself + its direct neighbors
   64         allNames = M.keys validNeighbors
   65         numberedGroups = zip [0..] [ name : M.findWithDefault [] name validNeighbors | name <- allNames ]
   66         nameToId = M.fromList [ (name, gid) | (gid, members) <- numberedGroups, name <- members ]
   67         idToNames = M.fromList [ (gid, members) | (gid, members) <- numberedGroups ]
   68     in SynonymDB nameToId idToNames
   69   where
   70     addPair :: M.Map Text [Text] -> (Text, Text) -> M.Map Text [Text]
   71     addPair acc (raw1, raw2) =
   72         let n1 = normalizeName raw1
   73             n2 = normalizeName raw2
   74             o1 = T.strip raw1
   75             o2 = T.strip raw2
   76         in if T.null n1 || T.null n2 || n1 == n2 then acc
   77            else M.insertWith (++) n1 [o2] $ M.insertWith (++) n2 [o1] acc

   79 -- | Load a SynonymDB from a CSV file, using a binary cache for speed.
   80 --   On first load: parse CSV → build SynonymDB → save .cache.zst
   81 --   On subsequent loads: load .cache.zst directly (if newer than CSV)
   82 loadFromCSVFileWithCache :: FilePath -> IO (Either String SynonymDB)
   83 -- never enteredloadFromCSVFileWithCache csvPath = do
   84     let cachePath = csvPath ++ ".cache.zst"
   85     cached <- loadCache cachePath csvPath
   86     case cached of
   87         Just db -> return (Right db)
   88         Nothing -> do
   89             csvData <- BL.readFile csvPath
   90             case buildFromCSV csvData of
   91                 Left err -> return (Left err)
   92                 Right db -> do
   93                     saveCache cachePath db
   94                     return (Right db)
   95   where
   96     loadCache cachePath srcPath = do
   97         exists <- doesFileExist cachePath
   98         if not exists then return Nothing
   99         else catch
  100             (do cacheTime <- getModificationTime cachePath
  101                 srcTime <- getModificationTime srcPath
  102                 if cacheTime < srcTime then return Nothing
  103                 else do
  104                     compressed <- BS.readFile cachePath
  105                     case Zstd.decompress compressed of
  106                         Zstd.Decompress raw -> do
  107                             let !db = decodeEx raw
  108                             result <- evaluate (force db)
  109                             return (Just result)
  110                         _ -> return Nothing)
  111             (\(_ :: SomeException) -> return Nothing)
  112     saveCache cachePath db = catch
  113         (BS.writeFile cachePath (Zstd.compress 1 (encode db)))
  114         (\(_ :: SomeException) -> return ())

  116 -- | Merge multiple SynonymDBs into one (later entries take priority on ID conflicts).
  117 mergeSynonymDBs :: [SynonymDB] -> SynonymDB
  118 -- entered 4 timesmergeSynonymDBs [] = emptySynonymDB
  119 mergeSynonymDBs [db] = db
  120 mergeSynonymDBs dbs =
  121     let -- Collect all groups from all DBs, re-number them
  122         allGroups = concatMap (\db -> M.elems (synIdToNames db)) dbs
  123         numberedGroups = zip [0..] allGroups
  124         nameToId = M.fromList [(normalizeName name, gid) | (gid, names) <- numberedGroups, name <- names]
  125         idToNames = M.fromList numberedGroups
  126     in SynonymDB nameToId idToNames

  132 -- | Normalize a name for lookup in the synonym database
  133 --
  134 -- Normalization rules:
  135 -- - Lowercase
  136 -- - Strip leading/trailing whitespace
  137 -- - Collapse multiple spaces to single space
  138 -- - Strip ", in ground" suffix (ecoinvent resource naming)
  139 -- - Strip "/kg" suffix (SimaPro unit convention)
  140 -- - Remove punctuation: commas, parentheses, quotes
  141 normalizeName :: Text -> Text
  142 -- entered 222 timesnormalizeName name =
  143     let -- Lowercase and strip
  144         t1 = T.strip $ T.toLower name
  145         -- Collapse whitespace
  146         t2 = T.unwords $ T.words t1
  147         -- Strip ", in ground" suffix
  148         t3 = stripSuffix ", in ground" $ stripSuffix " in ground" t2
  149         -- Strip "/kg" suffix
  150         t4 = stripSuffix "/kg" t3
  151         -- Remove punctuation
  152         t5 = T.filter (`notElem` (",()'\"" :: String)) t4
  153         -- Collapse whitespace again (from removed punctuation)
  154         t6 = T.unwords $ T.words t5
  155     in t6
  156   where
  157     stripSuffix :: Text -> Text -> Text
  158     stripSuffix suffix txt =
  159         if suffix `T.isSuffixOf` txt
  160             then T.dropEnd (T.length suffix) txt
  161             else txt
