  149 -- | Strip a trailing database tag from a product name.
  150 -- Matches patterns like "(WFLDB)", "(AGRIBALYSE)", "(SALCA)" — a parenthesized
  151 -- suffix where the content is all uppercase letters.
  152 -- Returns Just strippedName if a tag was found, Nothing otherwise.
  153 stripTrailingDBTag :: Text -> Maybe Text
  154 -- entered 29 timesstripTrailingDBTag name
  155     | T.null name = Nothing
  156     | T.last name /= ')' = Nothing
  157     | otherwise =
  158         let withoutClose = T.init name  -- drop trailing ')'
  159             (before, tag) = T.breakOnEnd "(" withoutClose
  160         in if T.null before
  161             then Nothing  -- no opening paren found
  162             else let prefix = T.init before  -- drop the '(' at end of 'before'
  163                      stripped = T.strip prefix
  164                  in if not (T.null tag)
  165                        && T.all (\c -> isUpper c || c == '-') tag
  166                        && T.any isAlpha tag  -- at least one letter
  167                        && not (T.null stripped)
  168                     then Just stripped
  169                     else Nothing

  171 -- | Strip a trailing @\/LOCATION MARKER@ suffix from a product name.
  172 -- WFLDB names use the pattern @ProductName (WFLDB)\/CA U@ where the suffix
  173 -- encodes location (2-3 uppercase letters) and unit type (1 uppercase letter).
  174 stripTrailingLocationSuffix :: Text -> Maybe Text
  175 -- entered 26 timesstripTrailingLocationSuffix name =
  176     case T.breakOnEnd "/" name of
  177         ("", _) -> Nothing
  178         (beforeSlash, afterSlash) ->
  179             case T.words afterSlash of
  180                 [loc, marker]
  181                     | T.length loc >= 2, T.length loc <= 3
  182                     , T.all isUpper loc
  183                     , T.length marker == 1
  184                     , T.all isUpper marker
  185                     -> Just (T.stripEnd (T.init beforeSlash))
  186                 _ -> Nothing

  188 -- | Extract product name prefixes from a compound name.
  189 -- Tries splitting at each separator and returns candidate prefixes (stripped).
  190 -- Also tries stripping a trailing database tag like "(WFLDB)" and a trailing
  191 -- location suffix like "/CA U" (WFLDB convention).
  192 -- Returns empty list if no separator is found and no tag detected.
  193 extractProductPrefixes :: Text -> [Text]
  194 -- entered 27 timesextractProductPrefixes name =
  195     let separatorPrefixes =
  196             [ T.strip prefix
  197             | sep <- compoundSeparators
  198             , let (prefix, rest) = T.breakOn sep name
  199             , not (T.null rest)       -- separator was found
  200             , not (T.null prefix)     -- non-empty prefix
  201             ]
  202         tagStripped = case stripTrailingDBTag name of
  203             Just stripped -> [stripped]
  204             Nothing       -> []
  205         locationSuffixStripped = case stripTrailingLocationSuffix name of
  206             Just stripped -> stripped : case stripTrailingDBTag stripped of
  207                 Just alsoTagStripped -> [alsoTagStripped]
  208                 Nothing              -> []
  209             Nothing -> []
  210     in separatorPrefixes ++ tagStripped ++ locationSuffixStripped

  212 -- | Extract a location code from any bracket pattern in a name.
  213 -- Tries {XX} first (standard LCA geography notation), then [XX] with
  214 -- validation to avoid chemical notation like [thio] or metadata like [Dummy].
  215 extractBracketedLocation :: Text -> Text
  216 -- entered 5 timesextractBracketedLocation name =
  217     case extractFromBrackets '{' '}' name of
  218         Just loc | not (T.null loc) -> loc
  219         _ -> case extractFromBrackets '[' ']' name of
  220             Just loc | looksLikeGeo loc -> loc
  221             _ -> ""
  222   where
  223     -- Accept only short codes starting uppercase (e.g. "GLO", "FR", "RER", "RoW")
  224     looksLikeGeo t =
  225         T.length t >= 2 && T.length t <= 3
  226         && isUpper (T.head t)
  227     extractFromBrackets :: Char -> Char -> Text -> Maybe Text
  228     extractFromBrackets open close txt =
  229         let (_, afterOpen) = T.breakOn (T.singleton open) txt
  230         in if T.null afterOpen
  231             then Nothing
  232             else let inside = T.drop 1 afterOpen  -- skip the open bracket
  233                      (content, afterClose) = T.breakOn (T.singleton close) inside
  234                  in if T.null afterClose
  235                     then Nothing
  236                     else Just (T.strip content)

  238 -- | Build an indexed database for fast cross-DB lookups
  239 -- This should be called once when a database is loaded
  240 buildIndexedDatabase :: Text -> SynonymDB -> SimpleDatabase -> IndexedDatabase
  241 -- entered 5 timesbuildIndexedDatabase dbName synDB db =
  242     let entries = buildSupplierEntries db
  243         -- Index by normalized product name + extracted prefixes
  244         byName = M.fromListWith (++)
  245             [ (normalizeText name, [entry])
  246             | (prodName, entry) <- entries
  247             , name <- prodName : extractProductPrefixes prodName
  248             , not (T.null (normalizeText name))
  249             ]
  250         -- Index by synonym group (for synonym matching)
  251         bySynonym = M.fromListWith (++)
  252             [ (groupId, [entry])
  253             | (prodName, entry) <- entries
  254             , Just groupId <- [lookupSynonymGroup synDB (normalizeName prodName)]
  255             ]
  256     in IndexedDatabase
  257         { idbName = dbName
  258         , idbByProductName = byName
  259         , idbBySynonymGroup = bySynonym
  260         }

  323 -- | Find a supplier across all loaded databases (using pre-built indexes)
  324 -- This is the fast O(1) lookup version
  325 findSupplierInIndexedDBs
  326     :: LinkingContext
  327     -> Text           -- ^ Product name to find
  328     -> Text           -- ^ Location of the consumer
  329     -> Text           -- ^ Unit of the exchange
  330     -> CrossDBLinkResult
  331 -- entered 5 timesfindSupplierInIndexedDBs LinkingContext{..} productName location unit =
  332     let normalizedName = normalizeText productName
  333         -- Try exact match first (O(1) lookup)
  334         exactCandidates = concatMap (lookupExact normalizedName) lcIndexedDatabases
  335         -- Try synonym match if no exact match
  336         synonymCandidates = if null exactCandidates
  337             then case lookupSynonymGroup lcSynonymDB (normalizeName productName) of
  338                 Just groupId -> concatMap (lookupBySynonym groupId) lcIndexedDatabases
  339                 Nothing -> []
  340             else []
  341         -- Fallback: try prefix-based splitting for compound names (e.g. SimaPro)
  342         prefixCandidates = if null exactCandidates && null synonymCandidates
  343             then tryPrefixes (extractProductPrefixes productName)
  344             else []
  345         allCandidates = exactCandidates ++ synonymCandidates ++ prefixCandidates
  346         -- Effective location: if raw location is empty, try extracting from compound name
  347         effectiveLocation = if T.null location
  348             then extractBracketedLocation productName
  349             else location
  350     in if null allCandidates
  351         then CrossDBNotLinked NoNameMatch
  352         else
  353             -- Check unit compatibility first
  354             let unitCompatible = filter (\(_, se) -> unitsAreCompatible lcUnitConfig unit (seUnit se)) allCandidates
  355             in if null unitCompatible
  356                 then
  357                     -- All candidates failed unit check — report the first supplier's unit
  358                     let (_, firstSe) = head allCandidates
  359                     in CrossDBNotLinked (UnitIncompatible unit (seUnit firstSe))
  360                 else
  361                     -- Score by effective location
  362                     let scoredCandidates = map (scoreEntry effectiveLocation) unitCompatible
  363                         !best = maximumBy (comparing cdbScore) scoredCandidates
  364                     in if cdbScore best >= lcThreshold
  365                         then
  366                             -- Build warnings (only when original location was explicit)
  367                             let warnings = if T.null location then [] else buildWarnings effectiveLocation (cdbLocation best)
  368                             in CrossDBLinked (cdbActivityUUID best) (cdbProductUUID best)
  369                                             (cdbDatabaseName best) (cdbScore best)
  370                                             (cdbProductName best) (cdbLocation best) warnings
  371                         else CrossDBNotLinked (LocationUnavailable effectiveLocation)
  372   where
  373     lookupExact :: Text -> IndexedDatabase -> [(Text, SupplierEntry)]
  374     lookupExact name idb =
  375         [(idbName idb, entry) | entry <- fromMaybe [] (M.lookup name (idbByProductName idb))]
  376 
  377     lookupBySynonym :: Int -> IndexedDatabase -> [(Text, SupplierEntry)]
  378     lookupBySynonym groupId idb =
  379         [(idbName idb, entry) | entry <- fromMaybe [] (M.lookup groupId (idbBySynonymGroup idb))]
  380 
  381     -- Try each prefix from compound name splitting, return first match
  382     tryPrefixes :: [Text] -> [(Text, SupplierEntry)]
  383     tryPrefixes [] = []
  384     tryPrefixes (p:ps) =
  385         let normalized = normalizeText p
  386             candidates = concatMap (lookupExact normalized) lcIndexedDatabases
  387         in if null candidates
  388             then -- Also try synonym match for this prefix
  389                 case lookupSynonymGroup lcSynonymDB (normalizeName p) of
  390                     Just groupId ->
  391                         let synCandidates = concatMap (lookupBySynonym groupId) lcIndexedDatabases
  392                         in if null synCandidates then tryPrefixes ps else synCandidates
  393                     Nothing -> tryPrefixes ps
  394             else candidates

  396     scoreEntry :: Text -> (Text, SupplierEntry) -> CrossDBCandidate
  397     scoreEntry queryLoc (dbName, SupplierEntry{..}) =
  398         let locScore = matchLocation lcLocationHierarchy queryLoc seLocation
  399             nameScore = 50
  400             !totalScore = nameScore + locScore
  401         in CrossDBCandidate
  402             { cdbActivityUUID = seActivityUUID
  403             , cdbProductUUID = seProductUUID
  404             , cdbDatabaseName = dbName

  442 -- | Match locations with hierarchy fallback
  443 --
  444 -- Returns:
  445 --   30 = Exact match
  446 --   20 = Subregion match (e.g., FR ⊂ Europe)
  447 --   10 = Global fallback (GLO or RoW)
  448 --    5 = Different but not blocking
  449 matchLocation :: M.Map Text [Text] -> Text -> Text -> Int
  450 -- entered 12 timesmatchLocation hier queryLoc candidateLoc
  451     | queryLoc == candidateLoc                           = 30  -- Exact
  452     | isSubregionOf hier queryLoc candidateLoc           = 20  -- Widening (FR→GLO, FR→RER)

  454     | isSubregionOf hier candidateLoc queryLoc           = 0   -- Narrowing (GLO→FR) — blocked
  455     | otherwise                                          = 5   -- Unrelated

  578 -- | Check if two units are compatible for linking
  579 unitsAreCompatible :: UC.UnitConfig -> Text -> Text -> Bool
  580 -- entered 3 timesunitsAreCompatible cfg unit1 unit2
  581     | normalizeText unit1 == normalizeText unit2 = True  -- Same unit (after normalization)
  582     | otherwise = UC.unitsCompatible cfg unit1 unit2     -- Dimensionally compatible
