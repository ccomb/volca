  599 -- | Build word-token search index: lowercased word → IntSet of ProcessIds

  602 buildSearchIndex :: V.Vector Activity -> M.Map Text IS.IntSet
  603 -- never enteredbuildSearchIndex activities =

  608             words_ = T.words (T.toLower (activityName a))
  609                   ++ T.words (T.toLower (activityLocation a))
  610         in foldl' (\m w -> MS.insertWith IS.union w (IS.singleton pid) m) acc words_
