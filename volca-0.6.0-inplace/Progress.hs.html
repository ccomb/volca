  125 -- | Report progress with consistent formatting and thread-safe output
  126 reportProgress :: ProgressLevel -> String -> IO ()
  127 -- entered 3604 timesreportProgress level message = do
  128     let prefix = case level of
  129             Info    -> ""
  130             Warning -> "[WARN] "
  131             Cache   -> "[CACHE] "
  132             Matrix  -> "[MATRIX] "
  133             Solver  -> "[SOLVER] "
  134             Error   -> "[ERROR] "
  135         formatted = prefix ++ message
  136     -- Append to in-memory buffer
  137     appendLogLine formatted
  138     -- Serialize all output to prevent thread-unsafe stderr corruption
  139     -- Catch encoding errors (safety net for Windows code page issues)
  140     withMVar progressOutputMutex $ \_ ->
  141         catch (hPutStrLn stderr formatted >> hFlush stderr)
  142               (\(_ :: SomeException) -> return ())
