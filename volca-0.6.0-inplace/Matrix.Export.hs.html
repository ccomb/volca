   41 -- | Extract matrix debug information from Database
   42 extractMatrixDebugInfo :: Database -> UUID -> Maybe Text -> IO MatrixDebugInfo
   43 -- never enteredextractMatrixDebugInfo database targetUUID flowFilter = do
   44     let activities = dbActivities database
   45         flows = dbFlows database
   46         techTriples = dbTechnosphereTriples database
   47         bioTriples = dbBiosphereTriples database
   48         activityIndexVec = dbActivityIndex database
   49         bioFlowUUIDs = dbBiosphereFlows database
   50         activityCount = dbActivityCount database
   51         bioFlowCount = dbBiosphereCount database
   52 
   53         targetProcessId = case findProcessIdByActivityUUID database targetUUID of
   54             Just pid -> pid
   55             Nothing -> error $ "Activity " <> show targetUUID <> " has no ProcessId in database for debug-matrices"
   56         demandVec = buildDemandVectorFromIndex activityIndexVec targetProcessId
   57         demandList = toList demandVec
   58 
   59         techTriplesInt = [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- U.toList techTriples]
   60         activityCountInt = fromIntegral activityCount
   61 
   62     supplyVec <- solveSparseLinearSystem techTriplesInt activityCountInt demandVec
   63     let supplyList = toList supplyVec
   64 
   65         bioTriplesInt = [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- U.toList bioTriples]
   66         bioFlowCountInt = fromIntegral bioFlowCount
   67         inventoryVec = applySparseMatrix bioTriplesInt bioFlowCountInt supplyVec
   68         inventoryList = toList inventoryVec
   69 
   70         filteredBioTriples = case flowFilter of
   71             Nothing -> bioTriples
   72             Just filterText ->
   73                 let matchingFlowIndices =
   74                         [ idx | (uuid, idx) <- zip (V.toList bioFlowUUIDs) ([0 ..] :: [Int]), Just flow <- [M.lookup uuid flows], T.toLower filterText `T.isInfixOf` T.toLower (flowName flow)
   75                         ]
   76                     matchingFlowIndicesInt32 = map fromIntegral matchingFlowIndices :: [Int32]
   77                  in U.filter (\(SparseTriple row _ _) -> row `elem` matchingFlowIndicesInt32) bioTriples
   78     return MatrixDebugInfo
   79             { mdActivities = activities
   80             , mdFlows = flows
   81             , mdTechTriples = techTriples
   82             , mdBioTriples = filteredBioTriples
   83             , mdActivityIndex = activityIndexVec
   84             , mdBioFlowUUIDs = bioFlowUUIDs
   85             , mdTargetUUID = targetUUID
   86             , mdTargetProcessId = targetProcessId
   87             , mdDatabase = database
   88             , mdSupplyVector = supplyList
   89             , mdDemandVector = demandList
   90             , mdInventoryVector = inventoryList
   91             }

  123 -- | Export supply chain data showing which activities contribute to target
  124 exportSupplyChainData :: FilePath -> MatrixDebugInfo -> IO ()
  125 -- never enteredexportSupplyChainData filePath debugInfo = do
  126     let database = mdDatabase debugInfo
  127         activities = mdActivities debugInfo
  128         activityIndexVec = mdActivityIndex debugInfo
  129         supplyVector = mdSupplyVector debugInfo
  130 
  131         supplyChainRows =
  132             [ csvRow processId activity idx supply
  133             | processId <- [toEnum 0 .. toEnum (V.length activities - 1)]
  134             , let activity = activities V.! fromEnum processId
  135             , let idx = fromIntegral (activityIndexVec V.! fromEnum processId) :: Int
  136             , let supply = if idx < length supplyVector then supplyVector !! idx else 0.0
  137             ]
  138 
  139         csvRow processId activity idx supply =
  140             [ T.unpack (processIdToText database processId)
  141             , T.unpack (activityName activity)
  142             , T.unpack (activityLocation activity)
  143             , show supply
  144             , show idx
  145             ]
  146 
  147         csvHeader = ["activity_id", "activity_name", "location", "supply_amount", "col_idx"]
  148         allRows = csvHeader : supplyChainRows
  149         csvContent = L.intercalate "\n" (map (L.intercalate ",") allRows)
  150 
  151     putStrLn $ "DEBUG: Writing supply chain data with " ++ show (length supplyChainRows) ++ " activities"
  152     putStrLn $ "DEBUG: Supply vector length: " ++ show (length supplyVector)

  155 -- | Export biosphere matrix contributions
  156 exportBiosphereMatrixData :: FilePath -> MatrixDebugInfo -> IO ()
  157 -- never enteredexportBiosphereMatrixData filePath debugInfo = do
  158     let database = mdDatabase debugInfo
  159         flows = mdFlows debugInfo
  160         activities = mdActivities debugInfo
  161         bioTriples = mdBioTriples debugInfo
  162         bioFlowUUIDs = mdBioFlowUUIDs debugInfo
  163         activityIndex = mdActivityIndex debugInfo
  164         supplyVector = mdSupplyVector debugInfo
  165 
  166         matrixRows =
  167             [ csvRow bioTriple
  168             | bioTriple@(SparseTriple _row _col value) <- U.toList bioTriples
  169             , abs value > 1e-20
  170             ]
  171 
  172         csvRow (SparseTriple row col value) =
  173             let rowInt = fromIntegral row :: Int
  174                 colInt = fromIntegral col :: Int
  175              in [ maybe "unknown" show (getFlowUUID rowInt)
  176                 , maybe "unknown" (T.unpack . flowName) (getFlow rowInt)
  177                 , maybe "unknown" T.unpack (getFlowUnit rowInt)
  178                 , maybe "unknown" (T.unpack . processIdToText database) (getActivityProcessId colInt)
  179                 , maybe "unknown" (T.unpack . activityName) (lookupActivity colInt)
  180                 , show value
  181                 , show (realContribution colInt)
  182                 ]
  183           where
  184             getFlowUUID :: Int -> Maybe UUID
  185             getFlowUUID rowIdx =
  186                 if rowIdx < V.length bioFlowUUIDs
  187                     then Just (bioFlowUUIDs V.! rowIdx)
  188                     else Nothing
  189             getFlow :: Int -> Maybe Flow
  190             getFlow rowIdx = getFlowUUID rowIdx >>= flip M.lookup flows
  191             getFlowUnit :: Int -> Maybe Text
  192             getFlowUnit rowIdx = fmap (getUnitNameForFlow (dbUnits database)) (getFlow rowIdx)
  193             getActivityProcessId :: Int -> Maybe ProcessId
  194             getActivityProcessId colIdx =
  195                 L.find (\pid -> fromIntegral (activityIndex V.! fromEnum pid) == colIdx)
  196                        [toEnum 0 .. toEnum (V.length activities - 1)]
  197             lookupActivity :: Int -> Maybe Activity
  198             lookupActivity colIdx = do
  199                 processId <- getActivityProcessId colIdx
  200                 if fromEnum processId < V.length activities
  201                     then Just (activities V.! fromEnum processId)
  202                     else Nothing
  203             realContribution :: Int -> Double
  204             realContribution colIdx =
  205                 if colIdx < length supplyVector
  206                     then value * (supplyVector !! colIdx)
  207                     else 0.0
  208 
  209         csvHeader = ["flow_id", "flow_name", "unit", "activity_id", "activity_name", "matrix_value", "contribution"]
  210         allRows = csvHeader : matrixRows
  211         csvContent = L.intercalate "\n" (map (L.intercalate ",") allRows)
  212 
  213     putStrLn $ "DEBUG: Writing biosphere matrix with " ++ show (length matrixRows) ++ " entries"
  214     writeFile filePath csvContent

  228 -- | Escape text for CSV output (semicolon delimiter)
  229 escapeCsvField :: Text -> Text
  230 -- entered 120 timesescapeCsvField text
  231     | T.any (\c -> c == ';' || c == '"' || c == '\n' || c == '\r') text =
  232         "\"" <> T.replace "\"" "\"\"" text <> "\""
  233     | otherwise = text

  235 -- | Export ie_index.csv (Intermediate Exchanges)
  236 exportIEIndex :: FilePath -> Database -> IO ()
  237 -- entered 6 timesexportIEIndex filePath db = do
  238     let activities = dbActivities db
  239         processIdTable = dbProcessIdTable db
  240         flows = dbFlows db
  241 
  242         rows = V.toList $ V.imap (\idx (_actUuid, prodUuid) ->
  243                 let activity = activities V.! idx
  244                     refProduct = case [ex | ex <- exchanges activity, exchangeIsReference ex] of
  245                         (ex:_) -> case M.lookup (exchangeFlowId ex) flows of
  246                             Just flow -> flowName flow
  247                             Nothing -> T.pack (show prodUuid)
  248                         [] -> T.pack (show prodUuid)
  249                     unit = activityUnit activity
  250                 in escapeCsvField (activityName activity) <> ";" <>
  251                    escapeCsvField (activityLocation activity) <> ";" <>
  252                    escapeCsvField refProduct <> ";" <>
  253                    escapeCsvField unit <> ";" <>
  254                    T.pack (show idx)
  255             ) processIdTable
  256 
  257         header = "activityName;geography;product;unitName;index"
  258         content = T.unlines (header : rows)

  260     TIO.writeFile filePath content
  261     putStrLn $ "Exported " ++ show (length rows) ++ " activities to " ++ filePath

  263 -- | Export ee_index.csv (Elementary Exchanges)
  264 exportEEIndex :: FilePath -> Database -> IO ()

  270                 case M.lookup flowUuid flows of
  271                     Just flow ->
  272                         let category = flowCategory flow
  273                             (compartment, subcompartment) = case T.splitOn "/" category of
  274                                 [] -> ("unspecified", "")

  282                            T.pack (show idx)

  284                         escapeCsvField (T.pack (show flowUuid)) <> ";unknown;;;" <> T.pack (show idx)
  285             ) (V.toList bioFlowUUIDs) ([0..] :: [Int])

  293 -- | Export A_public.csv (Technosphere Matrix)
  294 exportAMatrix :: FilePath -> Database -> IO ()
  295 -- entered 6 timesexportAMatrix filePath db = do
  296     let techTriples = dbTechnosphereTriples db
  297         activityCount = dbActivityCount db
  298 
  299         diagonalEntries = [T.pack $ show i ++ ";" ++ show i ++ ";1.0;;;;;"
  300                           | i <- [0..fromIntegral activityCount - 1 :: Int]]
  301 
  302         offDiagonalRows = U.foldr (\(SparseTriple row col value) acc ->
  303                 let rowStr = T.pack $ show row ++ ";" ++ show col ++ ";" ++
  304                             show (-value) ++ ";;;;;"

  306             ) [] techTriples

  308         header = "row;column;coefficient;uncertainty type;varianceWithPedigreeUncertainty;minValue;mostLikelyValue;maxValue"
  309         allRows = diagonalEntries ++ offDiagonalRows
