   50 -- | Extract matrix debug information from Database
   51 extractMatrixDebugInfo :: Database -> UUID -> Maybe Text -> IO MatrixDebugInfo
   52 extractMatrixDebugInfo database targetUUID flowFilter = do
   53     let activities = dbActivities database
   54         flows = dbFlows database
   55         techTriples = dbTechnosphereTriples database
   56         bioTriples = dbBiosphereTriples database
   57         activityIndexVec = dbActivityIndex database
   58         bioFlowUUIDs = dbBiosphereFlows database
   59         activityCount = dbActivityCount database
   60         bioFlowCount = dbBiosphereCount database
   61 
   62         targetProcessId = case findProcessIdByActivityUUID database targetUUID of
   63             Just pid -> pid
   64             Nothing -> error $ "Activity " <> show targetUUID <> " has no ProcessId in database for debug-matrices"
   65         demandVec = buildDemandVectorFromIndex activityIndexVec targetProcessId
   66         demandList = toList demandVec
   67 
   68         techTriplesInt = [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- U.toList techTriples]
   69         activityCountInt = fromIntegral activityCount
   70 
   71     supplyVec <- solveSparseLinearSystem techTriplesInt activityCountInt demandVec
   72     let supplyList = toList supplyVec
   73 
   74         bioTriplesInt = [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- U.toList bioTriples]
   75         bioFlowCountInt = fromIntegral bioFlowCount
   76         inventoryVec = applySparseMatrix bioTriplesInt bioFlowCountInt supplyVec
   77         inventoryList = toList inventoryVec
   78 
   79         filteredBioTriples = case flowFilter of
   80             Nothing -> bioTriples
   81             Just filterText ->
   82                 let matchingFlowIndices =
   83                         [ idx | (uuid, idx) <- zip (V.toList bioFlowUUIDs) ([0 ..] :: [Int]), Just flow <- [M.lookup uuid flows], T.toLower filterText `T.isInfixOf` T.toLower (flowName flow)
   84                         ]
   85                     matchingFlowIndicesInt32 = map fromIntegral matchingFlowIndices :: [Int32]
   86                  in U.filter (\(SparseTriple row _ _) -> row `elem` matchingFlowIndicesInt32) bioTriples
   87     return MatrixDebugInfo
   88             { mdActivities = activities
   89             , mdFlows = flows
   90             , mdTechTriples = techTriples
   91             , mdBioTriples = filteredBioTriples
   92             , mdActivityIndex = activityIndexVec
   93             , mdBioFlowUUIDs = bioFlowUUIDs
   94             , mdTargetUUID = targetUUID
   95             , mdTargetProcessId = targetProcessId
   96             , mdDatabase = database
   97             , mdSupplyVector = supplyList
   98             , mdDemandVector = demandList
   99             , mdInventoryVector = inventoryList
  100             }

  111 -- | Export supply chain data showing which activities contribute to target
  112 exportSupplyChainData :: FilePath -> MatrixDebugInfo -> IO ()
  113 exportSupplyChainData filePath debugInfo = do
  114     let database = mdDatabase debugInfo
  115         activities = mdActivities debugInfo
  116         activityIndexVec = mdActivityIndex debugInfo
  117         supplyVector = mdSupplyVector debugInfo
  118 
  119         supplyChainRows =
  120             [ csvRow processId activity idx supply
  121             | processId <- [toEnum 0 .. toEnum (V.length activities - 1)]
  122             , let activity = activities V.! fromEnum processId
  123             , let idx = fromIntegral (activityIndexVec V.! fromEnum processId) :: Int
  124             , let supply = if idx < length supplyVector then supplyVector !! idx else 0.0
  125             ]
  126 
  127         csvRow processId activity idx supply =
  128             [ T.unpack (processIdToText database processId)
  129             , T.unpack (activityName activity)
  130             , T.unpack (activityLocation activity)
  131             , show supply
  132             , show idx
  133             ]
  134 
  135         csvHeader = ["activity_id", "activity_name", "location", "supply_amount", "col_idx"]
  136         allRows = csvHeader : supplyChainRows
  137         csvContent = L.intercalate "\n" (map (L.intercalate ",") allRows)
  138 
  139     writeFile filePath csvContent
  140     reportProgress Info $ "Supply chain: " ++ show (length supplyChainRows) ++ " activities → " ++ filePath

  142 -- | Export biosphere matrix contributions
  143 exportBiosphereMatrixData :: FilePath -> MatrixDebugInfo -> IO ()
  144 exportBiosphereMatrixData filePath debugInfo = do
  145     let database = mdDatabase debugInfo
  146         flows = mdFlows debugInfo
  147         activities = mdActivities debugInfo
  148         bioTriples = mdBioTriples debugInfo
  149         bioFlowUUIDs = mdBioFlowUUIDs debugInfo
  150         activityIndex = mdActivityIndex debugInfo
  151         supplyVector = mdSupplyVector debugInfo
  152 
  153         matrixRows =
  154             [ csvRow bioTriple
  155             | bioTriple@(SparseTriple _row _col value) <- U.toList bioTriples
  156             , abs value > 1e-20
  157             ]
  158 
  159         csvRow (SparseTriple row col value) =
  160             let rowInt = fromIntegral row :: Int
  161                 colInt = fromIntegral col :: Int
  162              in [ maybe "unknown" show (getFlowUUID rowInt)
  163                 , maybe "unknown" (T.unpack . flowName) (getFlow rowInt)
  164                 , maybe "unknown" T.unpack (getFlowUnit rowInt)
  165                 , maybe "unknown" (T.unpack . processIdToText database) (getActivityProcessId colInt)
  166                 , maybe "unknown" (T.unpack . activityName) (lookupActivity colInt)
  167                 , show value
  168                 , show (realContribution colInt)
  169                 ]
  170           where
  171             getFlowUUID :: Int -> Maybe UUID
  172             getFlowUUID rowIdx =
  173                 if rowIdx < V.length bioFlowUUIDs
  174                     then Just (bioFlowUUIDs V.! rowIdx)
  175                     else Nothing
  176             getFlow :: Int -> Maybe Flow
  177             getFlow rowIdx = getFlowUUID rowIdx >>= flip M.lookup flows
  178             getFlowUnit :: Int -> Maybe Text
  179             getFlowUnit rowIdx = fmap (getUnitNameForFlow (dbUnits database)) (getFlow rowIdx)
  180             getActivityProcessId :: Int -> Maybe ProcessId
  181             getActivityProcessId colIdx =
  182                 L.find (\pid -> fromIntegral (activityIndex V.! fromEnum pid) == colIdx)
  183                        [toEnum 0 .. toEnum (V.length activities - 1)]
  184             lookupActivity :: Int -> Maybe Activity
  185             lookupActivity colIdx = do
  186                 processId <- getActivityProcessId colIdx
  187                 if fromEnum processId < V.length activities
  188                     then Just (activities V.! fromEnum processId)
  189                     else Nothing
  190             realContribution :: Int -> Double
  191             realContribution colIdx =
  192                 if colIdx < length supplyVector
  193                     then value * (supplyVector !! colIdx)
  194                     else 0.0
  195 
  196         csvHeader = ["flow_id", "flow_name", "unit", "activity_id", "activity_name", "matrix_value", "contribution"]
  197         allRows = csvHeader : matrixRows
  198         csvContent = L.intercalate "\n" (map (L.intercalate ",") allRows)
  199 
  200     writeFile filePath csvContent
  201     reportProgress Info $ "Biosphere matrix: " ++ show (length matrixRows) ++ " entries → " ++ filePath

  213 -- | Escape text for CSV output (semicolon delimiter)
  214 escapeCsvField :: Text -> Text
  215 escapeCsvField text
  216     | T.any (\c -> c == ';' || c == '"' || c == '\n' || c == '\r') text =
  217         "\"" <> T.replace "\"" "\"\"" text <> "\""
  218     | otherwise = text

  220 -- | Export ie_index.csv (Intermediate Exchanges)
  221 exportIEIndex :: FilePath -> Database -> IO ()
  222 exportIEIndex filePath db = do
  223     let activities = dbActivities db
  224         processIdTable = dbProcessIdTable db
  225         flows = dbFlows db
  226 
  227         rows = V.toList $ V.imap (\idx (_actUuid, prodUuid) ->
  228                 let activity = activities V.! idx
  229                     refProduct = case [ex | ex <- exchanges activity, exchangeIsReference ex] of
  230                         (ex:_) -> case M.lookup (exchangeFlowId ex) flows of
  231                             Just flow -> flowName flow
  232                             Nothing -> T.pack (show prodUuid)
  233                         [] -> T.pack (show prodUuid)
  234                     unit = activityUnit activity
  235                 in escapeCsvField (activityName activity) <> ";" <>
  236                    escapeCsvField (activityLocation activity) <> ";" <>
  237                    escapeCsvField refProduct <> ";" <>
  238                    escapeCsvField unit <> ";" <>
  239                    T.pack (show idx)
  240             ) processIdTable
  241 
  242         header = "activityName;geography;product;unitName;index"
  243         content = T.unlines (header : rows)
  244 
  245     TIO.writeFile filePath content
  246     reportProgress Info $ "ie_index: " ++ show (length rows) ++ " activities → " ++ filePath

  248 -- | Export ee_index.csv (Elementary Exchanges)
  249 exportEEIndex :: FilePath -> Database -> IO ()
  250 exportEEIndex filePath db = do
  251     let bioFlowUUIDs = dbBiosphereFlows db
  252         flows = dbFlows db
  253 
  254         rows = zipWith (\flowUuid idx ->
  255                 case M.lookup flowUuid flows of
  256                     Just flow ->
  257                         let category = flowCategory flow
  258                             (compartment, subcompartment) = case T.splitOn "/" category of
  259                                 [] -> ("unspecified", "")
  260                                 [c] -> (T.strip c, "")
  261                                 (c:s:_) -> (T.strip c, T.strip s)
  262                             unit = getUnitNameForFlow (dbUnits db) flow
  263                         in escapeCsvField (flowName flow) <> ";" <>
  264                            escapeCsvField compartment <> ";" <>
  265                            escapeCsvField subcompartment <> ";" <>
  266                            escapeCsvField unit <> ";" <>
  267                            T.pack (show idx)
  268                     Nothing ->
  269                         escapeCsvField (T.pack (show flowUuid)) <> ";unknown;;;" <> T.pack (show idx)
  270             ) (V.toList bioFlowUUIDs) ([0..] :: [Int])
  271 
  272         header = "name;compartment;subcompartment;unitName;index"
  273         content = T.unlines (header : rows)
  274 
  275     TIO.writeFile filePath content
  276     reportProgress Info $ "ee_index: " ++ show (length rows) ++ " biosphere flows → " ++ filePath

  278 -- | Export A_public.csv (Technosphere Matrix)
  279 exportAMatrix :: FilePath -> Database -> IO ()
  280 exportAMatrix filePath db = do
  281     let techTriples = dbTechnosphereTriples db
  282         activityCount = dbActivityCount db
  283         activities = dbActivities db
  284 
  285         -- Waste activities have their reference product stored as outputGroup=0 with a
  286         -- NEGATIVE amount (e.g., hard coal ash = -1.0 kg). These use -1.0 on the diagonal
  287         -- in the (I-A) convention. Production activities use 1.0.
  288         diagValue i =
  289             let act = activities V.! i
  290             in if any (\ex -> exchangeIsReference ex && exchangeAmount ex < 0) (exchanges act)
  291                then "-1.0" else "1.0"
  292 
  293         diagonalEntries = [T.pack (show i ++ ";" ++ show i ++ ";" ++ diagValue i) <> emptyCsvUncertainty
  294                           | i <- [0..fromIntegral activityCount - 1 :: Int]]
  295 
  296         offDiagonalRows = U.foldr (\(SparseTriple row col value) acc ->
  297                 let rowStr = T.pack (show row ++ ";" ++ show col ++ ";" ++ show (-value))
  298                              <> emptyCsvUncertainty
  299                 in rowStr : acc
  300             ) [] techTriples
  301 
  302         header = "row;column;coefficient;uncertainty type;varianceWithPedigreeUncertainty;minValue;mostLikelyValue;maxValue"
  303         allRows = diagonalEntries ++ offDiagonalRows
  304         content = T.unlines (header : allRows)
  305 
  306     TIO.writeFile filePath content
  307     reportProgress Info $ "A_public: " ++ show (length diagonalEntries) ++ " diagonal + " ++
  308                           show (U.length techTriples) ++ " off-diagonal entries → " ++ filePath
