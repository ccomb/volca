   21 -- | Normalize CAS number by stripping leading zeros from first segment.

  435     buildResult :: ParseState -> ProcessId -> Either String (Activity, [Flow], [Unit])
  436     buildResult st _pid =
  437         let name = case psActivityName st of
  438                 Just n -> n
  439                 Nothing -> "Unknown Activity"
  440             location = case psLocation st of
  441                 Just loc -> loc

  448             activity = Activity name description M.empty (psClassifications st) location refUnit (reverse $ psExchanges st) M.empty M.empty
  449             flows = reverse (psFlows st)

  468 {- | Apply cut-off strategy
  469 1. Remove zero-amount production exchanges (co-products)
  470 2. Assign single non-zero product as reference product
  471 3. Ensure single-output process structure
  472 4. VALIDATION: Fail if no reference product can be established
  473 -}
  474 applyCutoffStrategy :: Activity -> Either String Activity
  475 -- entered 106 timesapplyCutoffStrategy activity =
  476     let filteredExchanges = removeZeroAmountCoproducts (exchanges activity)
  477         updatedActivity = activity{exchanges = filteredExchanges}
  478         finalActivity =
  479             if hasReferenceProduct updatedActivity
  480                 then updatedActivity
  481                 else assignSingleProductAsReference updatedActivity
  482      in if hasReferenceProduct finalActivity
  483            then Right finalActivity
  484            else Left $ "Activity has no reference product: " ++ T.unpack (activityName activity)

  499 -- | Assign single product as reference product

  501 -- never enteredassignSingleProductAsReference activity =

  503         nonZeroProduction = [ex | ex <- productionExchanges, exchangeAmount ex /= 0.0]
