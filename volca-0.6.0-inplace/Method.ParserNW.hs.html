   40 -- | Pure parser. Takes a fallback name (used if no comment header found).
   41 parseNormWeightCSVBytes :: Text -> BS.ByteString -> Either String NormWeightSet
   42 -- entered 4 timesparseNormWeightCSVBytes fallbackName bs =
   43     let allLines = BC.lines bs
   44         (comments, rest) = span (\l -> BC.isPrefixOf "#" l || BS.null l) allLines
   45         name = extractName comments fallbackName
   46         delim = detectDelimiter rest
   47     in case dropHeader rest of
   48         [] -> Left "NW CSV: no data rows after header"
   49         rows ->
   50             let parsed = [ parseRow delim l | l <- rows, not (BS.null (BC.strip l)) ]
   51                 normMap = M.fromList [(cat, n) | (cat, n, _) <- parsed, not (T.null cat)]
   52                 weightMap = M.fromList [(cat, w) | (cat, _, w) <- parsed, not (T.null cat)]
   53             in if M.null normMap && M.null weightMap
   54                then Left "NW CSV: no valid rows parsed"
   55                else Right $ NormWeightSet name normMap weightMap
