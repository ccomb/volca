  294     -- Build exchange, flow, and unit from exchange data
  295     -- activityLoc is the activity's location for fallback
  296     buildExchange :: Int -> Maybe Text -> ExchangeData -> (Exchange, Flow, Unit)
  297     buildExchange datasetNum activityLoc edata =
  298         let flowId = generateFlowUUID datasetNum (exNumber edata) (exName edata) (exCategory edata)
  299             unitId = generateUnitUUID (exUnit edata)
  300 
  301             -- Determine flow type from input/output groups
  302             -- EcoSpold1 groups:
  303             -- Input: 1-3 = technosphere, 4 = resource (biosphere)
  304             -- Output: 0 = reference product, 1-3 = byproduct/co-product, 4 = emission (biosphere)
  305             inputGroup = exInputGroup edata
  306             outputGroup = exOutputGroup edata
  307 
  308             isBiosphere = inputGroup == "4" || outputGroup == "4"
  309             isInput = not (T.null inputGroup)
  310             isReferenceProduct = outputGroup == "0"
  311 
  312             -- Build category string
  313             category = if T.null (exSubCategory edata)
  314                        then exCategory edata
  315                        else exCategory edata <> "/" <> exSubCategory edata
  316 
  317             flowType = if isBiosphere then Biosphere else Technosphere
  318 
  319             -- Exchange location: use exchange's own location
  320             -- For technosphere: leave empty if not specified, so Loader can use name-only lookup
  321             -- For biosphere: fall back to activity location (biosphere flows don't need supplier linking)
  322             exchangeLocation = if T.null (exLocation edata)
  323                                then if isBiosphere
  324                                     then case activityLoc of
  325                                              Just loc -> loc
  326                                              Nothing -> ""
  327                                     else ""  -- Technosphere: leave empty for name-only lookup in Loader
  328                                else exLocation edata
  329 
  330             -- Set activityLinkId to nil - will be resolved later in Loader using
  331             -- (flowName, exchangeLocation) lookup against supplier activities
  332             exchange = if isBiosphere
  333                 then BiosphereExchange flowId (exMeanValue edata) unitId (inputGroup == "4") exchangeLocation
  334                 else TechnosphereExchange
  335                         flowId
  336                         (exMeanValue edata)
  337                         unitId
  338                         isInput
  339                         isReferenceProduct
  340                         UUID.nil  -- Will be resolved in Loader.fixEcoSpold1ActivityLinks
  341                         Nothing
  342                         exchangeLocation
  343 
  344             cas  = if T.null (exCASNumber edata) then Nothing else Just (exCASNumber edata)
  345             flow = Flow flowId (exName edata) category Nothing unitId flowType M.empty cas Nothing
  346 
  347             unit = Unit unitId (exUnit edata) (exUnit edata) ""
  348 
  349         in (exchange, flow, unit)

  351     -- Build final result
  352     buildResult :: ParseState -> Either String (Activity, [Flow], [Unit], Int, M.Map UUID Int)
  353     buildResult st =
  354         let name = case psActivityName st of
  355                 Just n -> n
  356                 Nothing -> "Unknown Activity"
  357             location = case psLocation st of
  358                 Just loc -> loc
  359                 Nothing -> "GLO"
  360             refUnit = case psRefUnit st of
  361                 Just u -> u
  362                 Nothing -> "UNKNOWN_UNIT"
  363             description = reverse (psDescription st)
  364             classifications = M.fromList $ filter (not . T.null . snd)
  365                 [("Category", psActivityCategory st), ("SubCategory", psActivitySubCategory st)]
  366             activity = Activity name description M.empty classifications location refUnit (reverse $ psExchanges st) M.empty M.empty
  367             flows = reverse (psFlows st)
  368             units = reverse (psUnits st)
  369         in case applyCutoffStrategy activity of
  370             Right act -> Right (act, flows, units, psDatasetNumber st, psSupplierLinks st)
  371             Left err  -> Left err

  604     -- Build exchange, flow, and unit from exchange data (same logic as parseWithXeno)
  605     buildExchangeForAll :: Int -> Maybe Text -> ExchangeData -> (Exchange, Flow, Unit)
  606     buildExchangeForAll datasetNum activityLoc edata =
  607         let flowId = generateFlowUUID datasetNum (exNumber edata) (exName edata) (exCategory edata)
  608             unitId = generateUnitUUID (exUnit edata)
  609             inputGroup = exInputGroup edata
  610             outputGroup = exOutputGroup edata
  611             isBiosphere = inputGroup == "4" || outputGroup == "4"
  612             isInput = not (T.null inputGroup)
  613             isReferenceProduct = outputGroup == "0"
  614             category = if T.null (exSubCategory edata)
  615                        then exCategory edata
  616                        else exCategory edata <> "/" <> exSubCategory edata
  617             flowType = if isBiosphere then Biosphere else Technosphere
  618             exchangeLocation = if T.null (exLocation edata)
  619                                then if isBiosphere
  620                                     then case activityLoc of
  621                                              Just loc -> loc
  622                                              Nothing -> ""
  623                                     else ""
  624                                else exLocation edata
  625             exchange = if isBiosphere
  626                 then BiosphereExchange flowId (exMeanValue edata) unitId (inputGroup == "4") exchangeLocation
  627                 else TechnosphereExchange
  628                         flowId
  629                         (exMeanValue edata)
  630                         unitId
  631                         isInput
  632                         isReferenceProduct
  633                         UUID.nil
  634                         Nothing
  635                         exchangeLocation
  636             cas  = if T.null (exCASNumber edata) then Nothing else Just (exCASNumber edata)
  637             flow = Flow flowId (exName edata) category Nothing unitId flowType M.empty cas Nothing
  638             unit = Unit unitId (exUnit edata) (exUnit edata) ""
  639         in (exchange, flow, unit)

  641     -- Build final result for a single dataset
  642     buildResultForAll :: ParseState -> Either String (Activity, [Flow], [Unit], Int, M.Map UUID Int)
  643     buildResultForAll st =
  644         let name = case psActivityName st of
  645                 Just n -> n
  646                 Nothing -> "Unknown Activity"
  647             location = case psLocation st of
  648                 Just loc -> loc
  649                 Nothing -> "GLO"
  650             refUnit = case psRefUnit st of
  651                 Just u -> u
  652                 Nothing -> "UNKNOWN_UNIT"
  653             description = reverse (psDescription st)
  654             classifications = M.fromList $ filter (not . T.null . snd)
  655                 [("Category", psActivityCategory st), ("SubCategory", psActivitySubCategory st)]
  656             activity = Activity name description M.empty classifications location refUnit (reverse $ psExchanges st) M.empty M.empty
  657             flows = reverse (psFlows st)
  658             units = reverse (psUnits st)
  659         in case applyCutoffStrategy activity of
  660             Right act -> Right (act, flows, units, psDatasetNumber st, psSupplierLinks st)
  661             Left err  -> Left err
