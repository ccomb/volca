    4 -- | Plugin Configuration
    5 --
    6 -- Parses @[[plugin]]@ TOML sections and merges external plugin declarations
    7 -- with the built-in registry. Supports:
    8 --
    9 -- * Adding external plugins (any type, with executable path)
   10 -- * Overriding built-in plugin priorities (match by name)
   11 -- * Disabling built-in plugins (@enabled = false@)

   14     , PluginType(..)

   35 -- | Plugin type tag (matches TOML @type@ field)
   36 data PluginType
   37     = PTImporter
   38     | PTExporter
   39     | PTSearcher
   40     | PTMapper
   41     | PTTransform
   42     | PTValidator
   43     | PTAnalyzer
   44     | PTReporter
   45     deriving (-- never entered-- never entered-- never enteredShow, -- never entered-- never enteredEq)
   46 
   47 instance DecodeTOML PluginType where
   48     -- never enteredtomlDecoder = do
   49         t <- tomlDecoder :: Decoder Text
   50         case T.toLower t of
   51             "importer"  -> pure PTImporter
   52             "exporter"  -> pure PTExporter
   53             "searcher"  -> pure PTSearcher
   54             "mapper"    -> pure PTMapper
   55             "transform" -> pure PTTransform
   56             "validator" -> pure PTValidator
   57             "analyzer"  -> pure PTAnalyzer
   58             "reporter"  -> pure PTReporter
   59             other       -> fail $ "Unknown plugin type: " <> T.unpack other

   61 -- | A single @[[plugin]]@ TOML entry
   62 data PluginConfig = PluginConfig
   63     { -- entered 5 timespcName     :: !Text
   64     , -- entered 2 timespcType     :: !PluginType
   65     , -- entered oncepcPath     :: !(Maybe FilePath)    -- Nothing for built-in overrides
   66     , -- entered 2 timespcEnabled  :: !Bool
   67     , -- entered oncepcPriority :: !(Maybe Int)
   68     , -- never enteredpcFormatId :: !(Maybe Text)        -- For exporter/reporter
   69     , -- never enteredpcPhase    :: !(Maybe Text)        -- For validator: "pre-compute" or "post-compute"
   70     , -- never enteredpcMimeType :: !(Maybe Text)        -- For reporter
   71     } deriving (-- never entered-- never entered-- never enteredShow, -- never entered-- never enteredEq)
   72 
   73 instance DecodeTOML PluginConfig where
   74     -- never enteredtomlDecoder = do
   75         pcName     <- getField "name"
   76         pcType     <- getField "type"
   77         pcPath     <- getFieldOpt "path"
   78         pcEnabled  <- fromMaybe True <$> getFieldOpt "enabled"
   79         pcPriority <- getFieldOpt "priority"
   80         pcFormatId <- getFieldOpt "format-id"
   81         pcPhase    <- getFieldOpt "phase"
   82         pcMimeType <- getFieldOpt "mime-type"
   83         pure PluginConfig{..}

   85 -- | Build a registry from built-in defaults + TOML plugin configs.
   86 --
   87 -- Processing order:
   88 -- 1. Start with defaultRegistry
   89 -- 2. For each TOML entry:
   90 --    - If enabled=false and matches a built-in by name: remove it
   91 --    - If matches a built-in by name: override priority (or other fields)
   92 --    - If has a path: add as external plugin
   93 -- 3. Sort priority-based lists
   94 buildRegistry :: [PluginConfig] -> PluginRegistry
   95 -- entered 3 timesbuildRegistry configs =
   96     let base = defaultRegistry
   97         -- Process each config entry
   98         registry = foldl applyConfig base configs
   99     in registry
  100         { prMappers    = sortOn mhPriority (prMappers registry)
  101         , prSearchers  = sortOn shPriority (prSearchers registry)
  102         , prTransforms = sortOn thPriority (prTransforms registry)
  103         }
  104 
  105 applyConfig :: PluginRegistry -> PluginConfig -> PluginRegistry
  106 -- entered 2 timesapplyConfig reg pc
  107     | not (pcEnabled pc) = disablePlugin reg pc
  108     | otherwise = case pcType pc of
  109         PTMapper   -> applyMapper reg pc
  110         PTReporter -> applyReporter reg pc
  111         PTExporter -> applyExporter reg pc
  112         PTAnalyzer -> applyAnalyzer reg pc
  113         PTTransform -> applyTransform reg pc
  114         PTValidator -> applyValidator reg pc
  115         PTImporter  -> applyImporter reg pc
  116         PTSearcher  -> applySearcher reg pc
  117 
  118 -- | Remove a plugin by name
  119 disablePlugin :: PluginRegistry -> PluginConfig -> PluginRegistry
  120 -- entered oncedisablePlugin reg pc = case pcType pc of
  121     PTMapper   -> reg { prMappers   = filter ((/= pcName pc) . mhName) (prMappers reg) }
  122     PTReporter -> reg { prReporters = M.filter ((/= pcName pc) . rhName) (prReporters reg) }
  123     PTExporter -> reg { prExporters = M.filter ((/= pcName pc) . ehName) (prExporters reg) }
  124     PTAnalyzer -> reg { prAnalyzers = M.filter ((/= pcName pc) . ahName) (prAnalyzers reg) }
  125     PTSearcher  -> reg { prSearchers  = filter ((/= pcName pc) . shName) (prSearchers reg) }
  126     PTTransform -> reg { prTransforms = filter ((/= pcName pc) . thName) (prTransforms reg) }
  127     PTValidator -> reg { prValidators = filter ((/= pcName pc) . vhName) (prValidators reg) }
  128     PTImporter  -> reg { prImporters  = filter ((/= pcName pc) . ihName) (prImporters reg) }

  130 -- | Apply mapper config: override priority of existing, or add external
  131 applyMapper :: PluginRegistry -> PluginConfig -> PluginRegistry
  132 -- entered onceapplyMapper reg pc =
  133     case pcPath pc of
  134         Nothing ->
  135             -- Override priority of existing built-in
  136             let mappers' = map (\m -> if mhName m == pcName pc
  137                     then m { mhPriority = fromMaybe (mhPriority m) (pcPriority pc) }
  138                     else m) (prMappers reg)
  139             in reg { prMappers = mappers' }
  140         Just path ->
  141             -- Add external mapper
  142             let priority = fromMaybe 50 (pcPriority pc)
  143                 handle = makeExternalMapper (pcName pc) path priority
  144             in reg { prMappers = handle : prMappers reg }
  145 
  146 -- | Apply reporter config: add external reporter
  147 applyReporter :: PluginRegistry -> PluginConfig -> PluginRegistry
  148 -- never enteredapplyReporter reg pc = case pcPath pc of
  149     Nothing -> reg  -- Can't override built-in reporter fields (no meaningful overrides)
  150     Just path ->
  151         let fmtId = fromMaybe (pcName pc) (pcFormatId pc)
  152             mime  = fromMaybe "application/octet-stream" (pcMimeType pc)
  153             handle = makeExternalReporter (pcName pc) path fmtId mime
  154         in reg { prReporters = M.insert fmtId handle (prReporters reg) }
  155 
  156 -- | Apply exporter config: add external exporter

  222 makeExternalMapper :: Text -> FilePath -> Int -> MapperHandle
  223 -- never enteredmakeExternalMapper name path priority = MapperHandle
  224     { mhName     = name
  225     , mhBackend  = External path
  226     , mhPriority = priority
  227     , mhMatch    = \_ query -> do
  228         let req = case query of
  229                 MatchCF cf -> object ["action" .= ("match" :: Text), "query" .= object
  230                     [ "name" .= mcfFlowName cf
  231                     , "uuid" .= show (mcfFlowRef cf)
  232                     , "cas"  .= mcfCAS cf
  233                     ]]
  234                 MatchMaterial mat mCas -> object ["action" .= ("match" :: Text), "query" .= object
  235                     [ "name" .= mat, "cas" .= mCas ]]
  236                 MatchSupplier sName sLoc -> object ["action" .= ("match" :: Text), "query" .= object
  237                     [ "supplier" .= sName, "location" .= sLoc ]]
  238         result <- callPlugin path req
  239         case result of
  240             Right (Object obj) -> case (lookupKey "uuid" obj, lookupKey "strategy" obj) of
  241                 (Just (String u), Just (String s)) -> case UUID.fromText u of
  242                     Just uid -> pure $ Just $ MapResult uid s 0.7
  243                     Nothing  -> pure Nothing
  244                 _ -> pure Nothing
  245             _ -> pure Nothing
  246     }

  248 makeExternalReporter :: Text -> FilePath -> Text -> Text -> ReportHandle
  249 -- never enteredmakeExternalReporter name path fmtId mime = ReportHandle
  250     { rhName     = name
  251     , rhBackend  = External path
  252     , rhFormatId = fmtId
  253     , rhMimeType = mime
  254     , rhReport   = \value -> do
  255         result <- callPlugin path (object ["action" .= ("report" :: Text), "data" .= value])
  256         case result of
  257             Right (String t) -> pure $ BSL8.pack (T.unpack t)
  258             Right other      -> pure $ encode other
  259             Left err         -> pure $ BSL8.pack $ "Plugin error: " <> T.unpack err
  260     }
