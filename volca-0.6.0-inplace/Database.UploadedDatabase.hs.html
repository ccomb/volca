  103 -- | Parse meta.toml content
  104 -- Simple key=value parser (not a full TOML parser)
  105 parseMetaToml :: Text -> Maybe UploadMeta
  106 -- entered 14 timesparseMetaToml content = do
  107     let lines' = map T.strip $ T.lines content
  108         kvPairs = [(T.strip k, v) | line <- lines'
  109                          , not (T.null line)
  110                          , not (T.isPrefixOf "#" line)
  111                          , let (k, rest) = T.breakOn "=" line
  112                          , not (T.null rest)
  113                          , let v = T.strip $ T.drop 1 rest
  114                          ]
  115         getValue key = lookup key kvPairs
  116         unquote t = T.dropAround (== '"') t
  117 
  118     version <- getValue "version" >>= readMaybe . T.unpack
  119     displayName <- unquote <$> getValue "displayName"
  120     let description = unquote <$> getValue "description"
  121     format <- getValue "format" >>= parseFormat . unquote
  122     dataPath <- T.unpack . unquote <$> getValue "dataPath"
  123 
  124     return UploadMeta
  125         { umVersion = version

  127         , umDescription = description

  140 -- | Format meta.toml content
  141 formatMetaToml :: UploadMeta -> Text
  142 -- entered 18 timesformatMetaToml UploadMeta{..} = T.unlines $
  143     [ "version = " <> T.pack (show umVersion)
  144     , "displayName = " <> quote umDisplayName

  166 -- | Scan a directory for subdirectories with meta.toml
  167 scanUploadsIn :: FilePath -> IO [(Text, FilePath, UploadMeta)]
  168 -- never enteredscanUploadsIn dir = do
  169     exists <- doesDirectoryExist dir
  170     if not exists
  171         then return []
  172         else do
  173             entries <- listDirectory dir
  174             let fullPaths = [(T.pack entry, dir </> entry) | entry <- entries]
  175             dirsOnly <- filterM (doesDirectoryExist . snd) fullPaths
  176             results <- forM dirsOnly $ \(slug, dirPath) -> do
  177                 maybeMeta <- readUploadMeta dirPath
  178                 return $ case maybeMeta of
  179                     Just meta -> Just (slug, dirPath, meta)
  180                     Nothing -> Nothing
  181             return [r | Just r <- results]
