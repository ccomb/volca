   80 -- | Read meta.toml from an upload directory
   81 -- Returns Nothing if file doesn't exist or can't be parsed
   82 readUploadMeta :: FilePath -> IO (Maybe UploadMeta)
   83 -- never enteredreadUploadMeta uploadDir = do
   84     let metaPath = uploadDir </> metaFileName
   85     exists <- doesFileExist metaPath
   86     if not exists
   87         then return Nothing
   88         else do
   89             result <- try $ TIO.readFile metaPath
   90             case result of
   91                 Left (_ :: SomeException) -> return Nothing
   92                 Right content -> return $ parseMetaToml content
   93 
   94 -- | Write meta.toml to an upload directory
   95 writeUploadMeta :: FilePath -> UploadMeta -> IO ()
   96 -- never enteredwriteUploadMeta uploadDir meta = do
   97     let metaPath = uploadDir </> metaFileName
   98     TIO.writeFile metaPath (formatMetaToml meta)

  100 -- | Parse meta.toml content
  101 -- Simple key=value parser (not a full TOML parser)
  102 parseMetaToml :: Text -> Maybe UploadMeta
  103 -- never enteredparseMetaToml content = do
  104     let lines' = map T.strip $ T.lines content
  105         kvPairs = [(T.strip k, v) | line <- lines'
  106                          , not (T.null line)
  107                          , not (T.isPrefixOf "#" line)
  108                          , let (k, rest) = T.breakOn "=" line
  109                          , not (T.null rest)
  110                          , let v = T.strip $ T.drop 1 rest
  111                          ]
  112         getValue key = lookup key kvPairs
  113         unquote t = T.dropAround (== '"') t
  114 
  115     version <- getValue "version" >>= readMaybe . T.unpack
  116     displayName <- unquote <$> getValue "displayName"
  117     let description = unquote <$> getValue "description"
  118     format <- getValue "format" >>= parseFormat . unquote
  119     dataPath <- T.unpack . unquote <$> getValue "dataPath"
  120 
  121     return UploadMeta
  122         { umVersion = version
  123         , umDisplayName = displayName
  124         , umDescription = description
  125         , umFormat = format
  126         , umDataPath = dataPath
  127         }

  129 -- | Parse format string to DatabaseFormat
  130 parseFormat :: Text -> Maybe DatabaseFormat
  131 -- never enteredparseFormat "ecospold2" = Just EcoSpold2
  132 parseFormat "ecospold1" = Just EcoSpold1
  133 parseFormat "simapro" = Just SimaProCSV
  134 parseFormat "ilcd" = Just ILCDProcess
  135 parseFormat _ = Just UnknownFormat
  136 
  137 -- | Format meta.toml content
  138 formatMetaToml :: UploadMeta -> Text
  139 -- never enteredformatMetaToml UploadMeta{..} = T.unlines $
  140     [ "version = " <> T.pack (show umVersion)
  141     , "displayName = " <> quote umDisplayName
  142     ] ++
  143     maybe [] (\d -> ["description = " <> quote d]) umDescription ++
  144     [ "format = " <> quote (formatToText umFormat)
  145     , "dataPath = " <> quote (T.pack umDataPath)
  146     ]
  147   where
  148     quote t = "\"" <> escapeToml t <> "\""
  149 
  150     escapeToml = T.concatMap escape
  151       where
  152         escape '"' = "\\\""
  153         escape '\\' = "\\\\"
  154         escape '\n' = "\\n"
  155         escape c = T.singleton c
  156 
  157     formatToText EcoSpold2 = "ecospold2"
  158     formatToText EcoSpold1 = "ecospold1"
  159     formatToText SimaProCSV = "simapro"
  160     formatToText ILCDProcess = "ilcd"
  161     formatToText UnknownFormat = "unknown"
  162 
  163 -- | Scan a directory for subdirectories with meta.toml
  164 scanUploadsIn :: FilePath -> IO [(Text, FilePath, UploadMeta)]
  165 -- never enteredscanUploadsIn dir = do
  166     exists <- doesDirectoryExist dir
  167     if not exists
  168         then return []
  169         else do
  170             entries <- listDirectory dir
  171             let fullPaths = [(T.pack entry, dir </> entry) | entry <- entries]
  172             dirsOnly <- filterM (doesDirectoryExist . snd) fullPaths
  173             results <- forM dirsOnly $ \(slug, dirPath) -> do
  174                 maybeMeta <- readUploadMeta dirPath
  175                 return $ case maybeMeta of
  176                     Just meta -> Just (slug, dirPath, meta)
  177                     Nothing -> Nothing
  178             return [r | Just r <- results]

  180 -- | Discover all uploaded databases by scanning the uploads directory
  181 -- Scans ./uploads/databases/ first, then legacy ./uploads/ for backward compat
  182 discoverUploadedDatabases :: IO [(Text, FilePath, UploadMeta)]
  183 -- never entereddiscoverUploadedDatabases = do
  184     dbDir <- getDatabaseUploadsDir
  185     newResults <- scanUploadsIn dbDir
  186     if not (null newResults)
  187         then return newResults
  188         else do
  189             -- Legacy fallback: scan ./uploads/ directly
  190             base <- getDataDir
  191             let legacyDir = base </> "uploads"
  192             scanUploadsIn legacyDir
  193 
  194 -- | Discover all uploaded methods by scanning the methods upload directory
  195 discoverUploadedMethods :: IO [(Text, FilePath, UploadMeta)]
  196 -- never entereddiscoverUploadedMethods = do
  197     methodDir <- getMethodUploadsDir
  198     scanUploadsIn methodDir
