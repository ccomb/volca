  117 -- | Build a CompartmentMap from CSV content.
  118 -- CSV columns: source_medium, source_sub, source_qualifier, target_medium, target_sub, target_qualifier
  119 buildCompartmentMapFromCSV :: BL.ByteString -> Either String CompartmentMap
  120 -- never enteredbuildCompartmentMapFromCSV csvData =
  121     case decode HasHeader csvData of
  122         Left err -> Left $ "CSV parse error: " <> err
  123         Right rows ->
  124             let entries = V.toList (rows :: V.Vector (Text, Text, Text, Text, Text, Text))
  125                 pairs = [(( T.toLower (T.strip sm)
  126                           , T.toLower (T.strip ss)
  127                           , T.toLower (T.strip sq))
  128                          , Compartment (T.strip tm) (T.strip ts) (T.strip tq))
  129                         | (sm, ss, sq, tm, ts, tq) <- entries]
  130             in Right $ M.fromList pairs
