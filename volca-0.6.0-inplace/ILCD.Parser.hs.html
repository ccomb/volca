   54 -- | Parse an ILCD directory into a SimpleDatabase.
   55 -- Expects subdirectories: processes/, flows/, flowproperties/, unitgroups/
   56 parseILCDDirectory :: FilePath -> IO (Either Text SimpleDatabase)
   57 parseILCDDirectory dir = do
   58     reportProgress Info $ "Loading ILCD database from: " ++ dir
   59 
   60     -- Step 1: Parse unit groups and flow properties (small, sequential)
   61     unitGroupMap <- parseUnitGroups (dir </> "unitgroups")
   62     flowPropMap  <- parseFlowProperties (dir </> "flowproperties")
   63     reportProgress Info $ printf "Parsed %d unit groups, %d flow properties" (M.size unitGroupMap) (M.size flowPropMap)
   64 
   65     -- Step 2: Parse flows (reuse FlowResolver, parallel + cached)
   66     flowInfoMap <- parseFlowDirectory (dir </> "flows")
   67     reportProgress Info $ printf "Parsed %d flows" (M.size flowInfoMap)
   68 
   69     -- Step 3: Build FlowDB and UnitDB from parsed data
   70     let (flowDB, unitDB) = buildFlowAndUnitDB flowInfoMap flowPropMap unitGroupMap
   71 
   72     -- Step 4: Parse process XMLs in parallel
   73     processFiles <- listXMLFiles (dir </> "processes")
   74     reportProgress Info $ printf "Parsing %d ILCD process files..." (length processFiles)
   75     rawProcesses <- parseProcessFilesParallel processFiles
   76 
   77     reportProgress Info $ printf "Parsed %d processes, building activity map..." (length rawProcesses)
   78 
   79     -- Step 5: Build ActivityMap
   80     let activityMap = buildActivityMap flowInfoMap flowDB unitDB rawProcesses
   81 
   82     -- Step 6: Fix supplier links (name-based, like SimaPro)
   83     let simpleDb = SimpleDatabase activityMap flowDB unitDB
   84     fixedDb <- fixILCDActivityLinks simpleDb
   85     reportProgress Info $ printf "ILCD database loaded: %d activities, %d flows, %d units"
   86         (M.size $ sdbActivities fixedDb) (M.size $ sdbFlows fixedDb) (M.size $ sdbUnits fixedDb)
   87     return $ Right fixedDb
   88 
   89 -- | List XML files in a directory
   90 listXMLFiles :: FilePath -> IO [FilePath]
   91 listXMLFiles d = do
   92     fs <- listDirectory d
   93     return [d </> f | f <- fs, map toLower (takeExtension f) == ".xml"]

   99 parseUnitGroups :: FilePath -> IO (M.Map UUID (Text, Int))
  100 parseUnitGroups dir = do

  105 data UGState = UGState
  106     { ugUUID      :: !Text
  107     , ugRefIdx    :: !Int
  108     , ugUnits     :: ![(Int, Text)]
  109     , ugCurId     :: !Int
  110     , ugCurName   :: !Text
  111     , ugInUnit    :: !Bool
  112     , ugTextAccum :: ![BS.ByteString]
  113     }
  114 
  115 parseUnitGroupXML :: BS.ByteString -> Maybe (UUID, (Text, Int))
  116 parseUnitGroupXML bytes =
  117     case X.fold openTag attr endOpen txt closeTag cdata (UGState "" 0 [] (-1) "" False []) bytes of
  118         Left _  -> Nothing
  119         Right s -> do
  120             uuid <- UUID.fromText (ugUUID s)
  121             case lookup (ugRefIdx s) (ugUnits s) of
  122                 Just name -> Just (uuid, (name, ugRefIdx s))
  123                 Nothing   -> Nothing
  124   where
  125     openTag s tag
  126         | isElement tag "unit" = s { ugInUnit = True, ugTextAccum = [], ugCurId = -1, ugCurName = "" }
  127         | otherwise = s { ugTextAccum = [] }
  128 
  129     attr s name value
  130         | isElement name "dataSetInternalID" && ugInUnit s =
  131             case TR.decimal (bsToText value) of { Right (n, _) -> s { ugCurId = n }; Left _ -> s }
  132         | otherwise = s
  133 
  134     endOpen s _ = s
  135 
  136     txt s content =
  137         let trimmed = BS.dropWhile (== 32) $ BS.dropWhileEnd (== 32) content
  138         in if BS.null trimmed then s else s { ugTextAccum = trimmed : ugTextAccum s }
  139 
  140     closeTag s tag
  141         | isElement tag "UUID" && T.null (ugUUID s) =
  142             s { ugUUID = accum s, ugTextAccum = [] }
  143         | isElement tag "referenceToReferenceUnit" =
  144             case TR.decimal (accum s) of
  145                 Right (n, _) -> s { ugRefIdx = n, ugTextAccum = [] }
  146                 Left _       -> s { ugTextAccum = [] }
  147         | isElement tag "name" && ugInUnit s =
  148             s { ugCurName = accum s, ugTextAccum = [] }
  149         | isElement tag "unit" =
  150             s { ugInUnit = False, ugTextAccum = []
  151               , ugUnits = (ugCurId s, ugCurName s) : ugUnits s }
  152         | otherwise = s { ugTextAccum = [] }
  153 
  154     cdata = txt
  155     accum s = T.strip $ T.concat $ reverse $ map bsToText (ugTextAccum s)

  167 data FPState = FPState
  168     { fpUUID         :: !Text
  169     , fpUnitGroupRef :: !Text
  170     , fpInQuantRef   :: !Bool
  171     , fpTextAccum    :: ![BS.ByteString]
  172     }
  173 
  174 parseFlowPropertyXML :: BS.ByteString -> Maybe (UUID, UUID)
  175 parseFlowPropertyXML bytes =
  176     case X.fold openTag attr endOpen txt closeTag cdata (FPState "" "" False []) bytes of
  177         Left _  -> Nothing
  178         Right s -> (,) <$> UUID.fromText (fpUUID s) <*> UUID.fromText (fpUnitGroupRef s)
  179   where
  180     openTag s tag
  181         | isElement tag "quantitativeReference" = s { fpInQuantRef = True, fpTextAccum = [] }
  182         | otherwise = s { fpTextAccum = [] }
  183 
  184     attr s name value
  185         | fpInQuantRef s && isElement name "refObjectId" && T.null (fpUnitGroupRef s) =
  186             s { fpUnitGroupRef = bsToText value }
  187         | otherwise = s
  188 
  189     endOpen s _ = s
  190 
  191     txt s content =
  192         let trimmed = BS.dropWhile (== 32) $ BS.dropWhileEnd (== 32) content
  193         in if BS.null trimmed then s else s { fpTextAccum = trimmed : fpTextAccum s }
  194 
  195     closeTag s tag
  196         | isElement tag "UUID" && T.null (fpUUID s) =
  197             s { fpUUID = accum s, fpTextAccum = [] }
  198         | isElement tag "quantitativeReference" =
  199             s { fpInQuantRef = False, fpTextAccum = [] }
  200         | otherwise = s { fpTextAccum = [] }
  201 
  202     cdata = txt
  203     accum s = T.strip $ T.concat $ reverse $ map bsToText (fpTextAccum s)

  206 -- Build FlowDB and UnitDB from ILCD data
  207 --------------------------------------------------------------------------------

  209 buildFlowAndUnitDB
  210     :: M.Map UUID ILCDFlowInfo
  211     -> M.Map UUID UUID          -- flowProperty UUID → unitGroup UUID
  212     -> M.Map UUID (Text, Int)   -- unitGroup UUID → (refUnitName, refIdx)
  213     -> (FlowDB, UnitDB)
  214 buildFlowAndUnitDB flowInfoMap fpMap ugMap = (flows, allUnits)
  215   where
  216     flows = M.mapWithKey mkFlow flowInfoMap
  217 
  218     -- Collect all unique units (keyed by unitGroup UUID)
  219     allUnits = M.fromList
  220         [ (ugId, Unit ugId uName uName "")
  221         | (ugId, (uName, _)) <- M.toList ugMap
  222         ]
  223 
  224     mkFlow uuid info =
  225         let ftype = if ilcdFlowType info == "Elementary flow" then Biosphere else Technosphere
  226         in Flow
  227             { flowId = uuid
  228             , flowName = ilcdBaseName info
  229             , flowCategory = maybe "" (\(Compartment m _ _) -> m) (ilcdCompartment info)
  230             , flowSubcompartment = ilcdCompartment info >>= \(Compartment _ sc _) ->
  231                 if T.null sc then Nothing else Just sc
  232             , flowUnitId = resolveUnit info
  233             , flowType = ftype
  234             , flowSynonyms = M.empty
  235             , flowCAS = ilcdCAS info
  236             , flowSubstanceId = Nothing
  237             }
  238 
  239     resolveUnit info = maybe UUID.nil id $
  240         ilcdFlowPropertyRef info >>= (`M.lookup` fpMap) >>= \ugId ->
  241             ugId <$ M.lookup ugId ugMap  -- use unitGroup UUID as unit key

  247 data ProcState = ProcState
  248     { psUUID              :: !Text
  249     , psBaseName          :: !Text
  250     , psLocation          :: !Text
  251     , psRefFlowIdx        :: !Int
  252     , psExchanges         :: ![ILCDExchangeRaw]
  253     , psInExchange        :: !Bool
  254     , psExInternalId      :: !Int
  255     , psExFlowRef         :: !Text
  256     , psExDirection       :: !Text
  257     , psExAmount          :: !Double
  258     , psExLocation        :: !Text
  259     , psTextAccum         :: ![BS.ByteString]
  260     , psInName            :: !Bool
  261     , psClassifications   :: !(M.Map Text Text)
  262     , psPendingClassName  :: !Text
  263     , psInClass           :: !Bool
  264     }
  265 
  266 parseProcessXML :: BS.ByteString -> Maybe ILCDProcessRaw
  267 parseProcessXML bytes =
  268     case X.fold openTag attr endOpen txt closeTag cdata
  269         (ProcState "" "" "" 0 [] False (-1) "" "" 0 "" [] False M.empty "" False) bytes of
  270         Left _  -> Nothing
  271         Right s -> buildProcess s
  272   where
  273     openTag s tag
  274         | isElement tag "exchange" =
  275             s { psInExchange = True, psTextAccum = []
  276               , psExInternalId = -1, psExFlowRef = ""
  277               , psExDirection = "", psExAmount = 0, psExLocation = "" }
  278         | isElement tag "name" && not (psInExchange s) =
  279             s { psInName = True, psTextAccum = [] }
  280         | isElement tag "class" && not (psInExchange s) =
  281             s { psInClass = True, psTextAccum = [] }
  282         | otherwise = s { psTextAccum = [] }
  283 
  284     attr s name value
  285         | isElement name "dataSetInternalID" && psInExchange s =
  286             case TR.decimal (bsToText value) of
  287                 Right (n, _) -> s { psExInternalId = n }
  288                 Left _       -> s
  289         | isElement name "refObjectId" && psInExchange s && T.null (psExFlowRef s) =
  290             s { psExFlowRef = bsToText value }
  291         | isElement name "location" && not (psInExchange s) && T.null (psLocation s) =
  292             s { psLocation = bsToText value }
  293         | isElement name "name" && not (psInExchange s) && not (psInName s) =
  294             s { psPendingClassName = bsToText value }
  295         | otherwise = s
  296 
  297     endOpen s _ = s
  298 
  299     txt s content =
  300         let trimmed = BS.dropWhile (== 32) $ BS.dropWhileEnd (== 32) content
  301         in if BS.null trimmed then s else s { psTextAccum = trimmed : psTextAccum s }
  302 
  303     closeTag s tag
  304         | isElement tag "UUID" && T.null (psUUID s) =
  305             s { psUUID = accum s, psTextAccum = [] }
  306         | isElement tag "baseName" && psInName s =
  307             s { psBaseName = accum s, psTextAccum = [] }
  308         | isElement tag "name" && not (psInExchange s) =
  309             s { psInName = False, psTextAccum = [] }
  310         | isElement tag "referenceToReferenceFlow" =
  311             case TR.decimal (accum s) of
  312                 Right (n, _) -> s { psRefFlowIdx = n, psTextAccum = [] }
  313                 Left _       -> s { psTextAccum = [] }
  314         | isElement tag "exchangeDirection" && psInExchange s =
  315             s { psExDirection = accum s, psTextAccum = [] }
  316         | isElement tag "resultingAmount" && psInExchange s =
  317             s { psExAmount = parseDouble (accum s), psTextAccum = [] }
  318         | isElement tag "meanAmount" && psInExchange s && psExAmount s == 0 =
  319             s { psExAmount = parseDouble (accum s), psTextAccum = [] }
  320         | isElement tag "location" && psInExchange s =
  321             s { psExLocation = accum s, psTextAccum = [] }
  322         | isElement tag "class" && psInClass s =
  323             let classVal = accum s
  324                 key = psPendingClassName s
  325                 existing = M.findWithDefault "" key (psClassifications s)
  326                 val = if T.null existing then classVal else existing <> "/" <> classVal
  327             in s { psClassifications = if T.null classVal then psClassifications s
  328                                        else M.insert key val (psClassifications s)
  329                  , psInClass = False, psTextAccum = [] }
  330         | isElement tag "classification" =
  331             s { psPendingClassName = "", psTextAccum = [] }
  332         | isElement tag "exchange" =
  333             let ex = ILCDExchangeRaw
  334                     { ierInternalId = psExInternalId s
  335                     , ierFlowRef = maybe UUID.nil id (UUID.fromText (psExFlowRef s))
  336                     , ierDirection = psExDirection s
  337                     , ierAmount = psExAmount s
  338                     , ierLocation = psExLocation s
  339                     }
  340             in s { psInExchange = False, psExchanges = ex : psExchanges s, psTextAccum = [] }
  341         | otherwise = s { psTextAccum = [] }
  342 
  343     cdata = txt
  344     accum s = T.strip $ T.concat $ reverse $ map bsToText (psTextAccum s)
  345 
  346     parseDouble t = case TR.double t of { Right (v, _) -> v; Left _ -> 0 }
  347 
  348     buildProcess s = do
  349         uuid <- UUID.fromText (psUUID s)
  350         if T.null (psBaseName s) then Nothing
  351         else Just ILCDProcessRaw
  352             { iprUUID = uuid
  353             , iprName = psBaseName s
  354             , iprLocation = psLocation s
  355             , iprRefFlowIdx = psRefFlowIdx s
  356             , iprExchanges = reverse (psExchanges s)
  357             , iprClassifications = psClassifications s
  358             }

  360 -- | Parse process files in parallel using worker pattern
  361 parseProcessFilesParallel :: [FilePath] -> IO [ILCDProcessRaw]
  362 parseProcessFilesParallel files = do
  363     numWorkers <- getNumCapabilities
  364     let workers = distributeFiles numWorkers files
  365     workerResults <- mapConcurrently parseWorker workers
  366     return $ concat workerResults
  367   where
  368     parseWorker paths = do
  369         results <- mapM (\f -> parseProcessXML <$> BS.readFile f) paths
  370         return [r | Just r <- results]

  373 -- Build ActivityMap from raw processes
  374 --------------------------------------------------------------------------------
  375 
  376 buildActivityMap
  377     :: M.Map UUID ILCDFlowInfo
  378     -> FlowDB
  379     -> UnitDB
  380     -> [ILCDProcessRaw]
  381     -> ActivityMap
  382 buildActivityMap flowInfoMap flowDB unitDB procs =
  383     M.fromList
  384         [ ((iprUUID p, refFlowUUID), activity)
  385         | p <- procs
  386         , let refEx = findRefExchange p
  387         , let refFlowUUID = maybe (iprUUID p) ierFlowRef refEx
  388         , let activity = buildActivity flowInfoMap flowDB unitDB p
  389         ]
  390 
  391 findRefExchange :: ILCDProcessRaw -> Maybe ILCDExchangeRaw
  392 findRefExchange p =
  393     case filter (\e -> ierInternalId e == iprRefFlowIdx p) (iprExchanges p) of
  394         (e:_) -> Just e
  395         []    -> Nothing
  396 
  397 buildActivity :: M.Map UUID ILCDFlowInfo -> FlowDB -> UnitDB -> ILCDProcessRaw -> Activity
  398 buildActivity flowInfoMap flowDB unitDB p = Activity
  399     { activityName = iprName p
  400     , activityDescription = []
  401     , activitySynonyms = M.empty
  402     , activityClassification = iprClassifications p
  403     , activityLocation = iprLocation p
  404     , activityUnit = refUnit
  405     , exchanges = map (mkExchange (iprRefFlowIdx p)) (iprExchanges p)
  406     , activityParams = M.empty
  407     , activityParamExprs = M.empty
  408     }
  409   where
  410     refUnit = case findRefExchange p of
  411         Just re -> case M.lookup (ierFlowRef re) flowDB of
  412             Just flow -> case M.lookup (flowUnitId flow) unitDB of
  413                 Just u  -> unitName u
  414                 Nothing -> "kg"
  415             Nothing -> "kg"
  416         Nothing -> "kg"
  417 
  418     mkExchange refIdx raw =
  419         let flowUUID = ierFlowRef raw
  420             isInput = ierDirection raw == "Input"
  421             isRef = ierInternalId raw == refIdx
  422             isElementary = maybe False (\fi -> ilcdFlowType fi == "Elementary flow")
  423                                (M.lookup flowUUID flowInfoMap)
  424             fUnitId = maybe UUID.nil flowUnitId (M.lookup flowUUID flowDB)
  425         in if isElementary
  426             then BiosphereExchange flowUUID (ierAmount raw) fUnitId isInput (ierLocation raw)
  427             else TechnosphereExchange flowUUID (ierAmount raw) fUnitId isInput isRef
  428                     UUID.nil Nothing (ierLocation raw)

  434 -- | Flow UUID → (activityUUID, productUUID) for reference exchanges
  435 type SupplierIndex = M.Map UUID (UUID, UUID)
  436 
  437 fixILCDActivityLinks :: SimpleDatabase -> IO SimpleDatabase
  438 fixILCDActivityLinks db = do
  439     let idx = buildSupplierIndex (sdbActivities db)
  440     reportProgress Info $ printf "Built supplier index with %d entries for ILCD linking" (M.size idx)
  441     return db { sdbActivities = M.map (fixActivityExchanges idx) (sdbActivities db) }
  442 
  443 -- | Build a flow-UUID-keyed index of (activityUUID, productUUID) from reference exchanges.
  444 -- UUID-based: no name collisions, no indirection through flowDB.
  445 buildSupplierIndex :: ActivityMap -> SupplierIndex
  446 buildSupplierIndex activities =
  447     M.fromList
  448         [ (exchangeFlowId ex, (actUUID, prodUUID))
  449         | ((actUUID, prodUUID), act) <- M.toList activities
  450         , ex <- exchanges act
  451         , exchangeIsReference ex
  452         ]
  453 
  454 fixActivityExchanges :: SupplierIndex -> Activity -> Activity
  455 fixActivityExchanges idx act =
  456     act { exchanges = map fixEx (exchanges act) }
  457   where
  458     fixEx ex@(TechnosphereExchange fid amt uid isInp _ _ procLink loc)
  459         | isInp = case M.lookup fid idx of
  460             Just (actUUID, prodUUID) ->
  461                 TechnosphereExchange prodUUID amt uid isInp False actUUID procLink loc
  462             Nothing -> ex
  463     fixEx ex = ex
