    3 {- |
    4 Module      : Matrix
    5 Description : Matrix-based LCA calculations using MUMPS sparse direct solver
    6 
    7 This module implements matrix-based Life Cycle Assessment (LCA) calculations using
    8 the MUMPS sequential direct solver for high-performance sparse linear algebra.
    9 It solves the fundamental LCA equation: (I - A)^-1 * f = s, where:
   10 
   11 - I is the identity matrix
   12 - A is the technosphere matrix (activities x activities)
   13 - f is the final demand vector
   14 - s is the supply vector (scaling factors)
   15 
   16 The biosphere inventory is then calculated as: B * s = g, where:
   17 - B is the biosphere matrix (flows x activities)
   18 - g is the final inventory vector
   19 
   20 Key features:
   21 - Uses MUMPS direct solver (LU factorization) for numerical stability
   22 - Handles sparse matrices efficiently with coordinate triplet format
   23 - Supports large-scale problems (tested with 15K+ activities)
   24 
   25 Performance characteristics:
   26 - Matrix assembly: O(nnz) where nnz is number of non-zero entries
   27 - MUMPS factorization: O(n^1.5) for sparse LCA matrices
   28 - Forward/backward solve: O(n log n)
   29 - Memory usage: ~50-100 MB for typical Ecoinvent database
   30 -}

   90 -- | Clear cached solver for a database (call when unloading)
   91 clearCachedSolver :: Text -> IO ()
   92 clearCachedSolver dbName =
   93     modifyMVar_ cachedSolver $ \cache ->
   94         case M.lookup dbName cache of
   95             Just (solver, _) -> do
   96                 mumpsDestroy solver
   97                 return $ M.delete dbName cache
   98             Nothing -> return cache

  100 -- | Aggregate duplicate matrix entries by summing values for same (i,j) coordinates
  101 aggregateMatrixEntries :: [(Int, Int, Double)] -> [(Int, Int, Double)]
  102 aggregateMatrixEntries entries =
  103     let groupedEntries = M.fromListWith (+) [((i, j), val) | (i, j, val) <- entries]
  104      in [(i, j, val) | ((i, j), val) <- M.toList groupedEntries]

  113 {- |
  114 Fast solver using the globally cached pre-factorized MUMPS solver.
  115 
  116 This function uses the cached solver with pre-computed LU factorization,
  117 eliminating both matrix assembly and factorization time for concurrent requests.
  118 Achieves sub-second inventory calculations after server startup.
  119 
  120 The solver is looked up by database ID from the per-database cache, enabling
  121 instant switching between databases without re-factorization.
  122 -}
  123 solveSparseLinearSystemWithFactorization :: MatrixFactorization -> Vector -> IO Vector
  124 solveSparseLinearSystemWithFactorization factorization demandVec = do
  125     let dbId = mfDatabaseId factorization
  126     cachedSolvers <- readMVar cachedSolver
  127     case M.lookup dbId cachedSolvers of
  128         Nothing -> do
  129             reportMatrixOperation $ "No cached factorization found for database '" ++ T.unpack dbId ++ "', falling back to matrix assembly"
  130             let systemMatrix = mfSystemMatrix factorization
  131                 n = mfActivityCount factorization
  132                 techTriples = U.toList $ U.filter (\(SparseTriple i j _) -> i /= j) $ U.map (\(SparseTriple i j val) -> SparseTriple i j (-val)) systemMatrix
  133             solveSparseLinearSystemMUMPS [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- techTriples] (fromIntegral n) demandVec
  134 
  135         Just (_solver, n) -> do
  136             reportMatrixOperation $ "Using cached solver for database '" ++ T.unpack dbId ++ "' (" ++ show n ++ " activities) - ultra-fast solve"
  137 
  138             withProgressTiming Solver "MUMPS cached solve" $ do
  139                 result <- catch
  140                     (withMVar solverGlobalMutex $ \_ -> do
  141                         cachedSolvers' <- readMVar cachedSolver
  142                         case M.lookup dbId cachedSolvers' of
  143                             Nothing -> error $ "Cached solver for database '" ++ T.unpack dbId ++ "' disappeared during solve"
  144                             Just (mumpsSolver, _) -> do
  145                                 let rhsVec = VS.fromList $ toList demandVec
  146                                 solutionVS <- mumpsSolve mumpsSolver rhsVec
  147                                 return $ Just (VS.convert solutionVS :: Vector))
  148                     (\e -> do
  149                         reportMatrixOperation $ "MUMPS cached solver failed: " ++ show (e :: SomeException)
  150                         reportMatrixOperation "Falling back to fresh solver for this request"
  151                         return Nothing)
  152 
  153                 case result of
  154                     Just vec -> return vec
  155                     Nothing -> do
  156                         let systemMatrix = mfSystemMatrix factorization
  157                             techTriples = U.toList $ U.filter (\(SparseTriple i j _) -> i /= j) $ U.map (\(SparseTriple i j val) -> SparseTriple i j (-val)) systemMatrix
  158                         solveSparseLinearSystemMUMPS [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- techTriples] (fromIntegral n) demandVec

  160 {- |
  161 Solve the fundamental LCA equation (I - A) * x = b using MUMPS direct solver.
  162 
  163 This function:
  164 1. Constructs the (I - A) system matrix from technosphere triplets
  165 2. Uses MUMPS LU factorization for numerical stability
  166 3. Returns the supply vector (scaling factors) for all activities
  167 
  168 Performance: ~3s for 14,457 activities with 116K technosphere entries
  169 -}
  170 solveSparseLinearSystemMUMPS :: [(Int, Int, Double)] -> Int -> Vector -> IO Vector
  171 solveSparseLinearSystemMUMPS techTriples n demandVec =
  172     withMVar solverGlobalMutex $ \_ -> do
  173         let identityTriples = [(i, i, 1.0) | i <- [0 .. n - 1]]
  174             systemTechTriples = [(i, j, -value) | (i, j, value) <- techTriples]
  175             allTriples = aggregateMatrixEntries (identityTriples ++ systemTechTriples)
  176             (rows, cols, vals) = unzip3 allTriples
  177             nnz = length allTriples
  178 
  179         reportMatrixOperation $ "Matrix assembly completed - starting MUMPS direct solve for " ++ show n ++ " activities"
  180 
  181         withProgressTiming Solver "MUMPS solve" $ do
  182             solver <- mumpsCreate n nnz rows cols vals
  183             mumpsAnalyzeAndFactorize solver
  184             let rhsVec = VS.fromList $ toList demandVec
  185             solutionVS <- mumpsSolve solver rhsVec
  186             mumpsDestroy solver
  187 
  188             let solutionVec = VS.convert solutionVS :: Vector
  189             when (U.any (\x -> isInfinite x || isNaN x) solutionVec) $ do
  190                 reportMatrixOperation "ERROR: Solution contains infinity or NaN values"
  191                 reportMatrixOperation "Matrix is singular - likely missing reference flows or treatment activities"
  192 
  193             return solutionVec

  195 -- | Solve linear system (I - A) * x = b using MUMPS direct solver (wrapper)
  196 solveSparseLinearSystem :: [(Int, Int, Double)] -> Int -> Vector -> IO Vector
  197 solveSparseLinearSystem = solveSparseLinearSystemMUMPS

  199 -- | Efficient sparse matrix multiplication: result = A * vector
  200 applySparseMatrix :: [(Int, Int, Double)] -> Int -> Vector -> Vector
  201 applySparseMatrix sparseTriples nRows inputVec =
  202     let resultVec = U.create $ do
  203             result <- MU.new nRows
  204             forM_ [0 .. nRows - 1] $ \i -> MU.write result i 0.0
  205             forM_ sparseTriples $ \(i, j, val) -> do
  206                 if j < U.length inputVec
  207                     then do
  208                         oldVal <- MU.read result i
  209                         let newVal = oldVal + val * (inputVec U.! j)
  210                         MU.write result i newVal
  211                     else return ()
  212             return result
  213      in resultVec

  215 {- |
  216 Compute the scaling vector by solving (I - A)x = d.
  217 
  218 Returns the supply vector where x[i] is the scaling factor for activity i
  219 (how much of activity i is needed to produce one unit of the root activity).
  220 -}
  221 computeScalingVector :: Database -> ProcessId -> IO Vector
  222 computeScalingVector db rootProcessId = do
  223     let activityCount = dbActivityCount db
  224         techTriples = dbTechnosphereTriples db
  225         activityIndex = dbActivityIndex db
  226         demandVec = buildDemandVectorFromIndex activityIndex rootProcessId
  227     case dbCachedFactorization db of
  228         Just factorization -> solveSparseLinearSystemWithFactorization factorization demandVec
  229         Nothing -> solveSparseLinearSystem [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- U.toList techTriples] (fromIntegral activityCount) demandVec

  231 {- |
  232 Apply the biosphere matrix to a scaling vector: g = B * x.
  233 
  234 Converts the scaling vector (activity scaling factors) into a biosphere inventory
  235 (environmental flows) by multiplying with the biosphere intervention matrix.
  236 -}
  237 applyBiosphereMatrix :: Database -> Vector -> Inventory
  238 applyBiosphereMatrix db supplyVec =
  239     let bioFlowCount = dbBiosphereCount db
  240         bioTriples = dbBiosphereTriples db
  241         bioFlowIndex = M.fromList $ zip (V.toList $ dbBiosphereFlows db) [0..]
  242         inventoryVec = applySparseMatrix [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- U.toList bioTriples] (fromIntegral bioFlowCount) supplyVec
  243     in M.fromList [(uuid, inventoryVec U.! idx)
  244                   | (uuid, idx) <- M.toList bioFlowIndex
  245                   , idx < U.length inventoryVec]
  246 
  247 computeInventoryMatrix :: Database -> ProcessId -> IO Inventory
  248 computeInventoryMatrix db rootProcessId =
  249     applyBiosphereMatrix db <$> computeScalingVector db rootProcessId

  251 {- |
  252 Per-process LCIA contributions for one impact category.
  253 contribution[p] = Σ_f  B[f,p] * s[p] * CF[f]   (in the method's impact unit)
  254 
  255 Exact for cyclic systems because it uses the full matrix-solved scaling vector.
  256 -}
  257 computeProcessLCIAContributions
  258     :: Database
  259     -> Vector               -- ^ Scaling vector (from computeScalingVector)
  260     -> M.Map UUID Double    -- ^ CF map: flowUUID → CF value (single impact category)
  261     -> M.Map ProcessId Double  -- ^ ProcessId → LCIA contribution in impact unit
  262 computeProcessLCIAContributions db scalingVec cfMap =
  263     let actIdx      = dbActivityIndex db
  264         bioTriples  = dbBiosphereTriples db
  265         bioFlows    = dbBiosphereFlows db
  266         -- Build inverse map: matrix column index → ProcessId
  267         colToProcess :: M.Map Int ProcessId
  268         colToProcess = M.fromList
  269             [ (fromIntegral (actIdx V.! pid), fromIntegral pid)
  270             | pid <- [0 .. V.length actIdx - 1]
  271             ]
  272         step acc (SparseTriple flowRow colIdx bioVal) =
  273             let flowUUID = bioFlows V.! fromIntegral flowRow
  274             in case M.lookup flowUUID cfMap of
  275                 Nothing    -> acc
  276                 Just cfVal ->
  277                     case M.lookup (fromIntegral colIdx :: Int) colToProcess of
  278                         Nothing  -> acc
  279                         Just pid ->
  280                             let s = scalingVec U.! fromIntegral colIdx
  281                             in M.insertWith (+) pid (bioVal * s * cfVal) acc
  282     in U.foldl' step M.empty bioTriples

  284 {- |
  285 Sherman-Morrison rank-1 update for ingredient substitution (~4ms per variant).
  286 
  287 Swapping supplier oldSup -> newSup for activity 'row' is a rank-1 update to (I-A).
  288 Instead of re-factorizing (~3s), we reuse the cached factorization:
  289 
  290     x' = x - z * (v^T * x) / (1 + v^T * z)
  291 
  292 where z = inv(I-A) * u is one back-substitution, u = coeff*(e_oldSup - e_newSup),
  293 and v = e_row.
  294 
  295 Returns Left if the update is singular (|1 + v^T*z| < epsilon).
  296 -}
  297 shermanMorrisonVariant
  298     :: Database
  299     -> Maybe MatrixFactorization  -- ^ Cached factorization from SharedSolver (Nothing = full re-solve)
  300     -> Vector              -- ^ Original scaling vector x
  301     -> Int                 -- ^ Activity being modified (row = consumer index)
  302     -> Int                 -- ^ Old supplier index (to remove)
  303     -> Int                 -- ^ New supplier index (to add)
  304     -> Double              -- ^ Exchange coefficient a (amount consumed)
  305     -> IO (Either T.Text Vector)  -- ^ Variant scaling vector x'
  306 shermanMorrisonVariant db mFact x row oldSup newSup coeff = do
  307     let n = U.length x
  308         u = fromList [if i == oldSup then coeff
  309                       else if i == newSup then -coeff
  310                       else 0.0
  311                      | i <- [0 .. n - 1]]
  312     z <- case mFact of
  313         Just f  -> solveSparseLinearSystemWithFactorization f u
  314         Nothing -> do
  315             let techTriples = dbTechnosphereTriples db
  316                 activityCount = dbActivityCount db
  317             solveSparseLinearSystem
  318                 [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- U.toList techTriples]
  319                 (fromIntegral activityCount) u
  320     let vtx = x U.! row
  321         vtz = z U.! row
  322         denom = 1.0 + vtz
  323     if abs denom < 1e-12
  324         then return $ Left "Sherman-Morrison update is singular \x2014 substitution creates a degenerate supply chain"
  325         else do
  326             let scale = vtx / denom
  327             return $ Right $ U.imap (\i xi -> xi - scale * (z U.! i)) x

  329 {- |
  330 Compute inventory with cross-database dependencies.
  331 
  332 This function implements the block matrix back-substitution algorithm:
  333 1. Solve the local database system: (I - A_local) x s_local = f
  334 2. For each cross-database link, accumulate demand on the supplier
  335 3. Group demand by dependency database
  336 4. Recursively solve each dependency database
  337 5. Sum inventories from all databases
  338 -}
  339 computeInventoryWithDependencies
  340     :: (UUID -> UUID -> Maybe Database)  -- ^ Lookup dependency DB by supplier (actUUID, prodUUID)
  341     -> Database                          -- ^ Current database
  342     -> ProcessId                         -- ^ Root activity
  343     -> IO Inventory
  344 computeInventoryWithDependencies lookupDb db rootProcessId = do
  345     let activityCount = dbActivityCount db
  346         bioFlowCount = dbBiosphereCount db
  347         techTriples = dbTechnosphereTriples db
  348         bioTriples = dbBiosphereTriples db
  349         activityIndex = dbActivityIndex db
  350         bioFlowIndex = M.fromList $ zip (V.toList $ dbBiosphereFlows db) [0..]
  351         demandVec = buildDemandVectorFromIndex activityIndex rootProcessId
  352 
  353     localSupplyVec <- case dbCachedFactorization db of
  354         Just factorization -> solveSparseLinearSystemWithFactorization factorization demandVec
  355         Nothing -> solveSparseLinearSystem
  356             [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- U.toList techTriples]
  357             (fromIntegral activityCount)
  358             demandVec
  359 
  360     let localInventoryVec = applySparseMatrix
  361             [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- U.toList bioTriples]
  362             (fromIntegral bioFlowCount)
  363             localSupplyVec
  364         localInventory = M.fromList
  365             [(uuid, localInventoryVec U.! idx) | (uuid, idx) <- M.toList bioFlowIndex, idx < U.length localInventoryVec]
  366 
  367         crossLinks = dbCrossDBLinks db
  368         processIdLookup = dbProcessIdLookup db
  369 
  370         crossDBDemands :: M.Map (UUID, UUID) Double
  371         crossDBDemands = foldr accumulateDemand M.empty crossLinks
  372           where
  373             accumulateDemand :: CrossDBLink -> M.Map (UUID, UUID) Double -> M.Map (UUID, UUID) Double
  374             accumulateDemand link acc =
  375                 let consumerKey = (cdlConsumerActUUID link, cdlConsumerProdUUID link)
  376                     supplierKey = (cdlSupplierActUUID link, cdlSupplierProdUUID link)
  377                 in case M.lookup consumerKey processIdLookup of
  378                     Nothing -> acc
  379                     Just consumerPid ->
  380                         let consumerIdx = fromIntegral $ activityIndex V.! fromIntegral consumerPid
  381                             consumerScale = localSupplyVec U.! consumerIdx
  382                             demand = cdlCoefficient link * consumerScale
  383                         in M.insertWith (+) supplierKey demand acc
  384 
  385     let processSupplier (supplierActUUID, supplierProdUUID) demand accIO = do
  386             acc <- accIO
  387             case lookupDb supplierActUUID supplierProdUUID of
  388                 Nothing -> return acc
  389                 Just depDb ->
  390                     case M.lookup (supplierActUUID, supplierProdUUID) (dbProcessIdLookup depDb) of
  391                         Nothing -> return acc
  392                         Just supplierPid -> do
  393                             depInventory <- computeInventoryWithDependencies lookupDb depDb supplierPid
  394                             let scaledInventory = M.map (* demand) depInventory
  395                             return $ M.unionWith (+) acc scaledInventory
  396     crossDBInventory <- M.foldrWithKey processSupplier (return M.empty) crossDBDemands
  397 
  398     return $ M.unionWith (+) localInventory crossDBInventory

  400 {- |
  401 Build the final demand vector f for LCA calculations.
  402 
  403 The demand vector represents external demand for products from each activity:
  404 - f[i] = 1.0 for the root activity (functional unit)
  405 - f[i] = 0.0 for all other activities
  406 -}
  407 buildDemandVectorFromIndex :: V.Vector Int32 -> ProcessId -> Vector
  408 buildDemandVectorFromIndex activityIndex rootProcessId =
  409     let n = V.length activityIndex
  410         rootIndex = if fromIntegral rootProcessId >= (0 :: Int) && fromIntegral rootProcessId < n
  411                     then fromIntegral $ activityIndex V.! fromIntegral rootProcessId
  412                     else error $ "FATAL: ProcessId not found in activity index: " ++ show rootProcessId
  413      in fromList [if i == rootIndex then 1.0 else 0.0 | i <- [0 .. n - 1 :: Int]]

  415 {- |
  416 Pre-compute matrix factorization for concurrent inventory calculations.
  417 
  418 This function builds the (I - A) system matrix from technosphere triplets and
  419 pre-computes the LU factorization during server startup. The resulting
  420 MatrixFactorization can be stored in the Database for fast concurrent solves.
  421 
  422 Performance: ~3s factorization time for full Ecoinvent, saves 2.9s per inventory request
  423 -}
  424 precomputeMatrixFactorization :: Text -> [(Int, Int, Double)] -> Int -> IO MatrixFactorization
  425 precomputeMatrixFactorization dbName techTriples n =
  426     withMVar solverGlobalMutex $ \_ -> do
  427         let identityTriples = [(i, i, 1.0) | i <- [0 .. n - 1]]
  428             systemTechTriples = [(i, j, -value) | (i, j, value) <- techTriples]
  429             systemMatrix = aggregateMatrixEntries (identityTriples ++ systemTechTriples)
  430             (rows, cols, vals) = unzip3 systemMatrix
  431             nnz = length systemMatrix
  432 
  433         reportMatrixOperation $ "Pre-computing factorization for database '" ++ T.unpack dbName ++ "' (" ++ show n ++ " activities, " ++ show nnz ++ " entries)"
  434 
  435         solver <- mumpsCreate n nnz rows cols vals
  436         mumpsAnalyzeAndFactorize solver
  437 
  438         modifyMVar_ cachedSolver $ \solvers -> return $ M.insert dbName (solver, n) solvers
  439 
  440         reportMatrixOperation $ "MUMPS solver for database '" ++ T.unpack dbName ++ "' factorized and cached"
  441 
  442         let factorization = MatrixFactorization
  443                 { mfSystemMatrix = U.fromList [SparseTriple (fromIntegral i) (fromIntegral j) v | (i, j, v) <- systemMatrix]
  444                 , mfActivityCount = fromIntegral n
  445                 , mfDatabaseId = dbName
  446                 }
  447 
  448         return factorization

  451 addFactorizationToDatabase :: Database -> MatrixFactorization -> Database
  452 addFactorizationToDatabase db factorization = db { dbCachedFactorization = Just factorization }
