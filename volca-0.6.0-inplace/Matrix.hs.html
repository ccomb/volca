    7 
    8 This module implements matrix-based Life Cycle Assessment (LCA) calculations using
    9 the PETSc library for high-performance sparse linear algebra. It solves the fundamental
   10 LCA equation: (I - A)⁻¹ * f = s, where:
   11

   16

   18 - B is the biosphere matrix (flows × activities)

   22 - Uses MUMPS direct solver through PETSc for numerical stability
   23 - Configurable through PETSC_OPTIONS environment variables

   25 - Supports large-scale problems (tested with 15K+ activities)
   26 
   27 Performance characteristics:
   28 - Matrix assembly: O(nnz) where nnz is number of non-zero entries
   29 - MUMPS factorization: O(n^1.5) for sparse LCA matrices

  156 solveSparseLinearSystemWithFactorization :: MatrixFactorization -> Vector -> IO Vector

  164             reportMatrixOperation $ "No cached factorization found for database '" ++ T.unpack dbId ++ "', falling back to matrix assembly"
  165             let systemMatrix = mfSystemMatrix factorization
  166                 n = mfActivityCount factorization

  168                 techTriples = U.toList $ U.filter (\(SparseTriple i j _) -> i /= j) $ U.map (\(SparseTriple i j val) -> SparseTriple i j (-val)) systemMatrix

  170 
  171         Just (_ksp, _petscMat, n) -> do
  172             reportMatrixOperation $ "Using cached solver for database '" ++ T.unpack dbId ++ "' (" ++ show n ++ " activities) - ultra-fast solve"
  173 
  174             -- Time the ultra-fast solve with thread synchronization and error handling
  175             result <- withProgressTiming Solver "PETSc thread-safe cached solve" $ do
  176                 -- Use withMVar to ensure thread-safe access to the cached KSP solver

  184                                 let rhsData = V.fromList $ map realToFrac $ toList demandVec
  185                                 let comm' = commWorld

  187                                 withVecNew comm' rhsData $ \rhs -> do
  188                                     withVecDuplicate rhs $ \solution -> do

  191                                         solutionData <- vecGetVS solution

  200                     Just solutionList -> return solutionList

  202                         -- Fallback to standard solver if cached solver fails
  203                         let systemMatrix = mfSystemMatrix factorization

  205                         fallbackVec <- solveSparseLinearSystemPETSc [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- techTriples] (fromIntegral n) demandVec

  210 {- |
  211 Solve the fundamental LCA equation (I - A) * x = b using PETSc MUMPS direct solver.
  212 
  213 This function:
  214 1. Constructs the (I - A) system matrix from technosphere triplets
  215 2. Uses PETSc's MUMPS direct solver for numerical stability
  216 3. Returns the supply vector (scaling factors) for all activities
  217 
  218 The solver configuration is controlled by PETSC_OPTIONS environment variable,
  219 allowing fine-tuning of MUMPS parameters for optimal performance.
  220 
  221 Performance: ~3s for 14,457 activities with 116K technosphere entries
  222 -}
  223 solveSparseLinearSystemPETSc :: [(Int, Int, Double)] -> Int -> Vector -> IO Vector
  224 -- entered 6 timessolveSparseLinearSystemPETSc techTriples n demandVec = do
  225     -- Serialize ALL PETSc operations to prevent concurrent matrix assembly/solving
  226     withMVar petscGlobalMutex $ \_ -> do
  227         -- Ensure PETSc is initialized globally (once and persistent)
  228         petscGlobalInit
  229 
  230         -- Build (I - A) system from sparse triplets
  231         -- Add identity matrix: I[i,i] = 1.0
  232         let identityTriples = [(i, i, 1.0) | i <- [0 .. n - 1]]
  233         let systemTechTriples = [(i, j, -value) | (i, j, value) <- techTriples]
  234         let allTriples = identityTriples ++ systemTechTriples
  235 
  236         -- Aggregate duplicate entries for proper matrix assembly
  237         let aggregatedTriples = aggregateMatrixEntries allTriples
  238 
  239         reportMatrixOperation $ "Matrix assembly completed - starting PETSc direct solve for " ++ show n ++ " activities"
  240 
  241         -- Convert to PETSc format
  242         let matrixData = buildPetscMatrixData aggregatedTriples n
  243         let rhsData = V.fromList $ map realToFrac $ toList demandVec
  244         let nzPattern = ConstNZPR (fromIntegral $ length aggregatedTriples, fromIntegral $ length aggregatedTriples)
  245 
  246         -- Use PETSc within the already initialized context (no re-initialization)
  247         let comm' = commWorld
  248         result <- withPetscMatrix comm' n n MatAij matrixData nzPattern InsertValues $ \mat ->
  249             withVecNew comm' rhsData $ \rhs -> do
  250                 let (_, _, _, matMutable) = fromPetscMatrix mat
  251 
  252                 -- Time the solver execution with enhanced error handling
  253                 withProgressTiming Solver "PETSc solve with environment options" $ do
  254                     -- Use explicit KSP setup to ensure PETSC_OPTIONS are consumed properly
  255                     withKsp comm' $ \ksp -> do
  256                         -- Set up operators first
  257                         kspSetOperators ksp matMutable matMutable
  258 
  259                         -- Explicitly consume PETSC_OPTIONS environment variables
  260                         kspSetFromOptions ksp
  261                         pc <- kspGetPc ksp
  262                         pcSetFromOptions pc
  263 
  264                         -- Debug: Report that options were applied
  265                         reportSolverOperation "Applied PETSC_OPTIONS to KSP and PC"
  266 
  267                         -- Configure solver
  268                         kspSetInitialGuessNonzero ksp False
  269                         kspSetUp ksp
  270 
  271                         -- Allocate solution vector and solve
  272                         withVecDuplicate rhs $ \solution -> do
  273                             kspSolve ksp rhs solution
  274 
  275                             -- Extract solution as list
  276                             solutionData <- vecGetVS solution
  277                             let solutionList = VS.toList solutionData
  278                             let solutionResult = map realToFrac solutionList
  279 
  280                             -- Check for infinity/NaN in solution (indicates singular/ill-conditioned matrix)
  281                             let hasInfinity = any (\x -> isInfinite x || isNaN x) solutionResult
  282                             when hasInfinity $ do
  283                                 reportMatrixOperation "ERROR: Solution contains infinity or NaN values"
  284                                 reportMatrixOperation "Matrix is singular - likely missing reference flows or treatment activities"
  285                                 reportMatrixOperation "Check: (1) all activities have reference flows, (2) treatment inputs included in matrix"
  286 
  287                             return solutionResult
  288 
  289         return $ fromList result

  295 -- | Efficient sparse matrix multiplication: result = A * vector
  296 applySparseMatrix :: [(Int, Int, Double)] -> Int -> Vector -> Vector
  297 -- entered 6 timesapplySparseMatrix sparseTriples nRows inputVec =
  298     let resultVec = U.create $ do
  299             result <- MU.new nRows
  300             -- Initialize to zero
  301             forM_ [0 .. nRows - 1] $ \i -> MU.write result i 0.0
  302 
  303             -- Apply sparse matrix
  304             forM_ sparseTriples $ \(i, j, val) -> do
  305                 if j < U.length inputVec
  306                     then do
  307                         oldVal <- MU.read result i
  308                         let newVal = oldVal + val * (inputVec U.! j)
  309                         MU.write result i newVal
  310                     else return ()
  311 
  312             return result
  313      in resultVec

  375 {- |
  376 Sherman-Morrison rank-1 update for ingredient substitution (~4ms per variant).
  377 
  378 Swapping supplier oldSup → newSup for activity 'row' is a rank-1 update to (I-A).
  379 Instead of re-factorizing (~3s), we reuse the cached factorization:
  380 
  381     x' = x - z · (v^T · x) / (1 + v^T · z)
  382 
  383 where z = inv(I-A) · u is one back-substitution, u = coeff·(e_oldSup - e_newSup),
  384 and v = e_row.
  385 
  386 Returns Left if the update is singular (|1 + v^T·z| < epsilon).
  387 -}
  388 shermanMorrisonVariant
  389     :: Database
  390     -> Vector              -- ^ Original scaling vector x
  391     -> Int                 -- ^ Activity being modified (row = consumer index)
  392     -> Int                 -- ^ Old supplier index (to remove)
  393     -> Int                 -- ^ New supplier index (to add)
  394     -> Double              -- ^ Exchange coefficient a (amount consumed)
  395     -> IO (Either T.Text Vector)  -- ^ Variant scaling vector x'
  396 -- never enteredshermanMorrisonVariant db x row oldSup newSup coeff = do
  397     let n = U.length x
  398         -- Build u = coeff · (e_oldSup - e_newSup)
  399         u = fromList [if i == oldSup then coeff
  400                       else if i == newSup then -coeff
  401                       else 0.0
  402                      | i <- [0 .. n - 1]]
  403     -- Solve (I-A)z = u using cached factorization (~2ms)
  404     z <- case dbCachedFactorization db of
  405         Just f  -> solveSparseLinearSystemWithFactorization f u
  406         Nothing -> do
  407             let techTriples = dbTechnosphereTriples db
  408                 activityCount = dbActivityCount db
  409             solveSparseLinearSystem
  410                 [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- U.toList techTriples]
  411                 (fromIntegral activityCount) u
  412     -- Apply rank-1 update formula
  413     let vtx = x U.! row       -- v^T · x where v = e_row
  414         vtz = z U.! row       -- v^T · z
  415         denom = 1.0 + vtz
  416     if abs denom < 1e-12
  417         then return $ Left "Sherman-Morrison update is singular — substitution creates a degenerate supply chain"
  418         else do
  419             let scale = vtx / denom
  420             return $ Right $ U.imap (\i xi -> xi - scale * (z U.! i)) x

  422 {- |
  423 Compute inventory with cross-database dependencies.
  424 
  425 This function implements the block matrix back-substitution algorithm:
  426 1. Solve the local database system: (I - A_local) × s_local = f
  427 2. For each cross-database link, accumulate demand on the supplier
  428 3. Group demand by dependency database
  429 4. Recursively solve each dependency database
  430 5. Sum inventories from all databases
  431 
  432 Arguments:
  433 - lookupDb: Function to look up a database by supplier UUID pair
  434 - db: The current database (e.g., Ginko)
  435 - rootProcessId: The root activity to solve for
  436 
  437 Returns:
  438 - Combined inventory from current database and all dependencies
  439 
  440 Note: Cross-DB links store raw exchange amounts. We need to multiply by the
  441 consumer's scaling factor (supply vector value) to get actual demand.
  442 -}
  443 computeInventoryWithDependencies
  444     :: (UUID -> UUID -> Maybe Database)  -- ^ Lookup dependency DB by supplier (actUUID, prodUUID)
  445     -> Database                          -- ^ Current database
  446     -> ProcessId                         -- ^ Root activity
  447     -> IO Inventory
  448 -- never enteredcomputeInventoryWithDependencies lookupDb db rootProcessId = do
  449     let -- Step 1: Compute local inventory (same as computeInventoryMatrix)
  450         activityCount = dbActivityCount db
  451         bioFlowCount = dbBiosphereCount db
  452         techTriples = dbTechnosphereTriples db
  453         bioTriples = dbBiosphereTriples db
  454         activityIndex = dbActivityIndex db
  455         bioFlowIndex = M.fromList $ zip (V.toList $ dbBiosphereFlows db) [0..]
  456 
  457         demandVec = buildDemandVectorFromIndex activityIndex rootProcessId
  458 
  459     -- Solve local system
  460     localSupplyVec <- case dbCachedFactorization db of
  461         Just factorization -> solveSparseLinearSystemWithFactorization factorization demandVec
  462         Nothing -> solveSparseLinearSystem
  463             [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- U.toList techTriples]
  464             (fromIntegral activityCount)
  465             demandVec
  466 
  467     let -- Calculate local inventory
  468         localInventoryVec = applySparseMatrix
  469             [(fromIntegral i, fromIntegral j, v) | SparseTriple i j v <- U.toList bioTriples]
  470             (fromIntegral bioFlowCount)
  471             localSupplyVec
  472         localInventory = M.fromList
  473             [(uuid, localInventoryVec U.! idx) | (uuid, idx) <- M.toList bioFlowIndex, idx < U.length localInventoryVec]
  474 
  475         -- Step 2: Compute demand for cross-DB links
  476         -- For each cross-DB link, the demand on the supplier is:
  477         --   demand = coefficient × supply[consumer]
  478         -- where supply[consumer] is the scaling factor for the consumer activity
  479         crossLinks = dbCrossDBLinks db
  480         processIdLookup = dbProcessIdLookup db
  481 
  482         -- Group links by supplier database (determined by looking up each supplier)
  483         -- and accumulate demand
  484         crossDBDemands :: M.Map (UUID, UUID) Double  -- (supplierActUUID, supplierProdUUID) -> total demand
  485         crossDBDemands = foldr accumulateDemand M.empty crossLinks
  486           where
  487             accumulateDemand :: CrossDBLink -> M.Map (UUID, UUID) Double -> M.Map (UUID, UUID) Double
  488             accumulateDemand link acc =
  489                 let consumerKey = (cdlConsumerActUUID link, cdlConsumerProdUUID link)
  490                     supplierKey = (cdlSupplierActUUID link, cdlSupplierProdUUID link)
  491                 in case M.lookup consumerKey processIdLookup of
  492                     Nothing -> acc  -- Consumer not found (shouldn't happen)
  493                     Just consumerPid ->
  494                         let consumerIdx = fromIntegral $ activityIndex V.! fromIntegral consumerPid
  495                             consumerScale = localSupplyVec U.! consumerIdx
  496                             -- Demand = amount per unit output × scaling factor
  497                             demand = cdlCoefficient link * consumerScale
  498                         in M.insertWith (+) supplierKey demand acc
  499 
  500     -- Step 3: For each supplier, find its database and compute its inventory
  501     -- Then sum all inventories
  502     let processSupplier (supplierActUUID, supplierProdUUID) demand accIO = do
  503             acc <- accIO
  504             case lookupDb supplierActUUID supplierProdUUID of
  505                 Nothing -> return acc  -- Supplier database not loaded (warning could be issued)
  506                 Just depDb ->
  507                     case M.lookup (supplierActUUID, supplierProdUUID) (dbProcessIdLookup depDb) of
  508                         Nothing -> return acc  -- Supplier not found in its database
  509                         Just supplierPid -> do
  510                             -- Recursively compute inventory (could recurse further if depDb has dependencies)
  511                             depInventory <- computeInventoryWithDependencies lookupDb depDb supplierPid
  512                             -- Scale by demand
  513                             let scaledInventory = M.map (* demand) depInventory
  514                             return $ M.unionWith (+) acc scaledInventory
  515     crossDBInventory <- M.foldrWithKey processSupplier (return M.empty) crossDBDemands
  516 
  517     -- Step 4: Sum local and cross-DB inventories
  518     return $ M.unionWith (+) localInventory crossDBInventory

  569 {- |
  570 Pre-compute matrix factorization for concurrent inventory calculations.
  571 
  572 This function builds the (I - A) system matrix from technosphere triplets and
  573 pre-computes the factorization during server startup. The resulting
  574 MatrixFactorization can be stored in the Database for fast concurrent solves.
  575 
  576 The database name is used as the key in the per-database solver cache, enabling
  577 instant switching between databases without re-factorization.
  578 
  579 Performance: ~3s factorization time for full Ecoinvent, saves 2.9s per inventory request
  580 -}
  581 precomputeMatrixFactorization :: Text -> [(Int, Int, Double)] -> Int -> IO MatrixFactorization
  582 -- never enteredprecomputeMatrixFactorization dbName techTriples n = do
  583     -- Serialize ALL PETSc operations to prevent concurrent matrix assembly/factorization
  584     withMVar petscGlobalMutex $ \_ -> do
  585         -- Ensure PETSc is initialized
  586         petscGlobalInit
  587 
  588         -- Build (I - A) system matrix
  589         let identityTriples = [(i, i, 1.0) | i <- [0 .. n - 1]]
  590         let systemTechTriples = [(i, j, -value) | (i, j, value) <- techTriples]
  591         let systemMatrix = aggregateMatrixEntries (identityTriples ++ systemTechTriples)
  592 
  593         reportMatrixOperation $ "Pre-computing factorization for database '" ++ T.unpack dbName ++ "' (" ++ show n ++ " activities, " ++ show (length systemMatrix) ++ " entries)"
  594 
  595         -- Create and cache the actual PETSc factorized solver
  596         let matrixData = buildPetscMatrixData systemMatrix n
  597         let nzPattern = ConstNZPR (fromIntegral $ length systemMatrix, fromIntegral $ length systemMatrix)
  598         let comm' = commWorld
  599 
  600         -- Create the PETSc matrix and factorize it
  601         mat <- petscMatrixCreate comm' n n MatAij matrixData nzPattern InsertValues
  602         ksp <- kspCreate comm'
  603         let (_, _, _, matMutable) = fromPetscMatrix mat
  604 
  605         kspSetOperators ksp matMutable matMutable
  606         kspSetFromOptions ksp
  607         pc <- kspGetPc ksp
  608         pcSetFromOptions pc
  609         kspSetInitialGuessNonzero ksp False
  610         kspSetUp ksp  -- This performs the actual factorization
  611 
  612         -- Cache the factorized solver by database name for per-database lookup
  613         modifyMVar_ cachedKspSolver $ \solvers -> return $ M.insert dbName (ksp, mat, n) solvers
  614 
  615         reportMatrixOperation $ "PETSc solver for database '" ++ T.unpack dbName ++ "' factorized and cached"
  616 
  617         -- Store the system matrix for fallback when cache miss or solver failure
  618         let factorization = MatrixFactorization
  619                 { mfSystemMatrix = U.fromList [SparseTriple (fromIntegral i) (fromIntegral j) v | (i, j, v) <- systemMatrix]
  620                 , mfActivityCount = fromIntegral n
  621                 , mfDatabaseId = dbName
  622                 }
  623 
  624         return factorization
