  223 -- | Parse a header line like "{key: value}" or "{value}"
  224 parseHeaderLine :: BS.ByteString -> Maybe (BS.ByteString, BS.ByteString)
  225 -- entered 418 timesparseHeaderLine line
  226     | BS8.isPrefixOf "{" line && BS8.isSuffixOf "}" line =
  227         let content = BS8.init (BS8.tail line)
  228         in case BS8.breakSubstring ": " content of
  229             (key, rest) | not (BS.null rest) -> Just (BS8.strip key, BS8.strip (BS.drop 2 rest))
  230             _ -> Just (BS8.strip content, "")
  231     | otherwise = Nothing

  233 -- | Update config from header line (takes ByteString, stores Text)
  234 updateConfigFromHeader :: SimaProConfig -> BS.ByteString -> BS.ByteString -> SimaProConfig
  235 -- entered 51 timesupdateConfigFromHeader cfg key value = case BS8.map toLower key of
  236     k | "simapro" `BS8.isPrefixOf` k -> cfg { spVersion = decodeBS key }
  237     "processes" -> cfg { spFileType = "processes" }
  238     "methods" -> cfg { spFileType = "methods" }
  239     "product stages" -> cfg { spFileType = "product stages" }
  240     "csv separator" -> cfg { spDelimiter = parseDelimiter value }
  241     "decimal separator" -> cfg { spDecimal = if BS.null value then ',' else BS8.head value }
  242     "short date format" -> cfg { spDateFormat = localDecodeBS value }
  243     _ -> cfg
  244   where
  245     localDecodeBS = TE.decodeUtf8With TEE.lenientDecode
  246     parseDelimiter v
  247         | v == "Semicolon" = ';'
  248         | v == "Comma"     = ','
  249         | v == "Tab"       = '\t'
  250         | otherwise        = ';'

  303 -- | Parse amount with configurable decimal separator (ByteString)
  304 parseAmount :: Char -> BS.ByteString -> Double
  305 -- entered 217 timesparseAmount decimalSep bs
  306     | BS.null bs = 0.0
  307     | otherwise =
  308         let normalized = if decimalSep == ','
  309                          then BS8.map (\c -> if c == ',' then '.' else c) bs
  310                          else bs
  311         in case reads (BS8.unpack normalized) of
  312             [(val, "")] -> val
  313             [(val, _)] -> val  -- Allow trailing characters
  314             _ -> 0.0

  316 -- | Split a CSV line by delimiter, respecting RFC 4180 quoted fields.
  317 splitCSV :: Char -> BS.ByteString -> [BS.ByteString]
  318 -- entered 325 timessplitCSV delim bs =
  319     let opts = Csv.defaultDecodeOptions
  320             { Csv.decDelimiter = fromIntegral (fromEnum delim) }
  321     in case Csv.decodeWith opts Csv.NoHeader (BL.fromStrict bs) of
  322         Right rows | not (V.null rows) -> V.toList (V.head rows)
  323         _ -> BS8.split delim bs  -- fallback to naive on parse error

  409 -- | Process a single line (ByteString)
  410 processLine :: ParseAcc -> BS.ByteString -> ParseAcc
  411 -- entered 615 timesprocessLine acc@ParseAcc{..} line
  412     -- Empty line handling
  413     | BS.null (BS8.strip line) = case paState of
  414         InProcessMeta _ -> acc { paState = BetweenBlocks }
  415         InSection _ -> acc { paState = BetweenBlocks }
  416         _ -> acc
  417 
  418     -- Header lines
  419     | Just (key, value) <- parseHeaderLine line, paState == InHeader =
  420         acc { paConfig = updateConfigFromHeader paConfig key value }
  421 
  422     -- Process block start
  423     | BS8.strip line == "Process" =
  424         acc { paState = BetweenBlocks
  425             , paCurrentBlock = emptyProcessBlock
  426             }
  427 
  428     -- End of block
  429     | BS8.strip line == "End" =
  430         let block = paCurrentBlock
  431             -- A block is valid if it has at least one product (process name not required)
  432             isValid = not (null (pbProducts block))
  433         in acc { paState = BetweenBlocks
  434                , paBlocks = if isValid then block : paBlocks else paBlocks
  435                , paCurrentBlock = emptyProcessBlock
  436                }
  437 
  438     -- Section detection
  439     | Just sec <- detectSection line =
  440         acc { paState = InSection sec }
  441 
  442     -- In a section, parse row (route db/project params to ParseAcc, process params to block)
  443     | InSection sec <- paState, not (BS.null (BS8.strip line)) =
  444         case sec of
  445             SecDbInputParams -> case parseParamRow paConfig line of
  446                 Just p  -> acc { paDbInputParams = p : paDbInputParams }
  447                 Nothing -> acc
  448             SecDbCalcParams -> case parseParamRow paConfig line of
  449                 Just p  -> acc { paDbCalcParams = p : paDbCalcParams }
  450                 Nothing -> acc
  451             SecProjInputParams -> case parseParamRow paConfig line of
  452                 Just p  -> acc { paProjInputParams = p : paProjInputParams }
  453                 Nothing -> acc
  454             SecProjCalcParams -> case parseParamRow paConfig line of
  455                 Just p  -> acc { paProjCalcParams = p : paProjCalcParams }
  456                 Nothing -> acc
  457             _ -> acc { paCurrentBlock = addRowToBlock paConfig sec line paCurrentBlock }
  458 
  459     -- Metadata key-value pairs
  460     | paState == BetweenBlocks || isMetadataKey (BS8.strip line) =
  461         if isMetadataKey (BS8.strip line)
  462         then acc { paState = InProcessMeta (BS8.strip line) }
  463         else case paState of
  464             InProcessMeta key -> acc { paCurrentBlock = setMetadata key line paCurrentBlock
  465                                      , paState = BetweenBlocks
  466                                      }
  467             _ -> acc
  468 
  469     -- Value for metadata key
  470     | InProcessMeta key <- paState =
  471         acc { paCurrentBlock = setMetadata key line paCurrentBlock
  472             , paState = BetweenBlocks
  473             }
  474 
  475     | otherwise = acc { paLineNum = paLineNum + 1 }

  561 -- | Extract location from SimaPro-style names
  562 -- Handles: "Name {FR}", "Name {Europe without Switzerland}| market for...", etc.
  563 -- Also handles WFLDB-style "/XX" suffix (e.g. "Ammonium nitrate .../CN").
  564 -- In SimaPro format, curly braces always denote geography.
  565 extractLocation :: Text -> (Text, Text)
  566 -- entered 113 timesextractLocation name =
  567     case T.breakOn "{" name of
  568         (_, rest) | not (T.null rest) ->
  569             case T.breakOn "}" (T.drop 1 rest) of
  570                 (loc, afterBrace) | not (T.null afterBrace) ->
  571                     let cleanLoc = T.strip loc
  572                     in if T.length cleanLoc >= 2
  573                        then (T.strip name, cleanLoc)
  574                        else (name, "")
  575                 _ -> (name, "")
  576         _ -> -- No {XX} found — try WFLDB-style /XX suffix
  577             case extractSlashLocation name of
  578                 Just (cleanName, loc) -> (cleanName, loc)
  579                 Nothing               -> (name, "")
  580   where
  581     -- Extract location from WFLDB-style slash suffixes like:
  582     --   "Product (WFLDB)/CN U"       → ("Product (WFLDB)", "CN")
  583     --   "Product/ha/GLO/I U"         → ("Product/ha", "GLO")
  584     -- Scans rightward through slash-separated segments for the first geo code.
  585     extractSlashLocation n = go n
  586       where
  587         go t = case T.breakOnEnd "/" t of
  588             ("", _) -> Nothing
  589             (before, suffix) -> case T.words (T.strip suffix) of
  590                 (loc:_) | isGeoCode loc ->
  591                     Just (T.strip (T.dropWhileEnd (== '/') before), loc)
  592                 _ -> go (T.dropWhileEnd (== '/') before)
  593     isGeoCode t = T.length t >= 2 && isUpper (T.head t)

  603 -- | Convert ProcessBlock to list of Activities (one per product)
  604 -- This matches EcoSpold behavior where multi-product processes create multiple activities
  605 -- Global params (db + project level) are passed in and combined with process-level params.
  606 processBlockToActivity :: ([(Text, Text)], [(Text, Text)], [(Text, Text)], [(Text, Text)])
  607                        -> ProcessBlock -> [(Activity, [Flow], [Unit])]
  608 -- entered 25 timesprocessBlockToActivity (dbInputPs, dbCalcPs, projInputPs, projCalcPs) ProcessBlock{..} =
  609     let -- Build parameter environment: input params first, then calculated params
  610         evalParam acc (name, rawVal) = case Expr.evaluate acc rawVal of
  611             Right v -> M.insert name v acc
  612             Left _  -> acc

  614         env1 = foldl' evalParam env0 (reverse projInputPs)
  615         env2 = foldl' evalParam env1 (reverse pbInputParams)
  616         env3 = foldl' evalParam env2 (reverse dbCalcPs)
  617         env4 = foldl' evalParam env3 (reverse projCalcPs)
  618         env  = foldl' evalParam env4 (reverse pbCalcParams)

  620         -- Raw expression map for re-evaluation

  622                 ++ reverse dbCalcPs ++ reverse projCalcPs ++ reverse pbCalcParams
  623         exprMap = M.fromList allExprs
  624 
  625         -- Extract location from process name if not specified
  626         (_, locFromName) = extractLocation pbName
  627         location = if T.null pbLocation || T.toLower pbLocation == "unspecified" then locFromName else pbLocation
  628 
  629         -- Convert avoided products (shared across all activities)
  630         avoidedExchanges = map (productToExchange env False) pbAvoidedProducts
  631 
  632         -- Convert technosphere inputs (shared across all activities)
  633         techExchanges = concatMap (techRowToExchange env True) (pbMaterials ++ pbElectricity)
  634         wasteExchanges = concatMap (techRowToExchange env True) pbWasteToTreatment
  635 
  636         -- Convert biosphere exchanges (shared across all activities)
  637         resourceExchanges = map (bioRowToExchange env True "resource") pbResources
  638         airExchanges = map (bioRowToExchange env False "air") pbEmissionsAir
  639         waterExchanges = map (bioRowToExchange env False "water") pbEmissionsWater
  640         soilExchanges = map (bioRowToExchange env False "soil") pbEmissionsSoil
  641         wasteFlowExchanges = map (bioRowToExchange env False "waste") pbFinalWaste
  642 
  643         sharedExchanges = avoidedExchanges ++
  644                           techExchanges ++ wasteExchanges ++
  645                           resourceExchanges ++ airExchanges ++ waterExchanges ++
  646                           soilExchanges ++ wasteFlowExchanges
  647 
  648         -- Collect all flows and units (shared)
  649         allFlows = collectFlows (pbProducts ++ pbAvoidedProducts)
  650                                 (pbMaterials ++ pbElectricity ++ pbWasteToTreatment)
  651                                 (pbResources ++ pbEmissionsAir ++ pbEmissionsWater ++
  652                                  pbEmissionsSoil ++ pbFinalWaste)
  653         allUnits = collectUnits (pbProducts ++ pbAvoidedProducts)
  654                                 (pbMaterials ++ pbElectricity ++ pbWasteToTreatment)
  655                                 (pbResources ++ pbEmissionsAir ++ pbEmissionsWater ++
  656                                  pbEmissionsSoil ++ pbFinalWaste)
  657 
  658         -- Create one activity per product
  659         makeActivity :: ProductRow -> (Activity, [Flow], [Unit])
  660         makeActivity prod =
  661             let productExchange = productToExchange env True prod
  662                 (cleanProductName, locFromProduct) = extractLocation (prName prod)
  663                 effectiveLoc = if T.null location then locFromProduct else location
  664                 activity = Activity
  665                     { activityName = cleanProductName
  666                     , activityDescription = if T.null pbComment then [] else [pbComment]
  667                     , activitySynonyms = M.empty
  668                     , activityClassification = M.fromList $ filter (not . T.null . snd)
  669                         [ ("Category type", pbCategoryType)
  670                         , ("Category", prCategory prod)
  671                         ]
  672                     , activityLocation = effectiveLoc
  673                     , activityUnit = prUnit prod
  674                     , exchanges = productExchange : sharedExchanges
  675                     , activityParams = env
  676                     , activityParamExprs = exprMap
  677                     }
  678             in (activity, allFlows, allUnits)
  679 
  680     in map makeActivity pbProducts

  799 -- | Ensure ByteString is proper UTF-8, converting from Windows-1252 if needed.
  800 -- SimaPro CSV files use Windows-1252 encoding (no encoding header in the format).
  801 -- Handles three cases:
  802 --   1. Already valid UTF-8 with no C1 controls -> pass through
  803 --   2. Valid UTF-8 but contains C1 controls (bad prior conversion) -> fix C1 chars
  804 --   3. Raw Windows-1252 bytes (not valid UTF-8) -> full Win-1252 decode
  805 ensureUtf8 :: BS.ByteString -> BS.ByteString
  806 -- entered 27 timesensureUtf8 bs = case TE.decodeUtf8' bs of
  807     Right text
  808         | T.any isC1Control text -> TE.encodeUtf8 (fixWindows1252Controls text)
  809         | otherwise -> bs  -- Already proper UTF-8, no changes needed
  810     Left _ -> TE.encodeUtf8 (decodeWindows1252 bs)
  811   where
  812     isC1Control c = c >= '\x0080' && c <= '\x009F'
  813 
  814 -- | Decode raw Windows-1252 bytes to Text.
  815 -- decodeLatin1 maps each byte to the same-valued Unicode codepoint,
  816 -- then fixWindows1252Controls corrects the 0x80-0x9F range.
  817 decodeWindows1252 :: BS.ByteString -> Text
  818 -- never entereddecodeWindows1252 = fixWindows1252Controls . TE.decodeLatin1
  819

  859 -- | Parse a SimaPro CSV file
  860 -- Handles Windows-1252/Latin-1 encoding common in SimaPro exports
  861 parseSimaProCSV :: FilePath -> IO ([Activity], FlowDB, UnitDB)
  862 -- entered 17 timesparseSimaProCSV path = do
  863     hPutStrLn stderr $ "Loading SimaPro CSV file: " ++ path
  864     startTime <- getCurrentTime
  865 
  866     -- Read as ByteString and convert from Windows-1252 to proper UTF-8.
  867     -- SimaPro CSV uses Windows-1252 encoding by default.
  868     rawContent <- BS.readFile path
  869     let !utf8Content = ensureUtf8 rawContent
  870     let bsLines = BS8.lines utf8Content
  871         -- Strip Windows \r from ByteStrings (fast, no allocation)
  872         lines' = map stripCR bsLines
  873         initialAcc = ParseAcc
  874             { paConfig = defaultConfig
  875             , paState = InHeader
  876             , paCurrentBlock = emptyProcessBlock
  877             , paBlocks = []
  878             , paLineNum = 0
  879             , paDbInputParams = []
  880             , paDbCalcParams = []
  881             , paProjInputParams = []
  882             , paProjCalcParams = []
  883             }
  884     -- Use strict foldl' to process lines (ByteString directly)
  885     finalAcc <- evaluate $ foldl' processLine initialAcc lines'
  886     let blocks = reverse (paBlocks finalAcc)
  887 
  888     -- Convert all blocks to activities (one activity per product) - PARALLEL
  889     let globalParams = ( paDbInputParams finalAcc, paDbCalcParams finalAcc
  890                        , paProjInputParams finalAcc, paProjCalcParams finalAcc )
  891     converted <- concat <$> mapConcurrently (evaluate . force . processBlockToActivity globalParams) blocks
  892     let activities = map (\(a,_,_) -> a) converted
  893         allFlows = concatMap (\(_,f,_) -> f) converted
  894         allUnits = concatMap (\(_,_,u) -> u) converted
  895 
  896     -- Build deduplicated maps
  897     let flowDB = M.fromList [(flowId f, f) | f <- allFlows]
  898         unitDB = M.fromList [(unitId u, u) | u <- allUnits]
  899 
  900     -- Force evaluation before returning
  901     let !numActivities = length activities
  902     let !numFlows = M.size flowDB
  903     let !numUnits = M.size unitDB
  904 
  905     endTime <- getCurrentTime
  906     let duration = realToFrac (diffUTCTime endTime startTime) :: Double
  907     hPutStrLn stderr $ printf "SimaPro parsing completed in %.2fs:" duration
  908     hPutStrLn stderr $ printf "  Activities: %d processes" numActivities
  909     hPutStrLn stderr $ printf "  Flows: %d unique" numFlows
  910     hPutStrLn stderr $ printf "  Units: %d unique" numUnits
  911 
  912     return (activities, flowDB, unitDB)
  913   where
  914     -- Strip Windows \r from ByteString (fast, often no-op)
  915     stripCR :: BS.ByteString -> BS.ByteString
  916     stripCR bs
  917         | BS.null bs = bs
  918         | BS8.last bs == '\r' = BS8.init bs
  919         | otherwise = bs
