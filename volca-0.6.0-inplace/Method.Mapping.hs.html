   74 -- | Map all method flows to database flows using mapper handles.
   75 -- Mappers are tried in priority order (assumed pre-sorted).
   76 mapMethodFlows
   77     :: [MapperHandle]
   78     -> MapContext
   79     -> Method
   80     -> IO [(MethodCF, Maybe (Flow, MatchStrategy))]
   81 -- never enteredmapMethodFlows mappers ctx method =
   82     mapM (\cf -> fmap (\r -> (cf, r)) (mapSingleFlow mappers ctx cf)) (methodFactors method)
   83 
   84 -- | Map a single CF using the mapper handle cascade.
   85 -- Each mapper is tried in order; the first match wins.
   86 mapSingleFlow
   87     :: [MapperHandle]
   88     -> MapContext
   89     -> MethodCF
   90     -> IO (Maybe (Flow, MatchStrategy))
   91 -- never enteredmapSingleFlow mappers ctx cf = go mappers
   92   where
   93     go [] = pure Nothing
   94     go (m:ms) = do
   95         result <- mhMatch m ctx (MatchCF cf)
   96         case result of
   97             Just mr | Just flow <- M.lookup (mrTargetId mr) (mcFlowsByUUID ctx) ->
   98                 pure $ Just (flow, strategyFromText (mrStrategy mr))
   99             _ -> go ms

  101 -- | Convenience wrapper: map method CFs using the given mappers + DB.
  102 mapMethodToFlows :: [MapperHandle] -> Database -> Method -> IO [(MethodCF, Maybe (Flow, MatchStrategy))]
  103 -- never enteredmapMethodToFlows mappers db = mapMethodFlows mappers (buildMapContext db)

  105 -- | Convert strategy text back to MatchStrategy
  106 strategyFromText :: Text -> MatchStrategy
  107 -- never enteredstrategyFromText t = case T.toLower t of
  108     "uuid"    -> ByUUID
  109     "cas"     -> ByCAS
  110     "name"    -> ByName
  111     "synonym" -> BySynonym
  112     "fuzzy"   -> ByFuzzy
  113     _         -> ByFuzzy  -- Unknown strategies map to fuzzy

  132 -- | Find flow by normalized name with compartment preference
  133 findFlowByNameComp :: M.Map Text [Flow] -> Text -> Maybe Compartment -> Maybe Flow
  134 -- entered oncefindFlowByNameComp flowsByName name mComp =
  135     case M.lookup (normalizeName name) flowsByName of
  136         Just flows@(_:_) -> Just (pickByCompartment flows mComp)

  143 -- | Find flow via synonym group with compartment preference
  144 findFlowBySynonymComp :: SynonymDB -> M.Map Text [Flow] -> Text -> Maybe Compartment -> Maybe Flow
  145 -- entered oncefindFlowBySynonymComp synDB flowsByName name mComp =
  146     case lookupSynonymGroup synDB name of
  147         Nothing -> Nothing
  148         Just groupId ->
  149             case getSynonyms synDB groupId of
  150                 Nothing -> Nothing
  151                 Just synonyms ->
  152                     let candidates = concatMap (lookupFlows flowsByName) synonyms
  153                     in if null candidates then Nothing
  154                        else Just (pickByCompartment candidates mComp)
  155   where
  156     lookupFlows fbn syn = M.findWithDefault [] (normalizeName syn) fbn
  157 
  158 -- | Pick the best flow match based on compartment preference.
  159 pickByCompartment :: [Flow] -> Maybe Compartment -> Flow
  160 -- never enteredpickByCompartment flows Nothing = head flows
  161 pickByCompartment flows (Just comp) =
  162     case find (exactCompMatch comp) flows of
  163         Just f  -> f
  164         Nothing -> case find (mediumMatch comp) flows of
  165             Just f  -> f
  166             Nothing -> head flows
  167   where
  168     exactCompMatch (Compartment med sub _) f =
  169         let cat = T.toLower (flowCategory f)
  170             subcomp = maybe "" T.toLower (flowSubcompartment f)
  171         in matchMedium med cat && (T.null sub || sub == subcomp || sub `T.isInfixOf` subcomp)
  172 
  173     mediumMatch (Compartment med _ _) f =
  174         matchMedium med (T.toLower (flowCategory f))
  175 
  176     matchMedium med cat
  177         | T.null med  = True
  178         | med == cat  = True
  179         | med `T.isInfixOf` cat = True
  180         | otherwise   = False

  182 -- | Compute statistics about mapping results
  183 computeMappingStats :: [(MethodCF, Maybe (Flow, MatchStrategy))] -> MappingStats
  184 -- never enteredcomputeMappingStats mappings = MappingStats
  185     { msTotal = length mappings
  186     , msByUUID = count ByUUID
  187     , msByCAS = count ByCAS
  188     , msByName = count ByName
  189     , msBySynonym = count BySynonym
  190     , msByFuzzy = count ByFuzzy
  191     , msUnmatched = length $ filter (isNothing . snd) mappings
  192     }
  193   where
  194     count strategy = length $ filter ((== Just strategy) . fmap snd . snd) mappings
