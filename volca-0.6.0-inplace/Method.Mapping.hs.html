   86 -- | Map a single CF using the mapper handle cascade.
   87 -- Each mapper is tried in order; the first match wins.
   88 mapSingleFlow
   89     :: [MapperHandle]
   90     -> MapContext
   91     -> MethodCF
   92     -> IO (Maybe (Flow, MatchStrategy))
   93 -- never enteredmapSingleFlow mappers ctx cf = go mappers

   95     go [] = pure Nothing
   96     go (m:ms) = do
   97         result <- mhMatch m ctx (MatchCF cf)

  153 -- | Pick the best flow match based on compartment preference.
  154 -- Returns Nothing for an empty candidate list.
  155 pickByCompartment :: [Flow] -> Maybe Compartment -> Maybe Flow
  156 -- entered 8 timespickByCompartment []    _          = Nothing
  157 pickByCompartment (f:_) Nothing    = Just f
  158 pickByCompartment (f:fs) (Just comp) = Just $
  159     case find (exactCompMatch comp) (f:fs) of
  160         Just m  -> m
  161         Nothing -> fromMaybe f (find (mediumMatch comp) (f:fs))
  162   where
  163     exactCompMatch (Compartment med sub _) fl =
  164         let cat    = T.toLower (flowCategory fl)
  165             subcomp = maybe "" T.toLower (flowSubcompartment fl)
  166         in matchMedium med cat && (T.null sub || sub == subcomp || sub `T.isInfixOf` subcomp)
  167 
  168     mediumMatch (Compartment med _ _) fl =
  169         matchMedium med (T.toLower (flowCategory fl))
  170 
  171     matchMedium med cat
  172         | T.null med        = True
  173         | med == cat        = True
  174         | med `T.isInfixOf` cat = True
  175         | otherwise         = False

  191 -- | Compute LCIA score from inventory and flow mappings.
  192 -- Scores each inventory flow by finding its best matching CF:
  193 --   1. UUID-matched flows use the exact CF from the mapping.
  194 --   2. For other flows, tries exact (name, medium, subcompartment) match.
  195 --   3. Falls back to (name, medium) for CF entries with unspecified subcompartment.
  196 -- This implements SimaPro's fallback behavior: "Air;(unspecified)" CFs apply to
  197 -- any air emission that lacks a more specific CF, without aggregating across media.
  198 -- Unit conversion is applied when the inventory flow's unit differs from the CF's unit.
  199 computeLCIAScore :: UnitConfig -> UnitDB -> FlowDB -> Inventory -> [(MethodCF, Maybe (Flow, MatchStrategy))] -> Double
  200 -- entered 8 timescomputeLCIAScore unitConfig unitDB flowDB inventory mappings =
  201     sum [ scoreFlow fid qty | (fid, qty) <- M.toList inventory, qty /= 0 ]
  202   where
  203     scoreFlow fid qty = case lookupCF fid of
  204         Nothing -> 0
  205         Just (cfVal, cfUnit) ->
  206             let flowUnit = maybe "" unitName (M.lookup fid flowDB >>= \f -> M.lookup (flowUnitId f) unitDB)
  207                 converted = if flowUnit == cfUnit || T.null cfUnit then qty
  208                             else fromMaybe qty (convertUnit unitConfig flowUnit cfUnit qty)
  209             in converted * cfVal
  210 
  211     lookupCF fid = case M.lookup fid uuidCF of
  212         Just cfv -> Just cfv
  213         Nothing  -> case M.lookup fid flowDB of
  214             Nothing   -> Nothing
  215             Just flow ->
  216                 let name    = normalizeName (flowName flow)
  217                     baseMed = normalizeMedium . T.takeWhile (/= '/') . T.toLower $ flowCategory flow
  218                     subcomp = T.toLower $ fromMaybe "" (flowSubcompartment flow)
  219                     strip3 (v, u, _) = (v, u)
  220                     exact   = strip3 <$> M.lookup (name, baseMed, subcomp) exactCF
  221                 in case exact of
  222                     Just _  -> exact
  223                     Nothing -> strip3 <$> M.lookup (name, baseMed) fallbackCF

  225     -- UUID-matched CFs: exact flow_id → (CF_value, CF_unit)

  227         [ (flowId flow, (mcfValue cf, mcfUnit cf))
  228         | (cf, Just (flow, ByUUID)) <- mappings ]

  230     -- Prefer higher-priority match strategies; within same strategy, take max CF
  231     preferBetter (v1, u1, s1) (v2, u2, s2)
  232         | stratPriority s1 < stratPriority s2 = (v1, u1, s1)
  233         | stratPriority s1 > stratPriority s2 = (v2, u2, s2)
  234         | v1 >= v2  = (v1, u1, s1)
  235         | otherwise = (v2, u2, s2)

  242     matchStrategy mflow = case mflow of

  244         Nothing     -> NoMatch

  246     -- Exact CFs for specific subcompartments: (name, medium, subcomp) → (CF, unit, strategy)

  251         , not (T.null subcomp) ]
