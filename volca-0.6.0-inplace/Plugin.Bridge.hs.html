    3 -- | External Plugin Bridge
    4 --
    5 -- Communicates with external plugin executables via JSON on stdin/stdout.
    6 -- Each request is a single JSON line; each response is a single JSON line.
    7 -- Plugins are spawned per-call (stateless). Future: persistent connections.
    8 module Plugin.Bridge
    9     ( callPlugin
   10     , callPluginBatch
   11     ) where
   12 
   13 import Data.Aeson (Value, encode, eitherDecode')
   14 import qualified Data.ByteString.Lazy as BSL
   15 import qualified Data.ByteString.Lazy.Char8 as BSL8
   16 import System.Exit (ExitCode(..))
   17 import System.Process (readProcessWithExitCode)
   18 import Data.Text (Text)
   19 import qualified Data.Text as T
   20 
   21 -- | Call an external plugin with a single JSON request.
   22 -- Spawns the plugin executable, sends JSON on stdin, reads JSON from stdout.
   23 callPlugin :: FilePath -> Value -> IO (Either Text Value)
   24 -- never enteredcallPlugin pluginPath request = do
   25     let input = BSL8.unpack (encode request)
   26     result <- readProcessWithExitCode pluginPath [] input
   27     case result of
   28         (ExitSuccess, stdout, _stderr) ->
   29             case eitherDecode' (BSL8.pack stdout) of
   30                 Right val -> pure (Right val)
   31                 Left err  -> pure (Left $ "Plugin JSON parse error: " <> T.pack err)
   32         (ExitFailure code, _stdout, stderr) ->
   33             pure $ Left $ "Plugin " <> T.pack pluginPath
   34                 <> " exited with code " <> T.pack (show code)
   35                 <> ": " <> T.pack stderr
   36 
   37 -- | Call an external plugin with a batch of JSON requests (JSON Lines).
   38 -- Sends all requests as newline-delimited JSON, reads newline-delimited responses.
   39 callPluginBatch :: FilePath -> [Value] -> IO (Either Text [Value])
   40 -- never enteredcallPluginBatch pluginPath requests = do
   41     let input = BSL8.unpack $ BSL.intercalate "\n" (map encode requests)
   42     result <- readProcessWithExitCode pluginPath [] input
   43     case result of
   44         (ExitSuccess, stdout, _stderr) ->
   45             let lines' = filter (not . BSL.null) $ BSL8.split '\n' (BSL8.pack stdout)
   46             in case parseLines lines' of
   47                 Right vals -> pure (Right vals)
   48                 Left err   -> pure (Left $ "Plugin JSON parse error: " <> T.pack err)
   49         (ExitFailure code, _stdout, stderr) ->
   50             pure $ Left $ "Plugin " <> T.pack pluginPath
   51                 <> " exited with code " <> T.pack (show code)
   52                 <> ": " <> T.pack stderr
   53 
   54 -- | traverse for lists using mapM from Prelude
   55 parseLines :: [BSL.ByteString] -> Either String [Value]
   56 -- never enteredparseLines = mapM eitherDecode'
