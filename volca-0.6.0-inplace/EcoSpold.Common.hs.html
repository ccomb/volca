   13 import qualified Data.ByteString as BS
   14 import Data.Text (Text)
   15 import qualified Data.Text as T
   16 import qualified Data.Text.Encoding as TE
   17 import qualified Data.Text.Read as TR

   19 -- | ByteString to Text conversion with UTF-8 decoding and XML entity decoding
   20 bsToText :: BS.ByteString -> Text
   21 bsToText = decodeXmlEntities . TE.decodeUtf8
   22 
   23 -- | Decode common XML entities that Xeno doesn't decode
   24 -- Xeno is a fast SAX parser but doesn't handle entity references
   25 decodeXmlEntities :: Text -> Text
   26 decodeXmlEntities = T.replace "&lt;" "<"
   27                   . T.replace "&gt;" ">"
   28                   . T.replace "&amp;" "&"
   29                   . T.replace "&quot;" "\""
   30                   . T.replace "&apos;" "'"
   31 
   32 -- | ByteString to Double conversion (strict - errors on parse failure)
   33 bsToDouble :: BS.ByteString -> Double
   34 bsToDouble bs = case TR.double (bsToText bs) of
   35     Right (val, _) -> val
   36     Left _ -> error $ "Failed to parse double from: " ++ show bs
   37 
   38 -- | ByteString to Int conversion (strict - errors on parse failure)
   39 bsToInt :: BS.ByteString -> Int
   40 bsToInt bs = case TR.decimal (bsToText bs) of
   41     Right (val, _) -> val
   42     Left _ -> error $ "Failed to parse int from: " ++ show bs
   43 
   44 -- | Check if element name matches (with or without namespace prefix)
   45 -- Handles both "tagName" and "prefix:tagName" forms
   46 isElement :: BS.ByteString -> BS.ByteString -> Bool
   47 isElement tagName expected =
   48     tagName == expected || BS.isSuffixOf (":" `BS.append` expected) tagName
   49 
   50 -- | Distribute a list evenly across N buckets (for parallel workers)
   51 distributeFiles :: Int -> [a] -> [[a]]
   52 distributeFiles n xs =
   53     let len      = length xs
   54         baseSize = len `div` n
   55         remainder = len `mod` n
   56         sizes    = replicate remainder (baseSize + 1) ++ replicate (n - remainder) baseSize
   57     in go sizes xs
   58   where
   59     go []     _  = []
   60     go _      [] = []
   61     go (s:ss) ys = let (h, t) = splitAt s ys in h : go ss t
