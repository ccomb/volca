   57 withKspSetup ::
   58   Comm ->
   59   KspType_ ->
   60   Mat -> -- linear operator
   61   Mat -> -- preconditioner
   62   Bool -> -- set initial solution guess to nonzero vector
   63   (KSP -> IO a) -> -- post-setup actions, i.e. solve with a r.h.s , etc.
   64   IO a
   65 -- never enteredwithKspSetup cc kt amat pmat ignz f = withKsp cc $ \ksp -> do
   66   kspSetOperators ksp amat pmat
   67   kspSetType ksp kt
   68   kspSetInitialGuessNonzero ksp ignz
   69   kspSetUp ksp
   70   f ksp

   96 -- | KSP setup that uses PETSC_OPTIONS for configuration (no hardcoded type)
   97 withKspSetupFromOptions ::
   98   Comm ->
   99   Mat -> -- linear operator
  100   Mat -> -- preconditioner
  101   Bool -> -- set initial solution guess to nonzero vector
  102   (KSP -> IO a) -> -- post-setup actions
  103   IO a
  104 -- never enteredwithKspSetupFromOptions cc amat pmat ignz f = withKsp cc $ \ksp -> do
  105   kspSetOperators ksp amat pmat
  106   kspSetFromOptions ksp  -- Read PETSC_OPTIONS instead of hardcoding type
  107   pc <- kspGetPC ksp
  108   pcSetFromOptions pc    -- Configure preconditioner from PETSC_OPTIONS (enables MUMPS options)
  109   kspSetInitialGuessNonzero ksp ignz
  110   kspSetUp ksp
  111   f ksp
