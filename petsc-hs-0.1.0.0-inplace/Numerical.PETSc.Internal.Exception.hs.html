  114 petscOutcome :: CInt -> Either PetscErrCode_ PetscSafeErrCode_
  115 -- entered 196 timespetscOutcome n | n==0 = Right NoError0
  116                | n==256 = Right NoError256
  117                | n==8288 = Right NoError8288  
  118                | n < maxErrCode && n >= minErrCode = Left (petscErrCodeFromInt n)
  119                | otherwise = error (unforeseenErrCodeStr n)
  120   where
  121     maxErrCode = 90
  122     minErrCode = 55
