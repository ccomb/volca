  137 joinVectors :: Storable t => [VS.Vector t] -> IO (VS.Vector t)
  138 -- never enteredjoinVectors [] = return $ VS.fromList []
  139 joinVectors [v] = return v
  140 joinVectors as = do
  141     let tot = sum (map vdim as)
  142     r <- createVector tot
  143     VS.unsafeWith r $ \ptr ->
  144         joiner as tot ptr
  145     return r
  146   where joiner [] _ _ = return ()
  147         joiner (v:cs) _ p = do
  148             let n = vdim v
  149             VS.unsafeWith v $ \pb -> copyArray p pb n
  150             joiner cs 0 (advancePtr p n)

  425 -- | Vector.mapM
  426 
  427 mapVectorM0 ::
  428   (Storable a, Storable b, Monad m) => (a -> m b) -> VS.Vector a -> m (VS.Vector b)
  429 -- never enteredmapVectorM0 f v = do
  430     w <- return $! unsafePerformIO $! createVector vd
  431     go w 0 (vd -1)
  432     return w
  433     where
  434       vd = vdim v
  435       go w' !k !t
  436         | k == t  = do
  437            x <- return $! unsafePerformIO $! VS.unsafeWith v (`peekElemOff` k) 
  438            y <- f x
  439            return $! unsafePerformIO $! VS.unsafeWith w' ( \q -> pokeElemOff q k y)
  440         | otherwise = do
  441            x <- return $! unsafePerformIO $! VS.unsafeWith v (`peekElemOff` k) 
  442            y <- f x
  443            _ <- return $! unsafePerformIO $! VS.unsafeWith w' ( \q -> pokeElemOff q k y )
  444            go w' (k+1) t
