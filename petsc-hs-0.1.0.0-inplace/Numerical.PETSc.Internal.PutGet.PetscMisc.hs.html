  186 petscOptionsSetValue :: OptionName -> String -> IO ()
  187 -- never enteredpetscOptionsSetValue opt val = chk0 $ petscOptionsSetValue0' (helper opt) val
  188  where
  189   helper o@(x : _)
  190     | x == '-' = o
  191     | otherwise = '-' : o
  192   helper [] = error "petscOptionsSetValue : no option name supplied"

  214 petscInit ::

  243 {- | git commit hash (for debugging)
  244 Evaluated at compile time via Template Haskell.
  245 Returns "(unknown)" if git is not available during compilation.
  246 -}
  247 gitHash :: String
  248 -- entered oncegitHash = $(do
  249     let trim = reverse . dropWhile (== '\n') . reverse
  250     result <- TH.runIO $ catch
  251         (trim <$> readProcess "git" ["rev-parse", "--verify", "HEAD"] [])
  252         (\(_ :: SomeException) -> return "(unknown)")

  268 -- | Version string
  269 {-# NOINLINE petscVersionString #-}
  270 petscVersionString :: String
  271 -- entered oncepetscVersionString = vstrRaw
  272  where
  273   vstrRaw = unsafePerformIO (petscGetVersion 200)
  274 
  275 petscGetVersion :: Int -> IO String
  276 -- entered oncepetscGetVersion l = do
  277   fp <- mallocPlainForeignPtrBytes l -- see Data.Bytestring.Internal.create
  278   withForeignPtr fp $ \p -> do
  279     pgv p (toCInt l)
  280     peekCString p
  281  where
  282   pgv v sz = chk0 (petscGetVersion0' v sz)
