  107 -- | filterMapV : if head of vector satisfies a predicate, apply a function to it
  108 
  109 filterMapV :: (a -> Bool) -> (a -> b) -> V.Vector a -> V.Vector b
  110 -- never enteredfilterMapV _ _ vv
  111   | V.null vv = V.empty
  112 filterMapV p f vv
  113   | p x = V.cons (f x) $ filterMapV p f (V.tail vv) 
  114   | otherwise = filterMapV p f (V.tail vv)
  115   where
  116       x = V.head vv
